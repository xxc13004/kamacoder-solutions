## 思路分析

直接枚举在第几位开始分割，然后将字符串拼接，按照字典序大小比较就行，找到一个最大字典序的字符串。

## 复杂度分析

1. 时间复杂度：O(n^2)；
2. 空间复杂度：O(n)。

### Python
``` python
import sys

def main() -> None:
    s = sys.stdin.readline().strip()
    n = len(s)
    ans = s
    
    # 找到最大的字符
    max_char = max(s)
    
    # 只考虑以最大字符开头的旋转
    for i in range(n):
        if s[i] == max_char:
            ns = s[i:] + s[:i]
            if ns > ans:
                ans = ns
    
    print(ans)

if __name__ == "__main__":
    main()
```
### Java
``` java
import java.io.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        PrintWriter pw = new PrintWriter(new BufferedWriter(new OutputStreamWriter(System.out)));
        
        String s = br.readLine().trim();
        int n = s.length();
        String ans = s;
        
        // 生成所有旋转字符串
        for (int i = 1; i <= n; i++) {
            // 旋转i个字符：s.substring(i) + s.substring(0, i)
            String ns = s.substring(i) + s.substring(0, i);
            if (ns.compareTo(ans) > 0) {
                ans = ns;
            }
        }
        
        pw.println(ans);
        pw.flush();
        pw.close();
        br.close();
    }
}
```

### Go
``` go
package main

import (
    "bufio"
    "fmt"
    "os"
)

func main() {
    reader := bufio.NewReader(os.Stdin)
    writer := bufio.NewWriter(os.Stdout)
    defer writer.Flush()
    
    var s string
    fmt.Fscan(reader, &s)
    
    n := len(s)
    // 构造循环字符串以便快速访问
    doubleS := s + s
    ans := s
    
    // 检查所有可能的旋转
    for i := 1; i < n; i++ {
        rotated := doubleS[i:i+n]
        if rotated > ans {
            ans = rotated
        }
    }
    
    fmt.Fprintln(writer, ans)
}
```
