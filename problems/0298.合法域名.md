# 模拟
[题目链接](https://kamacoder.com/problempage.php?pid=1378)
## 思路分析
根据题意进行模拟，按照点将字符串进行分割，此时得到所有的标签组成的数组，先判断数组的长度是不是小于2，以及字符串总长度小于等于255。然后判断每个标签的长度是不是1-63个字符之间。以及起始和结束是不是包含'-'。
## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(1)。

``` python
def main():
    s = input().strip()
    if len(s) > 255 or s[-1] == '.':
        print("false")
        return
    res = s.split('.')
    if len(res) == 1:
        print("false")
        return
    for x in res:
        if len(x) > 63:
            print("false")
            return
        if len(x) == 0 or not (x[0].isalnum() and x[-1].isalnum()):
            print("false")
            return
        for c in x:
            if not (c.isalnum() or c == '-'):
                print("false")
                return
    print("true")
```

### Java
``` java
import java.io.*;
import java.util.*;
public class Main {
    public static boolean isValid(char c) {
		return Character.isDigit(c) || (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z');
	}
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		String s = br.readLine();
		if (s.length() > 255 || s.charAt(s.length() - 1) == '.') {
			System.out.println("false");
			return;
		}
		String[] res = s.split("\\.");
		if (res.length == 1) {
			System.out.println("false");
			return;
		}
		for (String x : res) {
			if (x.length() == 0 || x.length() > 63) {
				System.out.println("false");
				return;
			}
			if (!isValid(x.charAt(0)) || !isValid(x.charAt(x.length() - 1))) {
				System.out.println("false");
				return;
			}
		}
		System.out.println("true");
	}
}
```

### Go
``` go
package main
import (
	"bufio"
	"fmt"
	"os"
)

func isValid(c byte) bool {
	return (c >= '0' && c <= '9') || (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z')
}

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Scan()
	s := scanner.Text()
	if len(s) > 255 || s[len(s)-1] == '.' {
		fmt.Println("false")
		return
	}
	temp := ""
	res := []string{}
	for i := 0; i < len(s); i++ {
		if s[i] == '.' {
			res = append(res, temp)
			temp = ""
		} else {
			temp += string(s[i])
		}
	}
	res = append(res, temp)
	if len(res) == 1 {
		fmt.Println("false")
		return
	}
	for _, x := range res {
		if len(x) > 63 {
			fmt.Println("false")
			return
		}
		if len(x) == 0 || !isValid(x[0]) || !isValid(x[len(x)-1]) {
			fmt.Println("false")
			return
		}
	}
	fmt.Println("true")
}
```
