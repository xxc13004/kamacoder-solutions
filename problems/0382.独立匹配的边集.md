# 模拟
[题目链接](https://kamacoder.com/problempage.php?pid=1462)
## 思路分析
本题要求判断给定的边集是否是“独立匹配”。根据题意，独立匹配 X 需要满足两个条件：
1. **匹配条件**：X 中任意两条边不共享端点。这意味着在边集 X 中，每个顶点最多只能出现一次。
2. **独立条件**：任何一条不属于 X 的边，至多与 X 中的边共享一个顶点。反过来说，如果图中有一条边不属于 X，但它的两个端点都被 X 中的边覆盖了，那么这就违反了独立条件。

基于此，我们可以对每一次查询的边集进行如下判定：
1. 建立一个布尔数组 `in_V_X` 来记录**顶点**是否被边集 X 覆盖，建立一个布尔数组 `in_X` 记录边的**编号**是否在边集 X 中。
2. 遍历当前查询的边集 X 中的每一条边，获取其两个端点 u 和 v。如果 u 或 v 已经被标记过，说明共享了端点，违反了条件1，直接判定为 `No`。否则将 `in_V_X[u]` 和 `in_V_X[v]` 标记为 true，并将 `in_X[当前边编号]` 标记为 true。
3. 如果满足了条件1，我们再遍历图中所有的 m 条边（遍历编号 1 到 m）。对于某条边 i，它的端点是 u 和 v。如果 `in_V_X[u]` 和 `in_V_X[v]` 同时为 true（即这两个端点都被边集 X 覆盖了），我们就必须检查边 i 是否属于边集 X（即检查 `in_X[i]`）。如果 `in_X[i]` 为 false，说明有一条非 X 中的边连接了两个 X 中的顶点，违反了条件2，判定为 `No`。（反之，如果in_X[i]为true，则表示是我们查询时打的标记，注意我们第二条中所说的in_X标记为true的情况，是符合题意的。）
4. 如果全部检查都通过，则判定为 `Yes`。

## 复杂度分析
1. 时间复杂度：O(q * m)；
2. 空间复杂度：O(n + m)。

## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<int> u(m + 1);
    vector<int> v(m + 1);

    for (int i = 1; i <= m; ++i) cin >> u[i];
    for (int i = 1; i <= m; ++i) cin >> v[i];

    int q;
    cin >> q;

    while (q--) {
        int c;
        cin >> c;
        vector<int> query_edges(c);
        for (int i = 0; i < c; ++i) {
            cin >> query_edges[i];
        }

        vector<bool> in_V_X(n + 1, false);
        vector<bool> in_X(m + 1, false);
        bool isValid = true;

        // 检查条件1：是否是匹配（不共享端点）
        for (int i = 0; i < c; ++i) {
            int edge_idx = query_edges[i];
            int node_u = u[edge_idx];
            int node_v = v[edge_idx];

            if (in_V_X[node_u] || in_V_X[node_v]) {
                isValid = false;
                break;
            }
            in_V_X[node_u] = true;
            in_V_X[node_v] = true;
            in_X[edge_idx] = true;
        }

        // 检查条件2：不属于X的边至多与X共享一个顶点
        if (isValid) {
            for (int i = 1; i <= m; ++i) {
                int node_u = u[i];
                int node_v = v[i];
                if (in_V_X[node_u] && in_V_X[node_v]) {
                    if (!in_X[i]) {
                        isValid = false;
                        break;
                    }
                }
            }
        }

        if (isValid) {
            cout << "Yes\n";
        } else {
            cout << "No\n";
        }
    }

    return 0;
}
```

### Python
``` python
import sys

def main():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    n = int(input_data[0])
    m = int(input_data[1])
    
    u = [0] + [int(x) for x in input_data[2 : 2 + m]]
    v = [0] + [int(x) for x in input_data[2 + m : 2 + 2 * m]]
    
    q = int(input_data[2 + 2 * m])
    idx = 3 + 2 * m
    
    out = []
    for _ in range(q):
        c = int(input_data[idx])
        query_edges = [int(x) for x in input_data[idx + 1 : idx + 1 + c]]
        idx += c + 1
        
        in_V_X = [False] * (n + 1)
        in_X = [False] * (m + 1)
        is_valid = True
        
        for edge_idx in query_edges:
            node_u = u[edge_idx]
            node_v = v[edge_idx]
            
            if in_V_X[node_u] or in_V_X[node_v]:
                is_valid = False
                break
            
            in_V_X[node_u] = True
            in_V_X[node_v] = True
            in_X[edge_idx] = True
            
        if is_valid:
            for i in range(1, m + 1):
                if in_V_X[u[i]] and in_V_X[v[i]]:
                    if not in_X[i]:
                        is_valid = False
                        break
                        
        if is_valid:
            out.append("Yes")
        else:
            out.append("No")
            
    print("\n".join(out))

if __name__ == '__main__':
    main()
```

``` java
import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());
        if (!st.hasMoreTokens()) return;
        
        int n = Integer.parseInt(st.nextToken());
        int m = Integer.parseInt(st.nextToken());
        
        int[] u = new int[m + 1];
        int[] v = new int[m + 1];
        
        readArray(br, st, u, m);
        readArray(br, st, v, m);
        
        st = new StringTokenizer(br.readLine());
        int q = Integer.parseInt(st.nextToken());
        
        StringBuilder sb = new StringBuilder();
        
        for (int i = 0; i < q; i++) {
            if (!st.hasMoreTokens()) {
                st = new StringTokenizer(br.readLine());
            }
            int c = Integer.parseInt(st.nextToken());
            int[] query_edges = new int[c];
            for (int j = 0; j < c; j++) {
                if (!st.hasMoreTokens()) {
                    st = new StringTokenizer(br.readLine());
                }
                query_edges[j] = Integer.parseInt(st.nextToken());
            }
            
            boolean[] in_V_X = new boolean[n + 1];
            boolean[] in_X = new boolean[m + 1];
            boolean isValid = true;
            
            for (int j = 0; j < c; j++) {
                int edgeIdx = query_edges[j];
                int nodeU = u[edgeIdx];
                int nodeV = v[edgeIdx];
                
                if (in_V_X[nodeU] || in_V_X[nodeV]) {
                    isValid = false;
                    break;
                }
                in_V_X[nodeU] = true;
                in_V_X[nodeV] = true;
                in_X[edgeIdx] = true;
            }
            
            if (isValid) {
                for (int j = 1; j <= m; j++) {
                    if (in_V_X[u[j]] && in_V_X[v[j]]) {
                        if (!in_X[j]) {
                            isValid = false;
                            break;
                        }
                    }
                }
            }
            
            if (isValid) {
                sb.append("Yes\n");
            } else {
                sb.append("No\n");
            }
        }
        System.out.print(sb.toString());
    }
    
    private static void readArray(BufferedReader br, StringTokenizer st, int[] arr, int len) throws IOException {
        int idx = 1;
        while (idx <= len) {
            if (!st.hasMoreTokens()) {
                String line = br.readLine();
                if (line == null) break;
                st = new StringTokenizer(line);
            }
            while (st.hasMoreTokens() && idx <= len) {
                arr[idx++] = Integer.parseInt(st.nextToken());
            }
        }
    }
}
```

``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
)

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)
	
	buf := make([]byte, 1024*1024)
	scanner.Buffer(buf, 1024*1024*10)

	if !scanner.Scan() {
		return
	}
	n, _ := strconv.Atoi(scanner.Text())
	scanner.Scan()
	m, _ := strconv.Atoi(scanner.Text())

	u := make([]int, m+1)
	v := make([]int, m+1)

	for i := 1; i <= m; i++ {
		scanner.Scan()
		u[i], _ = strconv.Atoi(scanner.Text())
	}
	for i := 1; i <= m; i++ {
		scanner.Scan()
		v[i], _ = strconv.Atoi(scanner.Text())
	}

	scanner.Scan()
	q, _ := strconv.Atoi(scanner.Text())

	for i := 0; i < q; i++ {
		scanner.Scan()
		c, _ := strconv.Atoi(scanner.Text())
		
		query_edges := make([]int, c)
		for j := 0; j < c; j++ {
			scanner.Scan()
			query_edges[j], _ = strconv.Atoi(scanner.Text())
		}

		in_V_X := make([]bool, n+1)
		in_X := make([]bool, m+1)
		isValid := true

		for j := 0; j < c; j++ {
			edgeIdx := query_edges[j]
			nodeU := u[edgeIdx]
			nodeV := v[edgeIdx]

			if in_V_X[nodeU] || in_V_X[nodeV] {
				isValid = false
				break
			}
			in_V_X[nodeU] = true
			in_V_X[nodeV] = true
			in_X[edgeIdx] = true
		}

		if isValid {
			for j := 1; j <= m; j++ {
				if in_V_X[u[j]] && in_V_X[v[j]] {
					if !in_X[j] {
						isValid = false
						break
					}
				}
			}
		}

		if isValid {
			fmt.Println("Yes")
		} else {
			fmt.Println("No")
		}
	}
}
```
