# 动态规划 / 前后缀分解
[题目链接](https://kamacoder.com/problempage.php?pid=1471)
## 思路分析
题目要求将给定的序列通过做加法（且只能加，每次+1，次数不限）变成一个严格的单峰序列，并使得总的加法操作次数最小。单峰序列的特点是存在一个峰顶位置，左侧严格单调递增，右侧严格单调递减。

为了使总操作次数（即增加量之和）最小，我们需要使得变化后的每个元素尽可能地小。
假设序列下标从 `0` 到 `n-1`，我们可以在 `O(N)` 时间内预处理出两个方向的数组：
1. **前缀 DP（变为递增）**：
定义 `L_val[i]` 为使前缀 `a[0...i]` 变为严格递增序列时，第 `i` 个位置的最小合法值；
定义 `L_cost[i]` 为相应的最小代价之和。
状态转移：
`L_val[i] = max(a[i], L_val[i-1] + 1)`
`L_cost[i] = L_cost[i-1] + (L_val[i] - a[i])`

2. **后缀 DP（变为递减）**：
定义 `R_val[i]` 为使后缀 `a[i...n-1]` 变为严格递减序列时（从后往前看是严格递增），第 `i` 个位置的最小合法值；
定义 `R_cost[i]` 为相应的最小代价之和。
状态转移：
`R_val[i] = max(a[i], R_val[i+1] + 1)`
`R_cost[i] = R_cost[i+1] + (R_val[i] - a[i])`

3. **寻找最优峰顶**：
枚举每一个可能的峰顶位置 `k`（`1 <= k <= n-2`）。为了满足单峰性质，位置 `k` 的值 `peak` 必须同时严格大于其左边修改后的值 `L_val[k-1]` 和右边修改后的值 `R_val[k+1]`，并且不能低于原本的值 `a[k]`。
即：`peak = max(a[k], max(L_val[k-1] + 1, R_val[k+1] + 1))`。
以 `k` 为峰顶的全局总代价为：`Total_cost = L_cost[k-1] + R_cost[k+1] + (peak - a[k])`。
遍历所有的 `k`，找出产生最小 `Total_cost` 的组合即可。

*注意：操作增加的总量极大，可能会超出32位整数范围（最大可达 `10^10` 级别），因此变量需使用64位整数。*

## 复杂度分析
1. 时间复杂度：O(N)；
2. 空间复杂度：O(N)。

## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    // 优化输入输出
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;
    if (!(cin >> n)) return 0;
    
    vector<long long> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    vector<long long> L_val(n), L_cost(n);
    L_val[0] = a[0];
    L_cost[0] = 0;
    for (int i = 1; i < n; i++) {
        L_val[i] = max(a[i], L_val[i-1] + 1);
        L_cost[i] = L_cost[i-1] + (L_val[i] - a[i]);
    }
    
    vector<long long> R_val(n), R_cost(n);
    R_val[n-1] = a[n-1];
    R_cost[n-1] = 0;
    for (int i = n - 2; i >= 0; i--) {
        R_val[i] = max(a[i], R_val[i+1] + 1);
        R_cost[i] = R_cost[i+1] + (R_val[i] - a[i]);
    }
    
    long long min_total = -1;
    for (int k = 1; k < n - 1; k++) {
        long long peak = max({a[k], L_val[k-1] + 1, R_val[k+1] + 1});
        long long total = L_cost[k-1] + R_cost[k+1] + (peak - a[k]);
        if (min_total == -1 || total < min_total) {
            min_total = total;
        }
    }
    
    cout << min_total << "\n";
    return 0;
}
```

### Python
``` python
import sys

def main():
    # 优化输入
    input_data = sys.stdin.read().split()
    if not input_data:
        return
        
    n = int(input_data[0])
    a = [int(x) for x in input_data[1:n+1]]
    
    L_val = [0] * n
    L_cost = [0] * n
    L_val[0] = a[0]
    L_cost[0] = 0
    for i in range(1, n):
        L_val[i] = max(a[i], L_val[i-1] + 1)
        L_cost[i] = L_cost[i-1] + (L_val[i] - a[i])
        
    R_val = [0] * n
    R_cost = [0] * n
    R_val[n-1] = a[n-1]
    R_cost[n-1] = 0
    for i in range(n-2, -1, -1):
        R_val[i] = max(a[i], R_val[i+1] + 1)
        R_cost[i] = R_cost[i+1] + (R_val[i] - a[i])
        
    min_total = -1
    for k in range(1, n-1):
        peak = max(a[k], L_val[k-1] + 1, R_val[k+1] + 1)
        total = L_cost[k-1] + R_cost[k+1] + (peak - a[k])
        if min_total == -1 or total < min_total:
            min_total = total
            
    print(min_total)

if __name__ == '__main__':
    main()
```

``` java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.IOException;
import java.util.StringTokenizer;

public class Main {
    public static void main(String[] args) throws IOException {
        // 优化输入读取，使用BufferedReader和StringTokenizer
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        String line = br.readLine();
        while (line != null && line.trim().isEmpty()) {
            line = br.readLine();
        }
        if (line == null) return;
        
        int n = Integer.parseInt(line.trim());
        long[] a = new long[n];
        int idx = 0;
        
        while (idx < n) {
            line = br.readLine();
            if (line == null) break;
            StringTokenizer st = new StringTokenizer(line);
            while (st.hasMoreTokens() && idx < n) {
                a[idx++] = Long.parseLong(st.nextToken());
            }
        }
        
        long[] L_val = new long[n];
        long[] L_cost = new long[n];
        L_val[0] = a[0];
        L_cost[0] = 0;
        for (int i = 1; i < n; i++) {
            L_val[i] = Math.max(a[i], L_val[i-1] + 1);
            L_cost[i] = L_cost[i-1] + (L_val[i] - a[i]);
        }
        
        long[] R_val = new long[n];
        long[] R_cost = new long[n];
        R_val[n-1] = a[n-1];
        R_cost[n-1] = 0;
        for (int i = n - 2; i >= 0; i--) {
            R_val[i] = Math.max(a[i], R_val[i+1] + 1);
            R_cost[i] = R_cost[i+1] + (R_val[i] - a[i]);
        }
        
        long min_total = -1;
        for (int k = 1; k < n - 1; k++) {
            long peak = Math.max(a[k], Math.max(L_val[k-1] + 1, R_val[k+1] + 1));
            long total = L_cost[k-1] + R_cost[k+1] + (peak - a[k]);
            if (min_total == -1 || total < min_total) {
                min_total = total;
            }
        }
        
        System.out.println(min_total);
    }
}
```

``` go
	if a > b {
		return a
	}
	return b
}
```
