# 模拟
[题目链接](https://kamacoder.com/problempage.php?pid=1383)
## 思路分析
所有老员工必须单独成组，然后将新员工安排到老员工没满的组，剩下的新员工（如果还有的话），单独成组。

统计老员工的数量cnt，新员工的数量就是n - cnt = k。那么最终的答案为max(cnt, cnt + ceil((k - cnt) / 3.0))。
## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(1)。

### Python
``` python
def main():
    n = int(input().strip())
    nums = list(map(int, input().split()))
    cnt = sum(1 for num in nums if num == 1)
    ans = max(cnt, cnt + (n - 2 * cnt + 2) // 3)
    print(ans)

if __name__ == "__main__":
    main()
```

### Java

``` java
import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());
        int n = Integer.parseInt(st.nextToken());
        int[] nums = new int[n];
        int cnt = 0;
        
        st = new StringTokenizer(br.readLine());
        for (int i = 0; i < n; i++) {
            nums[i] = Integer.parseInt(st.nextToken());
            if (nums[i] == 1) {
                cnt++;
            }
        }
        
        int ans = Math.max(cnt, cnt + (n - 2 * cnt + 2) / 3);
        System.out.println(ans);
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	
	// 读取第一行
	line, _ := reader.ReadString('\n')
	line = strings.TrimSpace(line)
	n, _ := strconv.Atoi(line)
	
	// 读取第二行
	line, _ = reader.ReadString('\n')
	line = strings.TrimSpace(line)
	parts := strings.Split(line, " ")
	
	cnt := 0
	for i := 0; i < n; i++ {
		num, _ := strconv.Atoi(parts[i])
		if num == 1 {
			cnt++
		}
	}
	
	ans := max(cnt, cnt + (n - 2*cnt + 2)/3)
	fmt.Println(ans)
}

func max(a, b int) int {
	if a > b {
		return a
	}
	return b
}
```
