## 思路分析

这道题有一个性质，因为这个矩阵是通过一个字符串右移操作形成的。所以一个连续的0在完成一次右移之后，再放到下一行与这段连续的0会有一个交集，这个交集中0的数量是减一的。所以要找最大的面积，只需要找到最大的连续的0的段长。注意这里需要考虑首尾的0也算连续的0，为了方便计算，只需要将两个一样的字符串拼接起来即可。

1. 三角形：如果最大的连续0长度为x，那么面积就是``` x * (x + 1) / 2 ```;
2. 矩形：设最大段长为x，那么每次换一行的时候交集的段长减一，所以这个矩形的某一条边长只会在[1, x]内。而高稳定+1，所以最终矩形面积为``` ans = max(ans, k * (x - k + 1)) ```。
3. 最终的结果就是矩形和三角形面积的最大值。

## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(n)。

### Python
``` python
import sys

def main():
    s = sys.stdin.readline().strip()
    n = len(s)
    
    # 找到最后一个1的位置
    p = n - 1
    while p >= 0 and s[p] == '0':
        p -= 1
    
    # 如果全是0
    if p < 0:
        print(n * n)
        return
    
    # 构造双倍字符串
    s1 = s + s
    max_gap = 0
    j = 0
    
    # 找到最长的连续0的长度
    for i in range(2 * n):
        if s1[i] == '0':
            continue
        max_gap = max(max_gap, i - j)
        j = i + 1
    
    # 计算三角形排列数量
    tri = max_gap * (max_gap + 1) // 2
    
    # 计算矩形排列的最大值
    rect = 0
    for i in range(1, max_gap + 1):
        edge = max_gap - (i - 1)
        rect = max(rect, i * edge)
    
    print(max(tri, rect))

if __name__ == "__main__":
    main()
```

### Java
``` java
import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        
        String s = br.readLine().trim();
        int n = s.length();
        
        // 找到最后一个1的位置
        int p = n - 1;
        while (p >= 0 && s.charAt(p) == '0') {
            p--;
        }
        
        // 如果全是0
        if (p < 0) {
            bw.write((long)n * n + "\n");
            bw.flush();
            return;
        }
        
        // 构造双倍字符串
        String s1 = s + s;
        int maxGap = 0;
        int j = 0;
        
        // 找到最长的连续0的长度
        for (int i = 0; i < 2 * n; i++) {
            if (s1.charAt(i) == '0') {
                continue;
            }
            maxGap = Math.max(maxGap, i - j);
            j = i + 1;
        }
        
        // 计算三角形排列数量
        long tri = (long)maxGap * (maxGap + 1) / 2;
        
        // 计算矩形排列的最大值
        long rect = 0;
        for (int i = 1; i <= maxGap; i++) {
            int edge = maxGap - (i - 1);
            rect = Math.max(rect, (long)i * edge);
        }
        
        // 输出较大值
        bw.write(Math.max(tri, rect) + "\n");
        bw.flush();
        bw.close();
        br.close();
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
)

func main() {
	in := bufio.NewReader(os.Stdin)
	out := bufio.NewWriter(os.Stdout)
	defer out.Flush()

	var s string
	fmt.Fscan(in, &s)
	n := len(s)

	// 找到最后一个1的位置
	p := n - 1
	for p >= 0 && s[p] == '0' {
		p--
	}

	// 如果全是0
	if p < 0 {
		fmt.Fprintln(out, n*n)
		return
	}

	// 构造双倍字符串
	s1 := s + s
	maxGap := 0
	j := 0

	// 找到最长的连续0的长度
	for i := 0; i < 2*n; i++ {
		if s1[i] == '0' {
			continue
		}
		if i-j > maxGap {
			maxGap = i - j
		}
		j = i + 1
	}

	// 计算三角形排列数量
	tri := maxGap * (maxGap + 1) / 2

	// 计算矩形排列的最大值
	rect := 0
	for i := 1; i <= maxGap; i++ {
		edge := maxGap - (i - 1)
		if i*edge > rect {
			rect = i * edge
		}
	}

	// 输出较大值
	if tri > rect {
		fmt.Fprintln(out, tri)
	} else {
		fmt.Fprintln(out, rect)
	}
}
```
