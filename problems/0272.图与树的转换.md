# Kruskal最小生成树算法
[题目链接](https://kamacoder.com/problempage.php?pid=1352)
## 思路分析
这里简单介绍一下最小生成树的定义：我们定义无向连通图的 最小生成树（Minimum Spanning Tree，MST）为边权和最小的生成树。

Kruskal 算法是一种常见并且好写的最小生成树算法，由 Kruskal 发明。该算法的基本思想是从小到大加入边，是个贪心算法。不熟悉的可以先学一下这个算法。

在了解了这个算法的原理后，由于我们只需要1号起点和n号起点的度数之差尽可能的大，那么可以考虑以下两种情况：

1. 让1号节点尽可能保留更多的边，n号节点尽可能保留更少的边；
2. 让n号节点尽可能保留更多的边，1号节点尽可能保留更少的边。

那么通过Kruskal算法得到最小生成树的时候，我们控制这个图的边权为：

1. 与1号节点相连的边边权为1，n号节点相连的边为-1，其他节点是0；
2. 与n号节点相连的边边权为1，1号节点相连的边为-1，其他节点是0；

上面两种情况分别可以对应上面提到的第二种和第一种情况。这样，我们最终求两种情况哪种情况得到的度数之差最小即可。由于要保留字典序最小的答案，所以要对Kruskal算法进行一些修改，即对边进行排序，如果两个边的权重不一样，选择权重小的，如果权重一样，编号小的边排在前面。这样就可以有限选到编号小的边。

## 复杂度分析
1. 时间复杂度：O(mlog(m))；
2. 空间复杂度：O(n + m)。

``` cpp
#include <bits/stdc++.h>
using namespace std;

struct DSU {
    vector<int> p;
    DSU(int n): p(n+1, -1) {}
    int find(int x) {
        return p[x] < 0 ? x : p[x] = find(p[x]);
    }
    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (p[a] > p[b]) swap(a, b);
        p[a] += p[b];
        p[b] = a;
        return true;
    }
};

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;
    struct Edge { int u, v, idx; };
    vector<Edge> edges(m);
    for(int i = 0; i < m; i++){
        cin >> edges[i].u >> edges[i].v;
        edges[i].idx = i + 1;
    }

    // Kruskal 模板：传入 cost 函数 f(e) 返回 -1/0/+1
    auto runKruskal = [&](auto&& f) -> tuple<vector<int>, int, int> {
        // 构造 (cost, idx, edge_id)
        struct Item { int cost, idx, id; };
        vector<Item> ord;
        ord.reserve(m);
        for(auto &e : edges) {
            ord.push_back({ f(e), e.idx, e.idx });
        }
        sort(ord.begin(), ord.end(), [](auto &a, auto &b){
            if (a.cost != b.cost) return a.cost < b.cost;
            return a.idx < b.idx;
        });

        DSU dsu(n);
        vector<int> picked;
        picked.reserve(n-1);
        vector<int> deg(n+1, 0);
        for(auto &it : ord){
            auto &e = edges[it.id - 1];
            if ((int)picked.size() == n-1) break;
            if (dsu.unite(e.u, e.v)) {
                picked.push_back(e.idx);
                if (e.u == 1 || e.v == 1) deg[1]++;
                if (e.u == n || e.v == n) deg[n]++;
            }
        }
        sort(picked.begin(), picked.end());
        return { picked, deg[1], deg[n] };
    };

    // 情况 1
    auto [p1, d1_1, d1_n] = runKruskal([&](auto &e){
        if (e.u == 1 || e.v == 1) return -1;
        if (e.u == n || e.v == n) return +1;
        return 0;
    });
    int diff1 = abs(d1_1 - d1_n);

    // 情况 2
    auto [p2, d2_1, d2_n] = runKruskal([&](auto &e){
        if (e.u == n || e.v == n) return -1;
        if (e.u == 1 || e.v == 1) return +1;
        return 0;
    });
    int diff2 = abs(d2_1 - d2_n);

    // 选更优或字典序更小
    const vector<int> *ans;
    if (diff1 > diff2) {
        ans = &p1;
    } else if (diff2 > diff1) {
        ans = &p2;
    } else {
        ans = (p1 < p2 ? &p1 : &p2);
    }

    // 输出
    for(int i = 0; i < (int)ans->size(); i++){
        if (i) cout << ' ';
        cout << (*ans)[i];
    }
    cout << "\n";
    return 0;
}
```

``` python
import sys
sys.setrecursionlimit(1 << 25)

n, m = map(int, sys.stdin.readline().split())
edges = []
for i in range(m):
    u, v = map(int, sys.stdin.readline().split())
    edges.append((u, v, i + 1))

class DSU:
    def __init__(self, n):
        self.p = [-1] * (n + 1)

    def find(self, x):
        if self.p[x] < 0:
            return x
        self.p[x] = self.find(self.p[x])
        return self.p[x]

    def union(self, x, y):
        x, y = self.find(x), self.find(y)
        if x == y:
            return False
        if self.p[x] > self.p[y]:
            x, y = y, x
        self.p[x] += self.p[y]
        self.p[y] = x
        return True

def kruskal(weight_func):
    items = []
    for u, v, idx in edges:
        cost = weight_func(u, v)
        items.append((cost, idx, u, v))
    items.sort()

    dsu = DSU(n)
    deg = [0] * (n + 1)
    res = []
    for _, idx, u, v in items:
        if len(res) == n - 1:
            break
        if dsu.union(u, v):
            res.append(idx)
            if u == 1 or v == 1:
                deg[1] += 1
            if u == n or v == n:
                deg[n] += 1
    res.sort()
    return res, deg[1], deg[n]

# case 1
res1, d1_1, d1_n = kruskal(lambda u, v: -1 if u == 1 or v == 1 else (1 if u == n or v == n else 0))
# case 2
res2, d2_1, d2_n = kruskal(lambda u, v: -1 if u == n or v == n else (1 if u == 1 or v == 1 else 0))

diff1 = abs(d1_1 - d1_n)
diff2 = abs(d2_1 - d2_n)
```

``` java
import java.io.*;
import java.util.*;
import java.util.function.BiFunction;


public class Main {
    static class Edge {
        int u, v, idx;
        Edge(int u, int v, int idx) {
            this.u = u;
            this.v = v;
            this.idx = idx;
        }
    }

    static class DSU {
        int[] parent;
        DSU(int n) {
            parent = new int[n + 1];
            Arrays.fill(parent, -1);
        }
        int find(int x) {
            return parent[x] < 0 ? x : (parent[x] = find(parent[x]));
        }
        boolean union(int a, int b) {
            a = find(a);
            b = find(b);
            if (a == b) return false;
            if (parent[a] > parent[b]) {
                int temp = a; a = b; b = temp;
            }
            parent[a] += parent[b];
            parent[b] = a;
            return true;
        }
    }

    static int n, m;
    static List<Edge> edges = new ArrayList<>();

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());
        n = Integer.parseInt(st.nextToken());
        m = Integer.parseInt(st.nextToken());
        for (int i = 1; i <= m; i++) {
            st = new StringTokenizer(br.readLine());
            int u = Integer.parseInt(st.nextToken()), v = Integer.parseInt(st.nextToken());
            edges.add(new Edge(u, v, i));
        }

        Result r1 = kruskal((u, v) -> (u == 1 || v == 1) ? -1 : ((u == n || v == n) ? 1 : 0));
        Result r2 = kruskal((u, v) -> (u == n || v == n) ? -1 : ((u == 1 || v == 1) ? 1 : 0));

        int diff1 = Math.abs(r1.d1 - r1.dn), diff2 = Math.abs(r2.d1 - r2.dn);
        List<Integer> ans;
        if (diff1 > diff2) ans = r1.list;
        else if (diff2 > diff1) ans = r2.list;
        else ans = (r1.list.toString().compareTo(r2.list.toString()) < 0) ? r1.list : r2.list;

        for (int i = 0; i < ans.size(); i++) {
            if (i > 0) System.out.print(" ");
            System.out.print(ans.get(i));
        }
        System.out.println();
    }

    static class Result {
        List<Integer> list;
        int d1, dn;
        Result(List<Integer> list, int d1, int dn) {
            this.list = list;
            this.d1 = d1;
            this.dn = dn;
        }
    }

    static Result kruskal(BiFunction<Integer, Integer, Integer> costFunc) {
        List<int[]> list = new ArrayList<>();
        for (Edge e : edges) {
            list.add(new int[]{ costFunc.apply(e.u, e.v), e.idx, e.u, e.v });
        }
        list.sort((a, b) -> a[0] != b[0] ? Integer.compare(a[0], b[0]) : Integer.compare(a[1], b[1]));

        DSU dsu = new DSU(n);
        List<Integer> res = new ArrayList<>();
        int[] deg = new int[n + 1];
        for (int[] item : list) {
            int idx = item[1], u = item[2], v = item[3];
            if (res.size() == n - 1) break;
            if (dsu.union(u, v)) {
                res.add(idx);
                if (u == 1 || v == 1) deg[1]++;
                if (u == n || v == n) deg[n]++;
            }
        }
        Collections.sort(res);
        return new Result(res, deg[1], deg[n]);
    }
}
```

``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"sort"
)

type Edge struct {
	u, v, idx int
}

type DSU struct {
	p []int
}

func NewDSU(n int) *DSU {
	p := make([]int, n+1)
	for i := range p {
		p[i] = -1
	}
	return &DSU{p}
}

func (d *DSU) Find(x int) int {
	if d.p[x] < 0 {
		return x
	}
	d.p[x] = d.Find(d.p[x])
	return d.p[x]
}

func (d *DSU) Union(x, y int) bool {
	x, y = d.Find(x), d.Find(y)
	if x == y {
		return false
	}
	if d.p[x] > d.p[y] {
		x, y = y, x
	}
	d.p[x] += d.p[y]
	d.p[y] = x
	return true
}

type Result struct {
	list    []int
	deg1, dgn int
}

func kruskal(edges []Edge, n int, cost func(u, v int) int) Result {
	type Item struct {
		cost, idx, u, v int
	}
	items := make([]Item, len(edges))
	for i, e := range edges {
		items[i] = Item{cost(e.u, e.v), e.idx, e.u, e.v}
	}
	sort.Slice(items, func(i, j int) bool {
		if items[i].cost != items[j].cost {
			return items[i].cost < items[j].cost
		}
		return items[i].idx < items[j].idx
	})

	dsu := NewDSU(n)
	deg := make([]int, n+1)
	var res []int
	for _, it := range items {
		if len(res) == n-1 {
			break
		}
		if dsu.Union(it.u, it.v) {
			res = append(res, it.idx)
			if it.u == 1 || it.v == 1 {
				deg[1]++
			}
			if it.u == n || it.v == n {
				deg[n]++
			}
		}
	}
	sort.Ints(res)
	return Result{res, deg[1], deg[n]}
}

func main() {
	reader := bufio.NewReader(os.Stdin)
	var n, m int
	fmt.Fscan(reader, &n, &m)

	edges := make([]Edge, m)
	for i := 0; i < m; i++ {
		fmt.Fscan(reader, &edges[i].u, &edges[i].v)
		edges[i].idx = i + 1
	}

	r1 := kruskal(edges, n, func(u, v int) int {
		if u == 1 || v == 1 {
			return -1
		} else if u == n || v == n {
			return 1
		}
		return 0
	})
	r2 := kruskal(edges, n, func(u, v int) int {
		if u == n || v == n {
			return -1
		} else if u == 1 || v == 1 {
			return 1
		}
		return 0
	})

	diff1 := abs(r1.deg1 - r1.dgn)
	diff2 := abs(r2.deg1 - r2.dgn)

	ans := r1.list
	if diff2 > diff1 || (diff1 == diff2 && less(r2.list, r1.list)) {
		ans = r2.list
	}
	for i, x := range ans {
		if i > 0 {
			fmt.Print(" ")
		}
		fmt.Print(x)
	}
	fmt.Println()
}

func abs(x int) int {
	if x < 0 {
		return -x
	}
	return x
}

func less(a, b []int) bool {
	for i := 0; i < len(a); i++ {
		if a[i] < b[i] {
			return true
		} else if a[i] > b[i] {
			return false
		}
	}
	return false
}
```
