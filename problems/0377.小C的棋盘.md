# Dijkstra最短路径
[题目链接](https://kamacoder.com/problempage.php?pid=1457)

## 思路分析
这道题需要找出战棋地图上有多少个**安全棋子**。安全棋子的定义是：**不被任何其他同色棋子攻击到**。同色棋子之间可以发生攻击，攻击的条件是最短有向路径距离不大于发起攻击棋子的攻击距离 k。

对于任何一种颜色，如果有多个同色棋子，我们就需要判断：对于任意一个同色棋子 u，是否能被除自己以外的任何一个同色棋子，在满足其攻击距离 <= k 的条件下触达。如果能，它就不安全。

对于单一颜色的一组源点跑多源最短路时，如果不加处理，距离为 0 的自身最短路会掩盖其他同色棋子到达它的短路信息。直接维护每个源点的全源最短路时间复杂度会比较高，不可接受。
但是，这里有一个关键点：**我们只需要知道到达目标顶点 v 的最短路能否由非 v 自身的某个其他源点贡献。** 
因此，我们利用一种**维护最小与次小源的最短路**的做法：
在每次使用 Dijkstra 计算多源最短路时，对图中的每个顶点维护两对二元组：
- `(dist[v][0], src[v][0])`：到达点 v 的**最小**距离及其对应的源点。
- `(dist[v][1], src[v][1])`：到达点 v 的**次小**距离及其对应的源点，**且要求与最小距离的源点不同** (`src[v][0] != src[v][1]`)。

因为对于任何源点集合中的点 u，它到自己的最小距离必定为 0（因为题目边长 d >= 1）。如果其他源点能到达 u，它在经过不断竞争后必定会去更新 u 的“次小源记录”。如果在最短路跑完后，u 维护的次小源记录 `dist[u][1] <= 颜色攻击距离 k`，就证明有另外的同色棋子能够攻击到 u，u 就判定为不安全。

## 复杂度分析
1. 时间复杂度：O(c * (N + M * logN))；其中，c 为颜色种数。
2. 空间复杂度：O(N + M)。

### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

const long long INF = 2e18; // 设置足够大

struct Edge {
    int to;
    long long w;
};

struct State {
    long long d;
    int u;
    int s;
    bool operator>(const State& other) const {
        return d > other.d;
    }
};

long long dist_arr[100005][2];
int src_arr[100005][2];

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m, c;
    if (!(cin >> n >> m >> c)) return 0;

    vector<int> colors(n + 1);
    vector<vector<int>> nodes_of_color(c + 1);
    for (int i = 1; i <= n; i++) {
        cin >> colors[i];
        nodes_of_color[colors[i]].push_back(i);
    }

    vector<long long> k_arr(c + 1);
    for (int i = 1; i <= c; i++) {
        cin >> k_arr[i];
    }

    vector<vector<Edge>> adj(n + 1);
    for (int i = 0; i < m; i++) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
    }

    vector<bool> is_safe(n + 1, false);

    for (int color = 1; color <= c; color++) {
        const auto& nodes = nodes_of_color[color];
        if (nodes.empty()) continue;
        if (nodes.size() == 1) {
            is_safe[nodes[0]] = true;
            continue;
        }

        // 初始化
        for (int i = 1; i <= n; i++) {
            dist_arr[i][0] = dist_arr[i][1] = INF;
            src_arr[i][0] = src_arr[i][1] = -1;
        }

        priority_queue<State, vector<State>, greater<State>> pq;
        for (int u : nodes) {
            dist_arr[u][0] = 0;
            src_arr[u][0] = u;
            pq.push({0, u, u});
        }

        long long k_limit = k_arr[color];

        while (!pq.empty()) {
            auto [d, u, s] = pq.top();
            pq.pop();

            if (d > k_limit) continue;

            if (s == src_arr[u][0] && d > dist_arr[u][0]) continue;
            if (s == src_arr[u][1] && d > dist_arr[u][1]) continue;
            if (s != src_arr[u][0] && s != src_arr[u][1]) continue;

            for (const auto& edge : adj[u]) {
                int v = edge.to;
                long long nd = d + edge.w;

                if (nd > k_limit) continue;

                if (s == src_arr[v][0]) {
                    if (nd < dist_arr[v][0]) {
                        dist_arr[v][0] = nd;
                        pq.push({nd, v, s});
                    }
                } else if (s == src_arr[v][1]) {
                    if (nd < dist_arr[v][1]) {
                        dist_arr[v][1] = nd;
                        if (dist_arr[v][1] < dist_arr[v][0]) {
                            swap(dist_arr[v][0], dist_arr[v][1]);
                            swap(src_arr[v][0], src_arr[v][1]);
                        }
                        pq.push({nd, v, s});
                    }
                } else {
                    if (nd < dist_arr[v][0]) {
                        dist_arr[v][1] = dist_arr[v][0];
                        src_arr[v][1] = src_arr[v][0];
                        dist_arr[v][0] = nd;
                        src_arr[v][0] = s;
                        pq.push({nd, v, s});
                    } else if (nd < dist_arr[v][1]) {
                        dist_arr[v][1] = nd;
                        src_arr[v][1] = s;
                        pq.push({nd, v, s});
                    }
                }
            }
        }

        for (int u : nodes) {
            if (dist_arr[u][1] <= k_limit) {
                is_safe[u] = false;
            } else {
                is_safe[u] = true;
            }
        }
    }

    int ans = 0;
    for (int i = 1; i <= n; i++) {
        if (is_safe[i]) ans++;
    }
    cout << ans << "\n";

    return 0;
}
```

``` python
import sys
import heapq

def main():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    n = int(input_data[0])
    m = int(input_data[1])
    c = int(input_data[2])
    
    idx = 3
    colors = [0] * (n + 1)
    nodes_of_color = [[] for _ in range(c + 1)]
    for i in range(1, n + 1):
        colors[i] = int(input_data[idx])
        nodes_of_color[colors[i]].append(i)
        idx += 1
        
    k_arr = [0] * (c + 1)
    for i in range(1, c + 1):
        k_arr[i] = int(input_data[idx])
        idx += 1
        
    adj = [[] for _ in range(n + 1)]
    for _ in range(m):
        u = int(input_data[idx])
        v = int(input_data[idx+1])
        w = int(input_data[idx+2])
        adj[u].append((v, w))
        idx += 3
        
    INF = 2 * 10**18
    is_safe = [False] * (n + 1)
    
    # 预先分配以提升速度
    dist = [[INF, INF] for _ in range(n + 1)]
    src = [[-1, -1] for _ in range(n + 1)]
    
    for color in range(1, c + 1):
        nodes = nodes_of_color[color]
        if not nodes:
            continue
        if len(nodes) == 1:
            is_safe[nodes[0]] = True
            continue
            
        for i in range(1, n + 1):
            dist[i][0] = dist[i][1] = INF
            src[i][0] = src[i][1] = -1
            
        pq = []
        for u in nodes:
            dist[u][0] = 0
            src[u][0] = u
            heapq.heappush(pq, (0, u, u))
            
        k_limit = k_arr[color]
        
        while pq:
            d, u, s = heapq.heappop(pq)
            
            if d > k_limit:
                continue
                
            if s == src[u][0] and d > dist[u][0]: continue
            if s == src[u][1] and d > dist[u][1]: continue
            if s != src[u][0] and s != src[u][1]: continue
                
            for v, w in adj[u]:
                nd = d + w
                if nd > k_limit: continue
                    
                if s == src[v][0]:
                    if nd < dist[v][0]:
                        dist[v][0] = nd
                        heapq.heappush(pq, (nd, v, s))
                elif s == src[v][1]:
                    if nd < dist[v][1]:
                        dist[v][1] = nd
                        if dist[v][1] < dist[v][0]:
                            dist[v][0], dist[v][1] = dist[v][1], dist[v][0]
                            src[v][0], src[v][1] = src[v][1], src[v][0]
                        heapq.heappush(pq, (nd, v, s))
                else:
                    if nd < dist[v][0]:
                        dist[v][1], src[v][1] = dist[v][0], src[v][0]
                        dist[v][0], src[v][0] = nd, s
                        heapq.heappush(pq, (nd, v, s))
                    elif nd < dist[v][1]:
                        dist[v][1], src[v][1] = nd, s
                        heapq.heappush(pq, (nd, v, s))
                        
        for u in nodes:
            if dist[u][1] <= k_limit:
                is_safe[u] = False
            else:
                is_safe[u] = True
                
    ans = sum(is_safe)
    print(ans)

if __name__ == '__main__':
    main()
```

``` java
import java.io.*;
import java.util.*;

public class Main {
    static class Edge {
        int to;
        long w;
        Edge(int to, long w) { this.to = to; this.w = w; }
    }

    static class State implements Comparable<State> {
        long d;
        int u;
        int s;
        State(long d, int u, int s) { this.d = d; this.u = u; this.s = s; }
        public int compareTo(State other) {
            return Long.compare(this.d, other.d);
        }
    }

    public static void main(String[] args) throws IOException {
        // Fast I/O
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = null;
        
        String line = br.readLine();
        if (line == null) return;
        st = new StringTokenizer(line);
        int n = Integer.parseInt(st.nextToken());
        int m = Integer.parseInt(st.nextToken());
        int c = Integer.parseInt(st.nextToken());

        int[] colors = new int[n + 1];
        List<List<Integer>> nodesOfColor = new ArrayList<>(c + 1);
        for (int i = 0; i <= c; i++) nodesOfColor.add(new ArrayList<>());
        
        st = new StringTokenizer(br.readLine());
        for (int i = 1; i <= n; i++) {
            colors[i] = Integer.parseInt(st.nextToken());
            nodesOfColor.get(colors[i]).add(i);
        }

        long[] kArr = new long[c + 1];
        st = new StringTokenizer(br.readLine());
        for (int i = 1; i <= c; i++) {
            kArr[i] = Long.parseLong(st.nextToken());
        }

        List<List<Edge>> adj = new ArrayList<>(n + 1);
        for (int i = 0; i <= n; i++) adj.add(new ArrayList<>());

        for (int i = 0; i < m; i++) {
            st = new StringTokenizer(br.readLine());
            int u = Integer.parseInt(st.nextToken());
            int v = Integer.parseInt(st.nextToken());
            long w = Long.parseLong(st.nextToken());
            adj.get(u).add(new Edge(v, w));
        }

        boolean[] isSafe = new boolean[n + 1];
        long[][] dist = new long[n + 1][2];
        int[][] src = new int[n + 1][2];
        final long INF = 2000000000000000000L;

        PriorityQueue<State> pq = new PriorityQueue<>();

        for (int color = 1; color <= c; color++) {
            List<Integer> nodes = nodesOfColor.get(color);
            if (nodes.isEmpty()) continue;
            if (nodes.size() == 1) {
                isSafe[nodes.get(0)] = true;
                continue;
            }

            for (int i = 1; i <= n; i++) {
                dist[i][0] = INF; dist[i][1] = INF;
                src[i][0] = -1; src[i][1] = -1;
            }

            pq.clear();
            for (int u : nodes) {
                dist[u][0] = 0;
                src[u][0] = u;
                pq.add(new State(0, u, u));
            }

            long kLimit = kArr[color];

            while (!pq.isEmpty()) {
                State state = pq.poll();
                long d = state.d;
                int u = state.u;
                int s = state.s;

                if (d > kLimit) continue;
                if (s == src[u][0] && d > dist[u][0]) continue;
                if (s == src[u][1] && d > dist[u][1]) continue;
                if (s != src[u][0] && s != src[u][1]) continue;

                for (Edge edge : adj.get(u)) {
                    int v = edge.to;
                    long nd = d + edge.w;

                    if (nd > kLimit) continue;

                    if (s == src[v][0]) {
                        if (nd < dist[v][0]) {
                            dist[v][0] = nd;
                            pq.add(new State(nd, v, s));
                        }
                    } else if (s == src[v][1]) {
                        if (nd < dist[v][1]) {
                            dist[v][1] = nd;
                            if (dist[v][1] < dist[v][0]) {
                                long tempD = dist[v][0]; dist[v][0] = dist[v][1]; dist[v][1] = tempD;
                                int tempS = src[v][0]; src[v][0] = src[v][1]; src[v][1] = tempS;
                            }
                            pq.add(new State(nd, v, s));
                        }
                    } else {
                        if (nd < dist[v][0]) {
                            dist[v][1] = dist[v][0];
                            src[v][1] = src[v][0];
                            dist[v][0] = nd;
                            src[v][0] = s;
                            pq.add(new State(nd, v, s));
                        } else if (nd < dist[v][1]) {
                            dist[v][1] = nd;
                            src[v][1] = s;
                            pq.add(new State(nd, v, s));
                        }
                    }
                }
            }

            for (int u : nodes) {
                if (dist[u][1] <= kLimit) isSafe[u] = false;
                else isSafe[u] = true;
            }
        }

        int ans = 0;
        for (int i = 1; i <= n; i++) {
            if (isSafe[i]) ans++;
        }
        System.out.println(ans);
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"container/heap"
	"fmt"
	"os"
	"strconv"
)

type Edge struct {
	to int
	w  int64
}

type State struct {
	d int64
	u int
	s int
}

type MinHeap []State

func (h MinHeap) Len() int           { return len(h) }
func (h MinHeap) Less(i, j int) bool { return h[i].d < h[j].d }
func (h MinHeap) Swap(i, j int)      { h[i], h[j] = h[j], h[i] }
func (h *MinHeap) Push(x interface{}) {
	*h = append(*h, x.(State))
}
func (h *MinHeap) Pop() interface{} {
	old := *h
	n := len(old)
	x := old[n-1]
	*h = old[0 : n-1]
	return x
}

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)
	buf := make([]byte, 1024*1024)
	scanner.Buffer(buf, 1024*1024*10)

	scanInt := func() int {
		if !scanner.Scan() {
			return 0
		}
		val, _ := strconv.Atoi(scanner.Text())
		return val
	}
	scanInt64 := func() int64 {
		if !scanner.Scan() {
			return 0
		}
		val, _ := strconv.ParseInt(scanner.Text(), 10, 64)
		return val
	}

	n := scanInt()
	if n == 0 {
		return
	}
	m := scanInt()
	c := scanInt()

	colors := make([]int, n+1)
	nodesOfColor := make([][]int, c+1)
	for i := 1; i <= n; i++ {
		colors[i] = scanInt()
		nodesOfColor[colors[i]] = append(nodesOfColor[colors[i]], i)
	}

	kArr := make([]int64, c+1)
	for i := 1; i <= c; i++ {
		kArr[i] = scanInt64()
	}

	adj := make([][]Edge, n+1)
	for i := 0; i < m; i++ {
		u := scanInt()
		v := scanInt()
		w := scanInt64()
		adj[u] = append(adj[u], Edge{v, w})
	}

	isSafe := make([]bool, n+1)
	dist := make([][2]int64, n+1)
	src := make([][2]int, n+1)
	const INF int64 = 2e18

	for color := 1; color <= c; color++ {
		nodes := nodesOfColor[color]
		if len(nodes) == 0 {
			continue
		}
		if len(nodes) == 1 {
			isSafe[nodes[0]] = true
			continue
		}

		for i := 1; i <= n; i++ {
			dist[i][0] = INF
			dist[i][1] = INF
			src[i][0] = -1
			src[i][1] = -1
		}

		pq := &MinHeap{}
		heap.Init(pq)
		for _, u := range nodes {
			dist[u][0] = 0
			src[u][0] = u
			heap.Push(pq, State{0, u, u})
		}

		kLimit := kArr[color]

		for pq.Len() > 0 {
			state := heap.Pop(pq).(State)
			d, u, s := state.d, state.u, state.s

			if d > kLimit {
				continue
			}
			if s == src[u][0] && d > dist[u][0] {
				continue
			}
			if s == src[u][1] && d > dist[u][1] {
				continue
			}
			if s != src[u][0] && s != src[u][1] {
				continue
			}

			for _, edge := range adj[u] {
				v := edge.to
				nd := d + edge.w

				if nd > kLimit {
					continue
				}

				if s == src[v][0] {
					if nd < dist[v][0] {
						dist[v][0] = nd
						heap.Push(pq, State{nd, v, s})
					}
				} else if s == src[v][1] {
					if nd < dist[v][1] {
						dist[v][1] = nd
						if dist[v][1] < dist[v][0] {
							dist[v][0], dist[v][1] = dist[v][1], dist[v][0]
							src[v][0], src[v][1] = src[v][1], src[v][0]
						}
						heap.Push(pq, State{nd, v, s})
					}
				} else {
					if nd < dist[v][0] {
						dist[v][1], src[v][1] = dist[v][0], src[v][0]
						dist[v][0], src[v][0] = nd, s
						heap.Push(pq, State{nd, v, s})
					} else if nd < dist[v][1] {
						dist[v][1] = nd
						src[v][1] = s
						heap.Push(pq, State{nd, v, s})
					}
				}
			}
		}

		for _, u := range nodes {
			if dist[u][1] <= kLimit {
				isSafe[u] = false
			} else {
				isSafe[u] = true
			}
		}
	}

	ans := 0
	for i := 1; i <= n; i++ {
		if isSafe[i] {
			ans++
		}
	}
	fmt.Println(ans)
}
```
