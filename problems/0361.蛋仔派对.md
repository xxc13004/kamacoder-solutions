# BFS + 贪心 + 排序
[题目链接](https://kamacoder.com/problempage.php?pid=1441)
## 思路分析

这道题比较长，可以将任务分解来看：

1. 每个蛋仔需要以最短的时间到达迷宫出口处，等待符合条件就逃离迷宫（逃离能量够了）；
2. 逃脱的蛋仔会留下自己的能量给其他蛋仔；
3. 每个蛋仔互不影响，自己行动时 其他蛋仔不受影响，也不消耗能量。

有了上面三个条件，我们就分开解决这些问题。

1. 第一个问题其实是一个多源的广搜问题，我们可以求出每个蛋仔是否能到达迷宫的出口，如果有一个不可以就输出-1，如果可以，就记录这个蛋仔到达出口处的最短路径，也就是消耗的时间的数量dist[i];
2. 这里需要用到贪心的思想，我们肯定希望逃离迷宫出口所需最小能量Ei的蛋仔先走，这样他就能留下自己的能量给迷宫，其他蛋仔也可以使用了;
3. 每个蛋仔逃离迷宫后剩余的能量就是Ki - dist[i]。

综上所述：

我们先求出所有蛋仔的最短到达出口时间dist[i]，实际做法可以从出口进行广搜，搜索到达每个格子的最短距离，然后按照三元组（Ei, Ri, dist[i]）对每个蛋仔进行排序，分别表示：

- Ei：蛋仔逃离所需的能量；
- Ri：蛋仔到达出口处所剩的能量，即逃离时留下的能量，Ri = Ki - dist[i]；
- dist[i]：蛋仔到达出口处所消耗的能量/时间，也就是最短路径。

最短按照顺序处理每个蛋仔即可，注意逃离的蛋仔，将其剩余的能量累加起来。

## 复杂度分析
1. 时间复杂度：O(NM + XlogX)，广搜 + 排序；
2. 空间复杂度：O(NM)。
## 代码实现
### Cpp
``` cpp
#include <bits/stdc++.h>
#define endl '\n'
const int N = 500000, MOD = 1000000007;
using namespace std;
using LL = long long;
using PIL = pair<int, LL>;
using PII = pair<int, int>;
using T3I = tuple<int, int, int>;
int dx[4] = {1, -1, 0, 0};
int dy[4] = {0, 0, 1, -1};
int main() {
    cin.tie(nullptr) -> sync_with_stdio(false);
    int P;
    cin >> P;
    while(P--){
        int N, M, X, Y;
        long long E;
        cin >> N >> M >> X >> Y >> E;

        // 蛋仔的坐标、逃离能量、初始能量
        vector<tuple<int,int,long long,long long>> eggs(X);
        for(int i = 0; i < X; i++){
            int x, y;
            long long Ei, Ki;
            cin >> y >> x >> Ei >> Ki;
            eggs[i] = {x-1, y-1, Ei, Ki};
        }

        // 逃离点的坐标，出口
        vector<pair<int,int>> exits(Y);
        for(int i = 0; i < Y; i++){
            int x, y;
            cin >> y >> x;
            exits[i] = {x-1, y-1};
        }
        // 迷宫
        vector<string> grid(N);
        for(int i = 0; i < N; i++)
            cin >> grid[i];
        // 存储每个格子到出口的最短距离
        vector<vector<int>> dist(N, vector<int>(M, MOD));
        queue<pair<int,int>> q;
        // 将出口加入队列，进行多源广搜
        for(auto &e : exits){
            dist[e.first][e.second] = 0;
            q.push({e.first, e.second});
        }

        while(!q.empty()){
            auto [x, y] = q.front(); q.pop();
            for(int d = 0; d < 4; d++){
                int nx = x + dx[d];
                int ny = y + dy[d];
                if(nx >= 0 && nx < N && ny >= 0 && ny < M){
                    if(grid[nx][ny] == '.' && dist[nx][ny] == MOD){
                        dist[nx][ny] = dist[x][y] + 1;
                        q.push({nx,ny});
                    }
                }
            }
        }

        vector<tuple<long long,long long,int>> arr;
        long long totalTime = 0;
        bool ok = true;

        // 计算三个核心的数值
        for(auto &egg : eggs){
            int x, y;
            long long Ei, Ki;
            tie(x,y,Ei,Ki) = egg;

            int d = dist[x][y];
            if(d==MOD || d>Ki){
                ok = false;
                break;
            }

            totalTime += d;
            arr.push_back({Ei, Ki-d, d});
        }

        // 有任何一个蛋仔出不去，都是-1
        if(!ok){
            cout << -1 << "\n";
            continue;
        }

        sort(arr.begin(), arr.end());

        long long cur = E;

        for(auto &e : arr){
            long long Ei, Ri;
            int d;
            tie(Ei, Ri, d) = e;
            // 当前最优解的情况，该蛋仔能量不够不能逃脱
            if(cur < Ei){
                ok = false;
                break;
            }
            cur += Ri;
        }

        if(!ok) cout << -1 << "\n";
        else cout << totalTime << "\n";
    }
    return 0;
}
```

``` python
import sys
from collections import deque

input = sys.stdin.readline
INF = 10**18

dx = [1,-1,0,0]
dy = [0,0,1,-1]

P = int(input())
for _ in range(P):
    N,M,X,Y,E = map(int,input().split())
    
    eggs=[]
    for _ in range(X):
        x,y,Ei,Ki=map(int,input().split())
        eggs.append((x-1,y-1,Ei,Ki))
    
    dist=[[INF]*M for _ in range(N)]
    q=deque()
    
    exits=[]
    for _ in range(Y):
        x,y=map(int,input().split())
        x-=1;y-=1
        dist[y][x]=0
        q.append((y,x))
    
    grid=[list(input().strip()) for _ in range(N)]
    
    while q:
        y,x=q.popleft()
        for i in range(4):
            ny=y+dy[i]
            nx=x+dx[i]
            if 0<=ny<N and 0<=nx<M:
                if grid[ny][nx]=='.' and dist[ny][nx]==INF:
                    dist[ny][nx]=dist[y][x]+1
                    q.append((ny,nx))
    
    arr=[]
    total=0
    ok=True
    
    for x,y,Ei,Ki in eggs:
        d=dist[y][x]
        if d==INF or d>Ki:
            ok=False
            break
        total+=d
        arr.append((Ei,Ki-d))
    
    if not ok:
        print(-1)
        continue
    
    arr.sort()
    cur=E
    
    for Ei,Ri in arr:
        if cur<Ei:
            ok=False
            break
        cur+=Ri
    
    print(total if ok else -1)
```

### Java
``` java
import java.io.*;
import java.util.*;

public class Main {

    static int[] dx = {1,-1,0,0};
    static int[] dy = {0,0,1,-1};
    static final int INF = 1_000_000_000;

    public static void main(String[] args) throws Exception{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringBuilder sb = new StringBuilder();
        int P = Integer.parseInt(br.readLine());

        while(P-- > 0){
            StringTokenizer st = new StringTokenizer(br.readLine());
            int N = Integer.parseInt(st.nextToken());
            int M = Integer.parseInt(st.nextToken());
            int X = Integer.parseInt(st.nextToken());
            int Y = Integer.parseInt(st.nextToken());
            long E = Long.parseLong(st.nextToken());

            int[][] eggPos = new int[X][2];
            long[] EiArr = new long[X];
            long[] KiArr = new long[X];

            for(int i=0;i<X;i++){
                st = new StringTokenizer(br.readLine());
                eggPos[i][0] = Integer.parseInt(st.nextToken())-1;
                eggPos[i][1] = Integer.parseInt(st.nextToken())-1;
                EiArr[i] = Long.parseLong(st.nextToken());
                KiArr[i] = Long.parseLong(st.nextToken());
            }

            Queue<int[]> q = new LinkedList<>();
            int[][] dist = new int[N][M];
            for(int i=0;i<N;i++)
                Arrays.fill(dist[i], INF);

            for(int i=0;i<Y;i++){
                st = new StringTokenizer(br.readLine());
                int x = Integer.parseInt(st.nextToken())-1;
                int y = Integer.parseInt(st.nextToken())-1;
                dist[y][x]=0;
                q.offer(new int[]{y,x});
            }

            char[][] grid = new char[N][];
            for(int i=0;i<N;i++)
                grid[i] = br.readLine().toCharArray();

            while(!q.isEmpty()){
                int[] cur = q.poll();
                int y=cur[0], x=cur[1];
                for(int d=0;d<4;d++){
                    int ny=y+dy[d], nx=x+dx[d];
                    if(ny>=0&&ny<N&&nx>=0&&nx<M){
                        if(grid[ny][nx]=='.' && dist[ny][nx]==INF){
                            dist[ny][nx]=dist[y][x]+1;
                            q.offer(new int[]{ny,nx});
                        }
                    }
                }
            }

            List<long[]> list = new ArrayList<>();
            long total=0;
            boolean ok=true;

            for(int i=0;i<X;i++){
                int x=eggPos[i][0], y=eggPos[i][1];
                int d=dist[y][x];
                if(d==INF||d>KiArr[i]){
                    ok=false;
                    break;
                }
                total+=d;
                list.add(new long[]{EiArr[i], KiArr[i]-d});
            }

            if(!ok){
                sb.append(-1).append("\n");
                continue;
            }

            list.sort(Comparator.comparingLong(a->a[0]));

            long curE=E;
            for(long[] a:list){
                if(curE<a[0]){
                    ok=false;
                    break;
                }
                curE+=a[1];
            }

            sb.append(ok?total:-1).append("\n");
        }
        System.out.print(sb);
    }
}
```

### Go
``` go
package main
import (
	"bufio"
	"fmt"
	"os"
	"sort"
)

var dx = []int{1, -1, 0, 0}
var dy = []int{0, 0, 1, -1}

const INF = int(1e9)

func main() {
	in := bufio.NewReader(os.Stdin)
	out := bufio.NewWriter(os.Stdout)
	defer out.Flush()

	var P int
	fmt.Fscan(in, &P)

	for ; P > 0; P-- {
		var N, M, X, Y int
		var E int64
		fmt.Fscan(in, &N, &M, &X, &Y, &E)

		type Egg struct {
			x, y int
			Ei, Ki int64
		}

		eggs := make([]Egg, X)
		for i := 0; i < X; i++ {
			fmt.Fscan(in, &eggs[i].x, &eggs[i].y, &eggs[i].Ei, &eggs[i].Ki)
			eggs[i].x--
			eggs[i].y--
		}

		dist := make([][]int, N)
		for i := range dist {
			dist[i] = make([]int, M)
			for j := range dist[i] {
				dist[i][j] = INF
			}
		}

		queue := make([][2]int, 0)

		for i := 0; i < Y; i++ {
			var x, y int
			fmt.Fscan(in, &x, &y)
			x--; y--
			dist[y][x] = 0
			queue = append(queue, [2]int{y, x})
		}

		grid := make([][]byte, N)
		for i := 0; i < N; i++ {
			var s string
			fmt.Fscan(in, &s)
			grid[i] = []byte(s)
		}

		for head := 0; head < len(queue); head++ {
			y := queue[head][0]
			x := queue[head][1]
			for d := 0; d < 4; d++ {
				ny := y + dy[d]
				nx := x + dx[d]
				if ny >= 0 && ny < N && nx >= 0 && nx < M {
					if grid[ny][nx] == '.' && dist[ny][nx] == INF {
						dist[ny][nx] = dist[y][x] + 1
						queue = append(queue, [2]int{ny, nx})
					}
				}
			}
		}

		type Node struct {
			Ei, Ri int64
		}

		nodes := make([]Node, 0)
		var total int64 = 0
		ok := true

		for _, e := range eggs {
			d := dist[e.y][e.x]
			if d == INF || int64(d) > e.Ki {
				ok = false
				break
			}
			total += int64(d)
			nodes = append(nodes, Node{e.Ei, e.Ki - int64(d)})
		}

		if !ok {
			fmt.Fprintln(out, -1)
			continue
		}

		sort.Slice(nodes, func(i, j int) bool {
			return nodes[i].Ei < nodes[j].Ei
		})

		cur := E
		for _, n := range nodes {
			if cur < n.Ei {
				ok = false
				break
			}
			cur += n.Ri
		}

		if ok {
			fmt.Fprintln(out, total)
		} else {
			fmt.Fprintln(out, -1)
		}
	}
}
```
