# 动态规划
[题目链接](https://kamacoder.com/problempage.php?pid=1460)
## 思路分析
本题需要在一个给定的数组中寻找“波峰区间”，并找出使得区间内峰顶最大值与两端最小值之差最大的区间。

一个“极大波峰区间”可以视为以某个元素作为峰顶 `m`，向左找到非递减序列的最左端点，向右找到非递增序列的最右端点。在这个最大的跨度范围内，区间最大值自然是 `array[m]`，而最小值则是两端点的值中较小的那个。极大化区间的跨度能确保两端的值尽可能地小，从而使差值 `array[m] - min(左端点值, 右端点值)` 尽可能大。

我们可以通过两趟遍历（类似动态规划思想）以 O(N) 的时间复杂度提前算出每个点向左、向右延伸的最远端点：
1. **向左找左边界**：从左往右遍历数组，如果当前元素大于等于前一个元素，则它的左边界等于前一个元素的左边界；否则左边界就是它自己。
2. **向右找右边界**：从右往左遍历数组，如果当前元素大于等于后一个元素，则它的右边界等于后一个元素的右边界；否则右边界就是它自己。

有了左右边界后，我们遍历每个元素作为峰顶，计算并维护全局最大的差值即可。

## 复杂度分析
1. 时间复杂度：O(N)；
2. 空间复杂度：O(N)。

## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    // 优化输入输出
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int m;
    if (!(cin >> m)) return 0;
    if (m == 0) {
        cout << 0 << "\n";
        return 0;
    }

    vector<int> arr(m);
    for (int i = 0; i < m; ++i) {
        cin >> arr[i];
    }

    // 记录每个元素向左最多能延伸到的底端位置
    vector<int> left_bound(m);
    left_bound[0] = 0;
    for (int i = 1; i < m; ++i) {
        if (arr[i - 1] <= arr[i]) {
            left_bound[i] = left_bound[i - 1];
        } else {
            left_bound[i] = i;
        }
    }

    // 记录每个元素向右最多能延伸到的底端位置
    vector<int> right_bound(m);
    right_bound[m - 1] = m - 1;
    for (int i = m - 2; i >= 0; --i) {
        if (arr[i + 1] <= arr[i]) {
            right_bound[i] = right_bound[i + 1];
        } else {
            right_bound[i] = i;
        }
    }

    // 遍历每一个峰顶，寻找最大的差值
    int max_diff = 0;
    for (int i = 0; i < m; ++i) {
        int cur_min = min(arr[left_bound[i]], arr[right_bound[i]]);
        max_diff = max(max_diff, arr[i] - cur_min);
    }

    cout << max_diff << "\n";
    return 0;
}
```

### Python
``` python
import sys

def solve():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    m = int(input_data[0])
    if m == 0:
        print(0)
        return
        
    arr = [int(x) for x in input_data[1:m+1]]
    
    # 记录每个元素向左最多能延伸到的底端位置
    left_bound = [0] * m
    left_bound[0] = 0
    for i in range(1, m):
        if arr[i - 1] <= arr[i]:
            left_bound[i] = left_bound[i - 1]
        else:
            left_bound[i] = i
            
    # 记录每个元素向右最多能延伸到的底端位置
    right_bound = [0] * m
    right_bound[m - 1] = m - 1
    for i in range(m - 2, -1, -1):
        if arr[i + 1] <= arr[i]:
            right_bound[i] = right_bound[i + 1]
        else:
            right_bound[i] = i
            
    # 计算差值
    max_diff = 0
    for i in range(m):
        cur_min = min(arr[left_bound[i]], arr[right_bound[i]])
        if arr[i] - cur_min > max_diff:
            max_diff = arr[i] - cur_min
            
    print(max_diff)

if __name__ == '__main__':
    solve()
```

### Java
``` java
import java.io.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StreamTokenizer st = new StreamTokenizer(br);
        
        if (st.nextToken() != StreamTokenizer.TT_EOF) {
            int m = (int) st.nval;
            if (m == 0) {
                System.out.println(0);
                return;
            }
            
            int[] arr = new int[m];
            for (int i = 0; i < m; i++) {
                st.nextToken();
                arr[i] = (int) st.nval;
            }
            
            int[] leftBound = new int[m];
            leftBound[0] = 0;
            for (int i = 1; i < m; i++) {
                if (arr[i - 1] <= arr[i]) {
                    leftBound[i] = leftBound[i - 1];
                } else {
                    leftBound[i] = i;
                }
            }
            
            int[] rightBound = new int[m];
            rightBound[m - 1] = m - 1;
            for (int i = m - 2; i >= 0; i--) {
                if (arr[i + 1] <= arr[i]) {
                    rightBound[i] = rightBound[i + 1];
                } else {
                    rightBound[i] = i;
                }
            }
            
            int maxDiff = 0;
            for (int i = 0; i < m; i++) {
                int curMin = Math.min(arr[leftBound[i]], arr[rightBound[i]]);
                maxDiff = Math.max(maxDiff, arr[i] - curMin);
            }
            System.out.println(maxDiff);
        }
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
)

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)
	
	if scanner.Scan() {
		var m int
		fmt.Sscanf(scanner.Text(), "%d", &m)
		if m == 0 {
			fmt.Println(0)
			return
		}
		
		arr := make([]int, m)
		for i := 0; i < m; i++ {
			scanner.Scan()
			fmt.Sscanf(scanner.Text(), "%d", &arr[i])
		}
		
		leftBound := make([]int, m)
		leftBound[0] = 0
		for i := 1; i < m; i++ {
			if arr[i-1] <= arr[i] {
				leftBound[i] = leftBound[i-1]
			} else {
				leftBound[i] = i
			}
		}
		
		rightBound := make([]int, m)
		rightBound[m-1] = m - 1
		for i := m - 2; i >= 0; i-- {
			if arr[i+1] <= arr[i] {
				rightBound[i] = rightBound[i+1]
			} else {
				rightBound[i] = i
			}
		}
		
		maxDiff := 0
		for i := 0; i < m; i++ {
			leftVal := arr[leftBound[i]]
			rightVal := arr[rightBound[i]]
			
			minVal := leftVal
			if rightVal < minVal {
				minVal = rightVal
			}
			
			diff := arr[i] - minVal
			if diff > maxDiff {
				maxDiff = diff
			}
		}
		
		fmt.Println(maxDiff)
	}
}
```
