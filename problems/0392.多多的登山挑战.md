# 二分答案与动态规划
[题目链接](https://kamacoder.com/problempage.php?pid=1472)
## 思路分析
题目的核心是寻找一条从营地 1 到营地 n 的路径，使得到达终点时所需的“最少补给包容量”尽可能小。由于我们只在容量不够时才获取补给包且永远不能丢下补给包，因此所需的最终容量必定等于该路径上所需经过的边权 W 的最大值（即 max We）。

由于题目保证边总是从编号小的营地指向编号大的营地（U_i < V_i），整个营地的图是一个有向无环图（DAG），天然存在拓扑序。
这就允许我们使用二分答案+动态规划的方法：
1. **二分最终的容量限度 K**：
   假设最终允许携带的补给包容量为 K。那么在这条路上，所有途径路段所需的包裹数都必须 <= K。
2. **动态规划验证**：
   在图上仅保留 W <= K 的边。
   定义 S[u] 表示在只走 W <= K 的边到达营地 u 的所有合法路径中，**累计能捡拾的最大补给包数量**。
   初始化 S[1] = A_1，其余为 -1。
   按照顶点编号递增的顺序 u = 1 -> n（即拓扑序）更新后续节点：
   若 S[u] ≠ -1，说明可以到达该节点。遍历 u 的所有邻边 (u, v, w)：
   由于我们需要越过这条边，首先必须满足边权限制即 W <= K，且到 u 时能捡到的补给包需要足以走过这条边，即 S[u] >= W。
   若满足上述条件，则营地 v 可以到达，且可以更新最大累计补给数为 S[v] = max(S[v], S[u] + A_v)。
3. 最后检查 S[n] ≠ -1 是否成立，若成立说明在容量不超过 K 的情况下可以抵达终点 n。寻找最小的符合条件的 K 即可。

## 复杂度分析
1. 时间复杂度：O((n + m) log (max W))；二分查找的最大界限为 max W <= 10^9，即进行约 30 次。每次二分利用已有拓扑序进行图遍历（DP状态转移），遍历所有节点与边消耗时间为 O(n + m)。整体表现优异。
2. 空间复杂度：O(n + m)；主要空间用于图的邻接表存储，以及长度为 n 的一维状态数组 S，内存开销呈线性。

## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Edge {
    int v;
    long long w;
};

int main() {
    // 优化输入输出
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n, m;
    if (!(cin >> n >> m)) return 0;
    
    vector<long long> A(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> A[i];
    }
    
    vector<vector<Edge>> adj(n + 1);
    long long max_w = 0;
    for (int i = 0; i < m; i++) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        if (w > max_w) {
            max_w = w;
        }
    }
    
    // DP检验函数
    auto check = [&](long long K) {
        vector<long long> S(n + 1, -1);
        S[1] = A[1];
        
        for (int u = 1; u < n; u++) {
            if (S[u] == -1) continue;
            for (auto& edge : adj[u]) {
                // 仅考虑不超载(K)且现有累加补给数足够越过当段路程的边
                if (edge.w <= K && S[u] >= edge.w) {
                    if (S[u] + A[edge.v] > S[edge.v]) {
                        S[edge.v] = S[u] + A[edge.v];
                    }
                }
            }
        }
        return S[n] != -1;
    };
    
    long long low = 1, high = max_w;
    long long ans = -1;
    
    // 二分找最小容量
    while (low <= high) {
        long long mid = low + (high - low) / 2;
        if (check(mid)) {
            ans = mid;
            high = mid - 1; // 尝试更小的容量
        } else {
            low = mid + 1;
        }
    }
    
    cout << ans << "\n";
    return 0;
}
```

### Python
``` python
import sys

def solve():
    # 优化读取：一次性读取所有内容，极大提高速度
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    n = int(input_data[0])
    m = int(input_data[1])
    
    A = [0] * (n + 1)
    idx = 2
    for i in range(1, n + 1):
        A[i] = int(input_data[idx])
        idx += 1
        
    adj = [[] for _ in range(n + 1)]
    max_w = 0
    for _ in range(m):
        u = int(input_data[idx])
        v = int(input_data[idx+1])
        w = int(input_data[idx+2])
        idx += 3
        adj[u].append((v, w))
        if w > max_w:
            max_w = w
            
    def check(K):
        S = [-1] * (n + 1)
        S[1] = A[1]
        for u in range(1, n):
            su = S[u]
            if su == -1:
                continue
            for v, w in adj[u]:
                if w <= K and su >= w:
                    nxt = su + A[v]
                    if nxt > S[v]:
                        S[v] = nxt
        return S[n] != -1

    ans = -1
    low = 1
    high = max_w
    
    # 二分查找
    while low <= high:
        mid = (low + high) // 2
        if check(mid):
            ans = mid
            high = mid - 1
        else:
            low = mid + 1
            
    print(ans)

if __name__ == '__main__':
    solve()
```

``` java
import java.io.InputStream;
import java.io.IOException;
import java.util.ArrayList;

class FastScanner {
    private InputStream stream;
    private byte[] buf = new byte[1024];
    private int head = 0;
    private int tail = 0;

    public FastScanner(InputStream stream) {
        this.stream = stream;
    }

    private int read() throws IOException {
        if (head >= tail) {
            head = 0;
            tail = stream.read(buf, 0, buf.length);
            if (tail <= 0) return -1;
        }
        return buf[head++];
    }

    public int nextInt() throws IOException {
        int c = read();
        while (c <= 32) {
            if (c == -1) return -1;
            c = read();
        }
        int res = 0;
        while (c > 32) {
            res = res * 10 + c - '0';
            c = read();
        }
        return res;
    }

    public long nextLong() throws IOException {
        int c = read();
        while (c <= 32) {
            if (c == -1) return -1;
            c = read();
        }
        long res = 0;
        while (c > 32) {
            res = res * 10 + c - '0';
            c = read();
        }
        return res;
    }
}

public class Main {
    static class Edge {
        int v;
        long w;
        Edge(int v, long w) {
            this.v = v;
            this.w = w;
        }
    }

    public static void main(String[] args) throws IOException {
        FastScanner scanner = new FastScanner(System.in);
        int n = scanner.nextInt();
        if (n == -1) return;
        int m = scanner.nextInt();

        long[] A = new long[n + 1];
        for (int i = 1; i <= n; i++) {
            A[i] = scanner.nextLong();
        }

        ArrayList<Edge>[] adj = new ArrayList[n + 1];
        for (int i = 1; i <= n; i++) {
            adj[i] = new ArrayList<>();
        }

        long max_w = 0;
        for (int i = 0; i < m; i++) {
            int u = scanner.nextInt();
            int v = scanner.nextInt();
            long w = scanner.nextLong();
            adj[u].add(new Edge(v, w));
            if (w > max_w) {
                max_w = w;
            }
        }

        long low = 1, high = max_w;
        long ans = -1;

        while (low <= high) {
            long mid = low + (high - low) / 2;
            long[] S = new long[n + 1];
            for (int i = 1; i <= n; i++) S[i] = -1;
            S[1] = A[1];

            for (int u = 1; u < n; u++) {
                if (S[u] == -1) continue;
                for (Edge edge : adj[u]) {
                    if (edge.w <= mid && S[u] >= edge.w) {
                        long nxt = S[u] + A[edge.v];
                        if (nxt > S[edge.v]) {
                            S[edge.v] = nxt;
                        }
                    }
                }
            }

            if (S[n] != -1) {
                ans = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }

        System.out.println(ans);
    }
}
```

``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
)

type Edge struct {
	v int
	w int64
}

func main() {
    // 扩展缓冲区用于应对庞大的图输入
	scanner := bufio.NewScanner(os.Stdin)
	buf := make([]byte, 1024*1024)
	scanner.Buffer(buf, 1024*1024*10)
	scanner.Split(bufio.ScanWords)

	if !scanner.Scan() {
		return
	}
	n, _ := strconv.Atoi(scanner.Text())
	scanner.Scan()
	m, _ := strconv.Atoi(scanner.Text())

	A := make([]int64, n+1)
	for i := 1; i <= n; i++ {
		scanner.Scan()
		val, _ := strconv.ParseInt(scanner.Text(), 10, 64)
		A[i] = val
	}

	adj := make([][]Edge, n+1)
	var max_w int64 = 0
	for i := 0; i < m; i++ {
		scanner.Scan()
		u, _ := strconv.Atoi(scanner.Text())
		scanner.Scan()
		v, _ := strconv.Atoi(scanner.Text())
		scanner.Scan()
		w, _ := strconv.ParseInt(scanner.Text(), 10, 64)
		adj[u] = append(adj[u], Edge{v, w})
		if w > max_w {
			max_w = w
		}
	}

	check := func(K int64) bool {
		S := make([]int64, n+1)
		for i := 1; i <= n; i++ {
			S[i] = -1
		}
		S[1] = A[1]
		for u := 1; u < n; u++ {
			su := S[u]
			if su == -1 {
				continue
			}
			for _, edge := range adj[u] {
				if edge.w <= K && su >= edge.w {
					nxt := su + A[edge.v]
					if nxt > S[edge.v] {
						S[edge.v] = nxt
					}
				}
			}
		}
		return S[n] != -1
	}

	var low int64 = 1
	var high int64 = max_w
	var ans int64 = -1

	for low <= high {
		mid := low + (high-low)/2
		if check(mid) {
			ans = mid
			high = mid - 1
		} else {
			low = mid + 1
		}
	}

	fmt.Println(ans)
}
```
