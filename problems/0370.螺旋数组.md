# 模拟
[题目链接](https://kamacoder.com/problempage.php?pid=1450)
## 思路分析

本题可以划分为两个主要部分：
1. **字符串解析**：
    - 输入的格式类似于标准的二维数组字符串（如 `[[1,2,3],[8,9,4],[7,6,5]]`）
    - 我们可以使用一个**有限状态机**的思路遍历该字符串，在遇到 `[` 时新建一行，在遇到 `,` 和 `]` 时截取积累的数字存入当前行。并在内层数组闭合 `]` 时将整行数据存入矩阵中。
2. **螺旋遍历**：
    - 通过设定四个边界（即 `top` 上边界、`bottom` 下边界、`left` 左边界、`right` 右边界）。
    - 依次按 **“向右” -> “向下” -> “向左” -> “向上”** 的顺序遍历：
      - 从 `left` 到 `right` 遍历上边界，完成后收缩上边界（`top++`）。
      - 从 `top` 到 `bottom` 遍历右边界，完成后收缩右边界（`right--`）。
      - 在确加上、右边界没有越界的前提下，从 `right` 到 `left` 遍历下边界，完成后收缩下边界（`bottom--`）。
      - 在确加上、右边界没有越界的前提下，从 `bottom` 到 `top` 遍历左边界，完成后收缩左边界（`left++`）。
    - 循环执行以上逻辑，直到边界交叉为止。

## 复杂度分析
1. 时间复杂度：O(L + N * M)。其中L为输入字符串的长度。
2. 空间复杂度：O(N * M)。

## 代码实现
### Cpp
```cpp
#include <iostream>
#include <vector>
#include <string>

using namespace std;

int main() {
    string s;
    if (!getline(cin, s)) return 0;

    // 1. 优化解析输入字符串，构建二维数组
    vector<vector<int>> matrix;
    vector<int> row;
    string num = "";

    for (int i = 0; i < s.length(); i++) {
        char c = s[i];
        if (c == '[') {
            row.clear();
        } else if (c == ']') {
            if (!num.empty()) {
                row.push_back(stoi(num));
                num = "";
            }
            if (!row.empty()) {
                matrix.push_back(row);
                row.clear();
            }
        } else if (c == ',') {
            if (!num.empty()) {
                row.push_back(stoi(num));
                num = "";
            }
        } else if (c != ' ' && c != '\r' && c != '\n') { // 防止末尾回车符被吸入
            num += c;
        }
    }

    if (matrix.empty() || matrix[0].empty()) {
        cout << "[]" << endl;
        return 0;
    }

    // 2. 螺旋遍历
    vector<int> res;
    int top = 0, bottom = matrix.size() - 1;
    int left = 0, right = matrix[0].size() - 1;

    while (top <= bottom && left <= right) {
        // 向右
        for (int j = left; j <= right; j++) res.push_back(matrix[top][j]);
        top++;
        
        // 向下
        for (int i = top; i <= bottom; i++) res.push_back(matrix[i][right]);
        right--;
        
        // 向左
        if (top <= bottom) {
            for (int j = right; j >= left; j--) res.push_back(matrix[bottom][j]);
            bottom--;
        }
        
        // 向上
        if (left <= right) {
            for (int i = bottom; i >= top; i--) res.push_back(matrix[i][left]);
            left++;
        }
    }

    // 3. 格式化输出
    cout << "[";
    for (int i = 0; i < res.size(); i++) {
        cout << res[i] << (i == res.size() - 1 ? "" : ",");
    }
    cout << "]" << endl;

    return 0;
}
```

```python
import json, sys

# 1. 使用 json.loads 直接将合法结构转化为二维数组
s = sys.stdin.read().strip()    
matrix = json.loads(s)

# 2. 螺旋遍历
res = []
top, bottom = 0, len(matrix) - 1
left, right = 0, len(matrix[0]) - 1

while top <= bottom and left <= right:
    # 向右
    for j in range(left, right + 1):
        res.append(matrix[top][j])
    top += 1
    
    # 向下
    for i in range(top, bottom + 1):
        res.append(matrix[i][right])
    right -= 1
    
    # 向左
    if top <= bottom:
        for j in range(right, left - 1, -1):
            res.append(matrix[bottom][j])
        bottom -= 1
        
    # 向上
    if left <= right:
        for i in range(bottom, top - 1, -1):
            res.append(matrix[i][left])
        left += 1  # 修复：左边界向右收缩

# 3. 格式化输出
print("[" + ",".join(map(str, res)) + "]")
```

### Java
```java
import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        if (!scanner.hasNextLine()) return;
        String s = scanner.nextLine().trim();

        // 1. 优化解析输入字符串，构建二维数组
        List<List<Integer>> matrix = new ArrayList<>();
        List<Integer> row = new ArrayList<>();
        StringBuilder num = new StringBuilder();

        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c == '[') {
                row.clear();
            } else if (c == ']') {
                if (num.length() > 0) {
                    row.add(Integer.parseInt(num.toString()));
                    num.setLength(0);
                }
                if (!row.isEmpty()) {
                    matrix.add(new ArrayList<>(row));
                    row.clear();
                }
            } else if (c == ',') {
                if (num.length() > 0) {
                    row.add(Integer.parseInt(num.toString()));
                    num.setLength(0);
                }
            } else if (c != ' ' && c != '\r' && c != '\n') {
                num.append(c);
            }
        }

        if (matrix.isEmpty() || matrix.get(0).isEmpty()) {
            System.out.println("[]");
            return;
        }

        // 2. 螺旋遍历
        List<Integer> res = new ArrayList<>();
        int top = 0, bottom = matrix.size() - 1;
        int left = 0, right = matrix.get(0).size() - 1;

        while (top <= bottom && left <= right) {
            // 向右
            for (int j = left; j <= right; j++) res.add(matrix.get(top).get(j));
            top++;
            
            // 向下
            for (int i = top; i <= bottom; i++) res.add(matrix.get(i).get(right));
            right--;
            
            // 向左
            if (top <= bottom) {
                for (int j = right; j >= left; j--) res.add(matrix.get(bottom).get(j));
                bottom--;
            }
            
            // 向上
            if (left <= right) {
                for (int i = bottom; i >= top; i--) res.add(matrix.get(i).get(left));
                left++;
            }
        }

        // 3. 格式化输出
        StringBuilder out = new StringBuilder("[");
        for (int i = 0; i < res.size(); i++) {
            out.append(res.get(i)).append(i == res.size() - 1 ? "" : ",");
        }
        out.append("]");
        System.out.println(out.toString());
    }
}
```

### Go
```go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	line, _ := reader.ReadString('\n')
	line = strings.TrimSpace(line)

	if len(line) == 0 {
		return
	}

	// 1. 优化解析输入字符串，使用 []byte 代替 strings.Builder
	var matrix [][]int
	var row []int
	var numBuf []byte // 用原生字节切片做缓存

	for i := 0; i < len(line); i++ {
		ch := line[i]
		if ch == '[' {
			row = []int{} // 遇到 '[' 则重置行缓存
		} else if ch == ']' {
			if len(numBuf) > 0 {
				val, _ := strconv.Atoi(string(numBuf))
				row = append(row, val)
				numBuf = numBuf[:0] // 清空缓存
			}
			if len(row) > 0 {
				matrix = append(matrix, row)
				row = nil // 防止外层的 ']' 重复添加
			}
		} else if ch == ',' {
			if len(numBuf) > 0 {
				val, _ := strconv.Atoi(string(numBuf))
				row = append(row, val)
				numBuf = numBuf[:0] // 清空缓存
			}
		} else if ch != ' ' && ch != '\r' && ch != '\n' {
			numBuf = append(numBuf, ch) // 将字符存入字节切片
		}
	}

	if len(matrix) == 0 || len(matrix[0]) == 0 {
		fmt.Println("[]")
		return
	}

	// 2. 螺旋遍历
	var res []string
	top, bottom := 0, len(matrix)-1
	left, right := 0, len(matrix[0])-1

	for top <= bottom && left <= right {
		// 向右
		for j := left; j <= right; j++ {
			res = append(res, strconv.Itoa(matrix[top][j]))
		}
		top++

		// 向下
		for i := top; i <= bottom; i++ {
			res = append(res, strconv.Itoa(matrix[i][right]))
		}
		right--

		// 向左
		if top <= bottom {
			for j := right; j >= left; j-- {
				res = append(res, strconv.Itoa(matrix[bottom][j]))
			}
			bottom--
		}

		// 向上
		if left <= right {
			for i := bottom; i >= top; i-- {
				res = append(res, strconv.Itoa(matrix[i][left]))
			}
			left++ // 注意：此处正确使用 left++ 收缩边界
		}
	}

	// 3. 格式化输出
	fmt.Printf("[%s]\n", strings.Join(res, ","))
}
```
