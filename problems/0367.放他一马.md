# 动态规划（状态机DP）
[题目链接](https://kamacoder.com/problempage.php?pid=1447)
## 思路分析

这道题的核心在于：击败怪物获得的经验值，仅仅与当前**已经击败的怪物总数对 10 取模**的结果有关。因此，我们不需要记录具体击败了多少只怪物，只需要记录击败怪物数量的个位数即可。由于这个状态只会有 0 到 9 这 10 种可能，这非常适合使用动态规划（状态机 DP）来解决。

1. **状态定义**：
   维护一个长度为 10 的数组 `dp`。`dp[j]` 表示当前操作后，**击败的怪物总数对 10 取模等于 $j$** 时，小美能获得的**最大经验值**。

2. **初始状态**：
   在遇到任何怪物之前，击败怪物总数为 0，此时 `dp[0] = 0`。其他的 `dp[1] ~ dp[9]` 设为 $-1$（或负无穷），表示该状态尚未达到（不合法状态）。

3. **状态转移**：
   对于遇到的第 $i$ 只怪物（编号为 $i$，生命值为 $a$），我们新建一个 `next_dp` 数组来保存这一轮的选择结果，并遍历上一轮合法的 `dp[j]`（即 `dp[j] != -1`）：
   - **选择放走**：击败的怪物数没有增加，状态 $j$ 不变。获得的经验值为 $i$。
     $$next\_dp[j] = \max(next\_dp[j], dp[j] + i)$$
   - **选择击败**：击败的怪物数增加 1，状态变为 $next\_j = (j + 1) \bmod 10$。
     根据题意，额外获得 $(x \bmod 10) \times a$ 经验值，加上原本的 $a$，总共获得 $(1 + (x \bmod 10)) \times a$ 经验值。因此当 $next\_j = 0$ 时，倍数为 $1$；当 $next\_j \neq 0$ 时，倍数为 $next\_j + 1$。$$next\_dp[next\_j] = \max(next\_dp[next\_j], dp[j] + a \times 倍数)$$

4. **最终答案**：
   遍历所有的怪物并完成状态转移后，`dp` 数组中 0 到 9 状态的最大值即为小美最多可以获得的经验值。
   *注意：最大经验值可能非常大，超过 32 位整数的范围，因此需要使用 64 位整数（`long long`）来保存经验值。*

## 复杂度分析
1. 时间复杂度：O(n)。对于每只怪物，状态转移只会有常数级（10 次）计算，总时间复杂度与 n 成正比。
2. 空间复杂度：O(1)。利用滚动数组优化，每次只需要保留上一个怪物的 10 个状态以及当前怪物的 10 个状态，额外空间消耗为常数级别（如果不存储全量输入数据，采用边读边处理的方式，则空间占用极低）。

## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    // 优化输入输出流
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;

    // dp[j] 代表击败怪物数量 mod 10 == j 时的最大经验值
    vector<long long> dp(10, -1);
    dp[0] = 0; // 初始状态

    for (int i = 1; i <= n; ++i) {
        long long a;
        cin >> a;

        vector<long long> next_dp(10, -1);
        for (int j = 0; j < 10; ++j) {
            if (dp[j] != -1) {
                // 选择放走，状态不变，获得经验值为 i
                next_dp[j] = max(next_dp[j], dp[j] + i);

                // 选择击败，状态转移到 next_j
                int next_j = (j + 1) % 10;
                long long multiplier = (next_j == 0 ? 1 : next_j + 1);
                next_dp[next_j] = max(next_dp[next_j], dp[j] + a * multiplier);
            }
        }
        dp = next_dp; // 滚动数组
    }

    // 在最终所有的状态中找出最大值
    long long ans = 0;
    for (int j = 0; j < 10; ++j) {
        ans = max(ans, dp[j]);
    }

    cout << ans << "\n";
    return 0;
}
```

### Python
``` python
import sys

def main():
    # 优化读取方式
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    n = int(input_data[0])
    
    dp = [-1] * 10
    dp[0] = 0
    
    for i in range(1, n + 1):
        a = int(input_data[i])
        next_dp = [-1] * 10
        
        for j in range(10):
            if dp[j] != -1:
                # 策略1：放走
                if dp[j] + i > next_dp[j]:
                    next_dp[j] = dp[j] + i
                
                # 策略2：击败
                next_j = (j + 1) % 10
                multiplier = 1 if next_j == 0 else next_j + 1
                defeat_exp = dp[j] + a * multiplier
                if defeat_exp > next_dp[next_j]:
                    next_dp[next_j] = defeat_exp
                    
        dp = next_dp

    print(max(dp))

if __name__ == '__main__':
    main()
```

### Java
``` java
import java.io.InputStream;
import java.io.IOException;

public class Main {
    public static void main(String[] args) throws IOException {
        FastScanner scanner = new FastScanner(System.in);
        int n = scanner.nextInt();
        
        long[] dp = new long[10];
        for (int i = 1; i < 10; i++) {
            dp[i] = -1;
        }
        
        for (int i = 1; i <= n; i++) {
            long a = scanner.nextLong();
            long[] nextDp = new long[10];
            for (int j = 0; j < 10; j++) {
                nextDp[j] = -1;
            }
            
            for (int j = 0; j < 10; j++) {
                if (dp[j] != -1) {
                    // 放走
                    nextDp[j] = Math.max(nextDp[j], dp[j] + i);
                    
                    // 击败
                    int nextJ = (j + 1) % 10;
                    long multiplier = (nextJ == 0) ? 1 : (nextJ + 1);
                    nextDp[nextJ] = Math.max(nextDp[nextJ], dp[j] + a * multiplier);
                }
            }
            dp = nextDp;
        }
        
        long ans = 0;
        for (int j = 0; j < 10; j++) {
            ans = Math.max(ans, dp[j]);
        }
        System.out.println(ans);
    }
    
    // 内部实现 FastScanner 以达到极致的 IO 优化性能
    static class FastScanner {
        InputStream is;
        byte[] buffer = new byte[1024 * 8];
        int head = 0, tail = 0;

        public FastScanner(InputStream is) {
            this.is = is;
        }

        private int read() throws IOException {
            if (head >= tail) {
                head = 0;
                tail = is.read(buffer, 0, buffer.length);
                if (tail <= 0) return -1;
            }
            return buffer[head++];
        }

        public int nextInt() throws IOException {
            long val = nextLong();
            return (int) val;
        }

        public long nextLong() throws IOException {
            int c = read();
            while (c <= 32) {
                if (c == -1) return -1;
                c = read();
            }
            long res = 0;
            while (c > 32) {
                res = res * 10 + c - '0';
                c = read();
            }
            return res;
        }
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
)

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords) // 优化输入，按词读取

	if !scanner.Scan() {
		return
	}
	n, _ := strconv.Atoi(scanner.Text())

	dp := make([]int64, 10)
	for i := 1; i < 10; i++ {
		dp[i] = -1
	}

	for i := 1; i <= n; i++ {
		scanner.Scan()
		a, _ := strconv.ParseInt(scanner.Text(), 10, 64)

		nextDp := make([]int64, 10)
		for j := 0; j < 10; j++ {
			nextDp[j] = -1
		}

		for j := 0; j < 10; j++ {
			if dp[j] != -1 {
				// 策略1：放走
				if dp[j]+int64(i) > nextDp[j] {
					nextDp[j] = dp[j] + int64(i)
				}

				// 策略2：击败
				nextJ := (j + 1) % 10
				var multiplier int64
				if nextJ == 0 {
					multiplier = 1
				} else {
					multiplier = int64(nextJ + 1)
				}

				defeatExp := dp[j] + a*multiplier
				if defeatExp > nextDp[nextJ] {
					nextDp[nextJ] = defeatExp
				}
			}
		}
		dp = nextDp
	}

	var ans int64 = 0
	for j := 0; j < 10; j++ {
		if dp[j] > ans {
			ans = dp[j]
		}
	}
	fmt.Println(ans)
}
```
