# 排序 贪心
[题目链接](https://kamacoder.com/problempage.php?pid=1442)
## 思路分析

因为苹果树上的果子是每次摘取后都会翻倍。所以应该对数组进行排序，从最小的苹果树摘起才能保证获得的苹果最多。

## 复杂度分析
1. 时间复杂度：O(nlogn)；
2. 空间复杂度：O(1)。
## 代码实现
### Cpp
``` cpp
#include <bits/stdc++.h>
#define endl '\n'
const int N = 500000, MOD = 1000000007;
using namespace std;
using LL = long long;
using PIL = pair<int, LL>;
using PII = pair<int, int>;
using T3I = tuple<int, int, int>;
int dir[]{-1, 0, 1, 0, -1};
int main() {
    cin.tie(nullptr) -> sync_with_stdio(false);
    int n;
    cin >> n;
    vector<int> nums(n);
    for (int i = 0; i < n; ++i) cin >> nums[i];
    sort(nums.begin(), nums.end());
    LL ans = nums[0];
    LL p = 2;
    for (int i = 1; i < n; ++i) {
        ans = (ans + p * nums[i]) % MOD;
        p *= 2;
        p %= MOD;
    }
    cout << ans << endl;
    return 0;
}
```

``` python
import sys

def main():
    input = sys.stdin.readline
    n = int(input())
    nums = list(map(int, input().split()))
    nums.sort()
    
    MOD = 1000000007
    ans = nums[0]
    p = 2
    
    for i in range(1, n):
        ans = (ans + p * nums[i]) % MOD
        p = (p * 2) % MOD
    
    print(ans)

if __name__ == "__main__":
    main()
```

### Go
``` go
package main

import (
    "bufio"
    "fmt"
    "os"
    "sort"
    "strconv"
    "strings"
)

const MOD = 1000000007
const BUFFER_SIZE = 64 * 1024 // 64KB 大缓冲

func main() {
    // 创建带有大缓冲的读取器
    reader := bufio.NewReaderSize(os.Stdin, BUFFER_SIZE)
    
    // 读取第一行
    line, _ := reader.ReadString('\n')
    n, _ := strconv.Atoi(strings.TrimSpace(line))
    
    // 读取第二行
    line, _ = reader.ReadString('\n')
    parts := strings.Fields(line)
    
    nums := make([]int, n)
    for i := 0; i < n; i++ {
        nums[i], _ = strconv.Atoi(parts[i])
    }
    
    sort.Ints(nums)
    
    ans := int64(nums[0])
    p := int64(2)
    
    for i := 1; i < n; i++ {
        ans = (ans + p * int64(nums[i])) % MOD
        p = (p * 2) % MOD
    }
    
    fmt.Println(ans)
}
```
