# 哈希表
[题目链接](https://kamacoder.com/problempage.php?pid=1422)
## 思路分析

由于查询的字符串长度最多不超过10，所以只需要提前将所有长度不超过10的子串全部存储到哈希表，然后直接输出每一个查询的结果即可。

## 复杂度分析
1. 时间复杂度：O(10n)；
2. 空间复杂度：O(10n)。

``` cpp
    for (int i = 0; i < n; ++i) {
        for (int j = 1; j <= 10; ++j) {
            if (i + j - 1 >= n) break;
            ump[s.substr(i, j)]++;
        }
    }
    vector<string> qs(q);
    for (int i = 0; i < q; ++i) cin >> qs[i];
    for (string &qsi : qs) { 
        if (ump.count(qsi)) cout << ump[qsi] << endl;
        else cout << 0 << endl;
    }
    return 0;
}
```

### Python
``` python
import sys
 
def main():
    input_data = sys.stdin.read().split()
    n, q = map(int, input_data[:2])
    s = input_data[2]
     
    # 预处理所有长度1-10的子串
    ump = {}
    for i in range(n):
        for j in range(1, 11):  # 长度1到10
            if i + j > n:
                break
            sub = s[i:i+j]
            ump[sub] = ump.get(sub, 0) + 1
     
    # 处理查询
    idx = 3  # 跳过前3个输入（n, q, s）
    out_lines = []
    for _ in range(q):
        query = input_data[idx]
        idx += 1
        out_lines.append(str(ump.get(query, 0)))
     
    sys.stdout.write("\n".join(out_lines))
 
if __name__ == "__main__":
    main()
```

### Java
``` java
import java.util.*;
import java.io.*;
 
public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        PrintWriter pw = new PrintWriter(new BufferedWriter(new OutputStreamWriter(System.out)));
         
        String[] firstLine = br.readLine().split(" ");
        int n = Integer.parseInt(firstLine[0]);
        int q = Integer.parseInt(firstLine[1]);
         
        String s = br.readLine();
        Map<String, Integer> map = new HashMap<>();
         
        // 预处理所有长度1-10的子串
        for (int i = 0; i < n; i++) {
            for (int j = 1; j <= 10; j++) {
                if (i + j > n) break;
                String sub = s.substring(i, i + j);
                map.put(sub, map.getOrDefault(sub, 0) + 1);
            }
        }
         
        // 处理查询
        while (q-- > 0) {
            String query = br.readLine();
            pw.println(map.getOrDefault(query, 0));
        }
         
        pw.flush();
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"os"
	"strconv"
)

func main() {
	const bufferSize = 1 << 20 // 1MB缓冲区
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Buffer(make([]byte, bufferSize), bufferSize)
	scanner.Split(bufio.ScanWords)

	// 快速读取整数
	readInt := func() int {
		scanner.Scan()
		n, _ := strconv.Atoi(scanner.Text())
		return n
	}

	// 读取n和q
	n := readInt()
	q := readInt()

	// 读取字符串s
	scanner.Scan()
	s := scanner.Text()

	// 预处理所有长度1-10的子串
	ump := make(map[string]int, n*10) // 预分配容量，提高性能

	// 使用字节数组操作，避免多次分配字符串
	sBytes := []byte(s)
	for i := 0; i < n; i++ {
		// 限制子串最大长度为10且不超过字符串长度
		maxLen := 10
		if i+maxLen > n {
			maxLen = n - i
		}
		
		// 生成所有子串
		for j := 1; j <= maxLen; j++ {
			sub := string(sBytes[i : i+j])
			ump[sub]++
		}
	}

	// 处理查询 - 使用buffered writer
	out := bufio.NewWriter(os.Stdout)
	defer out.Flush()

	// 复用缓冲区
	outputBuf := make([]byte, 0, 32)
	
	for i := 0; i < q; i++ {
		scanner.Scan()
		query := scanner.Text()
		
		if count, exists := ump[query]; exists {
			// 快速整数转字符串并输出
			outputBuf = outputBuf[:0]
			outputBuf = strconv.AppendInt(outputBuf, int64(count), 10)
			outputBuf = append(outputBuf, '\n')
			out.Write(outputBuf)
		} else {
			out.WriteString("0\n")
		}
	}
}
```
