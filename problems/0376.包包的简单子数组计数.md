# 脑筋急转弯
[题目链接](https://kamacoder.com/problempage.php?pid=1456)
## 思路分析
这道题的核心是找出满足向下取整的平均值等于全数组所有可能子数组平均值中的最大值 v 的子数组数量。

首先思考一个数学性质：一个数组的平均值，永远不可能超过这个数组中的最大元素。因此，所有子数组中平均值的最大值，一定就等于整个数组中最大的那个元素（假设最大元素为 max_val，只包含这个元素的子数组长度为1，平均值即为 max_val）。也就是说，题目要求的 v = max_val。

接下来思考：怎样的子数组，其平均值向下取整后能等于 max_val 呢？
如果一个子数组中包含了哪怕一个比 max_val 小的元素，那么这个子数组的元素总和一定严格小于 长度 * max_val。它的平均值就一定严格小于 max_val。因为全是整数，一个严格小于 max_val 的数字向下取整，最大只能是 max_val - 1，不可能等于 max_val。

因此，**只有当一个子数组里面的所有元素都等于 max_val 时，它的平均值向下取整才会等于 max_val**。

这样一来，问题就得到了简化：
1. 遍历一遍数组，找出数组中的最大值 max_val；
2. 找到所有连续的且值等于 max_val 的元素区间。假设某一段连续区间长度为 L，那么这段区间内能够形成的符合条件的子数组数量为 L * (L + 1) / 2；
3. 将所有这样连续区间的贡献累加即可。

注意点：需要用long long保存答案。

## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(1)。

## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    // 优化输入输出流
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;
    if (!(cin >> n)) return 0;
    
    vector<int> a(n);
    int max_val = -1;
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
        if (a[i] > max_val) {
            max_val = a[i];
        }
    }
    
    long long ans = 0;
    long long len = 0;
    
    for (int i = 0; i < n; ++i) {
        if (a[i] == max_val) {
            len++;
        } else {
            // 当遇到不等于最大值的元素时，结算之前的连续段
            ans += len * (len + 1) / 2;
            len = 0;
        }
    }
    // 加上最后一段可能以 max_val 结尾的区间
    ans += len * (len + 1) / 2;
    
    cout << ans << "\n";
    return 0;
}
```

### Python
``` python
import sys

def main():
    # 使用 sys.stdin.read 优化读取速度
    input_data = sys.stdin.read().split()
    n = int(input_data[0])
    a = [int(x) for x in input_data[1:n+1]]
    
    max_val = max(a)
    ans = 0
    length = 0
    
    for num in a:
        if num == max_val:
            length += 1
        else:
            # 结算区间，Python中大整数不会溢出
            ans += length * (length + 1) // 2
            length = 0
            
    ans += length * (length + 1) // 2
    
    print(ans)

if __name__ == '__main__':
    main()
```

### Java
``` java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.StreamTokenizer;

public class Main {
    public static void main(String[] args) throws Exception {
        // 使用 StreamTokenizer 优化读取速度
        StreamTokenizer in = new StreamTokenizer(new BufferedReader(new InputStreamReader(System.in)));
        
        if (in.nextToken() == StreamTokenizer.TT_EOF) return;
        int n = (int) in.nval;
        
        int[] a = new int[n];
        int max_val = -1;
        for (int i = 0; i < n; i++) {
            in.nextToken();
            a[i] = (int) in.nval;
            if (a[i] > max_val) {
                max_val = a[i];
            }
        }
        
        long ans = 0;
        long len = 0; // 使用 long 防止乘法时溢出
        
        for (int i = 0; i < n; i++) {
            if (a[i] == max_val) {
                len++;
            } else {
                ans += len * (len + 1) / 2;
                len = 0;
            }
        }
        ans += len * (len + 1) / 2;
        
        System.out.println(ans);
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
)

func main() {
    // 使用 Scanner 优化读取速度
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)
    // 增加缓冲区以防单行过长
	buf := make([]byte, 1024*1024)
	scanner.Buffer(buf, 1024*1024*10)

	if !scanner.Scan() {
		return
	}
	n, _ := strconv.Atoi(scanner.Text())

	a := make([]int, n)
	max_val := -1
	for i := 0; i < n; i++ {
		scanner.Scan()
		a[i], _ = strconv.Atoi(scanner.Text())
		if a[i] > max_val {
			max_val = a[i]
		}
	}

	var ans int64 = 0
	var length int64 = 0

	for i := 0; i < n; i++ {
		if a[i] == max_val {
			length++
		} else {
			ans += length * (length + 1) / 2
			length = 0
		}
	}
	ans += length * (length + 1) / 2

	fmt.Println(ans)
}
```
