## 思路分析

利用一个二维数组存储当前所有正在递减的子序列，每次遇到一个新元素的时候就与所有递减的子序列的最后一个元素进行比较。如果这个子序列的最后一个元素比当前元素大，那么就证明它可以插入到这个子序列中。在这期间会遇到三种情况：

1. 有一个子序列满足。这种情况直接插入到这个子序列末尾即可。
2. 没有子序列满足。直接新增一个子序列，将这个元素作为第一个元素。
3. 有多个子序列满足。这种情况插入到第一个满足的子序列末尾即可。假设最开始只有一个子序列，那么我们在依次匹配的时候，如果当前元素等于这个子序列末尾元素的时候，就不满足题意，这个时候，由于只有一个子序列，所以需要新增一个子序列。所以后续的所有的子序列末尾元素都一定大于等于前面的子序列的末尾元素。而我们肯定尽量往大于当前元素的最小的元素后面放，才能放的尽可能的多。

## 复杂度分析
1. 时间复杂度：O(n ^ 2)；
2. 空间复杂度：O(n)。

``` cpp
#include <bits/stdc++.h>
#define endl '\n'
const int N = 500000, MOD = 1000000007;
using namespace std;
using LL = long long;
using PIL = pair<int, LL>;
using PII = pair<int, int>;
using T3I = tuple<int, int, int>;
int t = 0, dir[]{1, 0, -1, 0, 1};
int main() {
    cin.tie(nullptr) -> sync_with_stdio(false);
    string s, temp;
    vector<int> nums;
    while (getline(cin, temp, ',')) {
        nums.push_back(stoi(temp));
    }
    int n = nums.size();
    vector<vector<int>> ans;
    ans.push_back({nums[0]});
    for (int i = 1; i < n; ++i) {
        int idx = -1; LL idxv = 2147483648;  // 表示最小的子序列号和最小的子序列末尾元素
        int m = ans.size();
        for (int j = 0; j < m; ++j) {
            auto &v = ans[j];
            if (v.back() > nums[i] && idxv > v.back()) idx = j, idxv = v.back();
        }
        if (idx == -1) ans.push_back({nums[i]});
        else ans[idx].push_back(nums[i]);
    }
    cout << ans.size() << endl;
    for (auto &v : ans) {
        int m = v.size();
        for (int j = 0; j < m - 1; ++j) cout << v[j] << ',';
        cout << v.back() << endl;
    }
    return 0;
}
```

### Python
``` python
import sys

def main():
    data = sys.stdin.read().strip()
    if not data:
        return
    nums = list(map(int, data.split(',')))

    ans = [[nums[0]]]

    for i in range(1, len(nums)):
        idx = -1
        idxv = 1 << 31  # 2147483648
        for j in range(len(ans)):
            if ans[j][-1] > nums[i] and idxv > ans[j][-1]:
                idx = j
                idxv = ans[j][-1]
        if idx == -1:
            ans.append([nums[i]])
        else:
            ans[idx].append(nums[i])

    print(len(ans))
    for v in ans:
        print(','.join(map(str, v)))

if __name__ == "__main__":
    main()
```

### Java
``` java
import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        String line = br.readLine();
        if (line == null || line.isEmpty()) return;

        String[] parts = line.split(",");
        int[] nums = new int[parts.length];
        for (int i = 0; i < parts.length; i++) {
            nums[i] = Integer.parseInt(parts[i]);
        }

        List<List<Integer>> ans = new ArrayList<>();
        ans.add(new ArrayList<>(List.of(nums[0])));

        for (int i = 1; i < nums.length; i++) {
            int idx = -1;
            long idxv = 2147483648L; // 2^31
            for (int j = 0; j < ans.size(); j++) {
                List<Integer> v = ans.get(j);
                if (v.get(v.size() - 1) > nums[i] && idxv > v.get(v.size() - 1)) {
                    idx = j;
                    idxv = v.get(v.size() - 1);
                }
            }
            if (idx == -1) {
                List<Integer> newList = new ArrayList<>();
                newList.add(nums[i]);
                ans.add(newList);
            } else {
                ans.get(idx).add(nums[i]);
            }
        }

        System.out.println(ans.size());
        for (List<Integer> v : ans) {
            StringBuilder sb = new StringBuilder();
            for (int j = 0; j < v.size(); j++) {
                if (j > 0) sb.append(',');
                sb.append(v.get(j));
            }
            System.out.println(sb);
        }
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	line, _ := reader.ReadString('\n')
	line = strings.TrimSpace(line)
	if line == "" {
		return
	}

	parts := strings.Split(line, ",")
	nums := make([]int, len(parts))
	for i, p := range parts {
		nums[i], _ = strconv.Atoi(p)
	}

	ans := make([][]int, 0)
	ans = append(ans, []int{nums[0]})

	for i := 1; i < len(nums); i++ {
		idx := -1
		idxv := int64(2147483648) // 2^31
		for j := 0; j < len(ans); j++ {
			v := ans[j]
			if v[len(v)-1] > nums[i] && idxv > int64(v[len(v)-1]) {
				idx = j
				idxv = int64(v[len(v)-1])
			}
		}
		if idx == -1 {
			ans = append(ans, []int{nums[i]})
		} else {
			ans[idx] = append(ans[idx], nums[i])
		}
	}

	fmt.Println(len(ans))
	for _, v := range ans {
		strVals := make([]string, len(v))
		for j, val := range v {
			strVals[j] = strconv.Itoa(val)
		}
		fmt.Println(strings.Join(strVals, ","))
	}
}
```
