# 前缀和+有序集合
[题目链接](https://kamacoder.com/problempage.php?pid=1453)
## 思路分析

这道题需要为每个长度 m (1 <= m <= n) 找到在最多执行 n-m 次删除操作后的最小代价。代价的计算规则为：`删除次数 * k + 剩余数组的前 m 项之和`。

由于我们在剩余数组中只对 **前 m 项** 求和，且每次删除需支付正数代价 k >= 1，为了让总代价最小，我们可以得出一个极其重要的结论：**在第 m 个保留项之后的所有原数组元素，我们完全不需要去删除它们**。因为它们会被排在新数组的第 m 项之后，既不参与前 m 项的求和，保留它们也不会增加我们的删除成本。

于是问题就等价于：在原数组中选择 m 个元素作为新数组的前 m 项。

假设我们确定的这 m 个元素里，最后一个（也就是第 m 个）元素在原数组中的位置是 x 。很显然，x 的取值范围必须满足 m <= x <= n。
为了让原数组第 x 个元素刚好成为新数组的第 m 项，我们必须：
1. 保留元素 a[x]。
2. 在 a[x] 之前（即原数组的前 x-1 个元素中），挑出 m-1 个元素予以保留。
3. 那么前 x-1 个元素中，剩下的那部分也就是 `(x - 1) - (m - 1) = x - m` 个元素，就必须被强制删除。

这样，一旦我们固定了第 m 个保留项在原数组中的位置 x，代价公式就完全确定了：

```
总代价 = (原数组前 x-1 个元素中选出的 m-1 个元素的和) + a[x] + (x - m) * k
```

为了让这个总代价最小，在确定了位置 x 之后，我们理应贪心地在原数组前 x-1 个元素中选出 **最小的 m-1 个元素**。

基于以上规律，我们可以设计如下算法：
1. 从左到右遍历 x，取值范围从 1 到 n，也就是逐个枚举第 m 项在原数组中的位置。
2. 维护一个动态的、从小到大排序的数组 `sorted_a`，用来记录 a[1] 到 a[x-1] 所有的元素。
3. 借助这个排好序的 `sorted_a`，我们可以通过计算前缀和，非常快速地获取“最小的 m-1 个元素的和”。
4. 对于当前枚举的每一个 x，它都有能力作为长度为 1 到 x 的序列的结尾。所以我们再套一个内层循环，枚举 m (1 <= m <= x)，并用上面推导的公式计算代价，如果比 `ans[m]` 现有的记录更小，就更新它。
5. 当一轮内层循环结束后，把当前的 a[x] 按照大小顺序插入到 `sorted_a` 中，保持其有序性，然后继续处理下一个 x。

## 复杂度分析
1. 时间复杂度：O(n^2)。对于每个位置 x，计算前缀和以及内层循环耗时 O(x)；向排序好的数组中插入元素也耗时 O(x)。整体处理一个测试用例耗费 O(n^2) 的时间。由于题目保证单个测试文件所有用例的 n 之和不超过 1000，最坏情况下运算量在 10^6 级别，完全不会超时。
2. 空间复杂度：O(n)。我们仅额外开辟了长度不超过 n 的排序数组 `sorted_a`、前缀和数组 `pref` 与答案数组 `ans`，占用的额外空间很小。

## 代码实现

### Cpp
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

void solve() {
    int n;
    long long k;
    cin >> n >> k;
    vector<long long> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    // ans 数组用于记录各个 m 下的最优解，初始赋予极大值
    vector<long long> ans(n + 1, 2e18); 
    vector<long long> sorted_a;

    for (int x = 1; x <= n; ++x) {
        // 计算 sorted_a 的前缀和，便于获取前 m-1 小的元素之和
        vector<long long> pref(sorted_a.size() + 1, 0);
        for (int i = 0; i < sorted_a.size(); ++i) {
            pref[i + 1] = pref[i] + sorted_a[i];
        }

        // x 作为保留的第 m 项，m 最大只能到 x
        for (int m = 1; m <= x; ++m) {
            long long cost = pref[m - 1] + a[x - 1] + (x - m) * k;
            if (cost < ans[m]) {
                ans[m] = cost;
            }
        }

        // 把当前元素 a[x-1] 插入 sorted_a 维持有序
        auto it = lower_bound(sorted_a.begin(), sorted_a.end(), a[x - 1]);
        sorted_a.insert(it, a[x - 1]);
    }

    for (int m = 1; m <= n; ++m) {
        cout << ans[m] << (m == n ? "" : " ");
    }
    cout << "\n";
}

int main() {
    // 优化标准输入输出
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t;
    if (cin >> t) {
        while (t--) {
            solve();
        }
    }
    return 0;
}
```

### Python
```python
import sys
from bisect import bisect_left

def main():
    input_data = sys.stdin.read().split()
    
    T = int(input_data[0])
    idx = 1
    out = []
    
    for _ in range(T):
        n = int(input_data[idx])
        k = int(input_data[idx+1])
        idx += 2
        
        a = [int(x) for x in input_data[idx : idx + n]]
        idx += n
        
        # ans 数组用于记录各个 m 下的最优解，初始赋予极大值
        ans = [float('inf')] * (n + 1)
        sorted_a = []
        
        for x in range(1, n + 1):
            val = a[x - 1]
            
            # 计算前缀和，获取前 m-1 小的元素之和
            pref = [0] * (len(sorted_a) + 1)
            for i in range(len(sorted_a)):
                pref[i + 1] = pref[i] + sorted_a[i]
                
            for m in range(1, x + 1):
                cost = pref[m - 1] + val + (x - m) * k
                if cost < ans[m]:
                    ans[m] = cost
                    
            # 二分查找插入点，维护有序状态
            pos = bisect_left(sorted_a, val)
            sorted_a.insert(pos, val)
            
        out.append(" ".join(str(ans[m]) for m in range(1, n + 1)))
        
    print("\n".join(out))

if __name__ == '__main__':
    main()
```

### Java
```java
import java.io.*;
import java.util.*;

public class Main {
    // 快速读取工具类
    static class FastReader {
        BufferedReader br;
        StringTokenizer st;

        public FastReader() {
            br = new BufferedReader(new InputStreamReader(System.in));
        }

        String next() {
            while (st == null || !st.hasMoreElements()) {
                try {
                    String line = br.readLine();
                    if(line == null) return null;
                    st = new StringTokenizer(line);
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            return st.nextToken();
        }
        
        int nextInt() {
            return Integer.parseInt(next());
        }

        long nextLong() {
            return Long.parseLong(next());
        }
    }

    public static void main(String[] args) {
        FastReader in = new FastReader();
        String tStr = in.next();
        if (tStr == null) return;
        int T = Integer.parseInt(tStr);
        
        for (int t = 0; t < T; t++) {
            int n = in.nextInt();
            long k = in.nextLong();
            
            long[] a = new long[n];
            for (int i = 0; i < n; i++) {
                a[i] = in.nextLong();
            }
            
            long[] ans = new long[n + 1];
            Arrays.fill(ans, Long.MAX_VALUE);
            
            List<Long> sortedA = new ArrayList<>();
            for (int x = 1; x <= n; x++) {
                // 计算前缀和
                long[] pref = new long[sortedA.size() + 1];
                for (int i = 0; i < sortedA.size(); i++) {
                    pref[i + 1] = pref[i] + sortedA.get(i);
                }
                
                for (int m = 1; m <= x; m++) {
                    long cost = pref[m - 1] + a[x - 1] + (x - m) * k;
                    if (cost < ans[m]) {
                        ans[m] = cost;
                    }
                }
                
                // 二分查找插入以维护排序
                int pos = Collections.binarySearch(sortedA, a[x - 1]);
                if (pos < 0) {
                    pos = -(pos + 1);
                }
                sortedA.add(pos, a[x - 1]);
            }
            
            StringBuilder sb = new StringBuilder();
            for (int m = 1; m <= n; m++) {
                sb.append(ans[m]).append(m == n ? "" : " ");
            }
            System.out.println(sb.toString());
        }
    }
}
```

### Go
```go
package main

import (
	"bufio"
	"fmt"
	"os"
	"sort"
	"strconv"
)

func main() {
    // 扩展缓冲区大小
	scanner := bufio.NewScanner(os.Stdin)
	const maxCapacity = 1024 * 1024 * 10
	buf := make([]byte, maxCapacity)
	scanner.Buffer(buf, maxCapacity)
	scanner.Split(bufio.ScanWords)

	if !scanner.Scan() {
		return
	}
	T, _ := strconv.Atoi(scanner.Text())

	for t := 0; t < T; t++ {
		scanner.Scan()
		n, _ := strconv.Atoi(scanner.Text())
		scanner.Scan()
		k, _ := strconv.ParseInt(scanner.Text(), 10, 64)

		a := make([]int64, n)
		for i := 0; i < n; i++ {
			scanner.Scan()
			a[i], _ = strconv.ParseInt(scanner.Text(), 10, 64)
		}

		ans := make([]int64, n+1)
		for i := 1; i <= n; i++ {
			ans[i] = 2e18 // 设定为远超可能最大代价的极大值
		}

		var sortedA []int64
		for x := 1; x <= n; x++ {
            // 计算前缀和
			pref := make([]int64, len(sortedA)+1)
			for i := 0; i < len(sortedA); i++ {
				pref[i+1] = pref[i] + sortedA[i]
			}

			val := a[x-1]
			for m := 1; m <= x; m++ {
				cost := pref[m-1] + val + int64(x-m)*k
				if cost < ans[m] {
					ans[m] = cost
				}
			}

			// 查找插入点保证切片有序状态
			pos := sort.Search(len(sortedA), func(i int) bool { return sortedA[i] >= val })
			sortedA = append(sortedA, 0)
			copy(sortedA[pos+1:], sortedA[pos:])
			sortedA[pos] = val
		}

		for m := 1; m <= n; m++ {
			if m > 1 {
				fmt.Print(" ")
			}
			fmt.Print(ans[m])
		}
		fmt.Println()
	}
}
```
