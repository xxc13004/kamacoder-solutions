# 动态规划
[题目链接](https://kamacoder.com/problempage.php?pid=1430)
## 思路分析

先对数组进行求和，设其和为m，如果m是奇数，那么肯定不能满足分割成两部分。如果为偶数，我们就用背包来求解。我们将k = m / 2，k就是背包的容量，如果最后dp[k]为1，那么就说明数组里面的数，能够组成当前和的一半，必定满足题意。

**状态定义**：定义dp[i]为当前元素能凑成和为i的方案的数量，由于我们不关心具体方案数，只关心存不存在，所以这里的取值只有0和1。

**状态转移**：设当前遍历到的元素为x，当前背包容量为j，我们可以考虑选或不选x，如果选，那么就要从dp[j-x]进行转移。如果dp[j-x]为1，那么dp[j]就是1，如果dp[j-x]为0，那么dp[j]保持不变。初始时刻，dp[0] = 1。

**转移方程**：

1. dp[j] = dp[j - x] || dp[j] if j >= x。

## 复杂度分析
1. 时间复杂度：O(nk)；
2. 空间复杂度：O(k)。

``` cpp
    if (sum % 2 != 0) {  // 不能被2整除，就不行
        cout << "false" << endl;
        return 0;
    }
    sum /= 2;
    vector<int> dp(sum + 1);
    dp[0] = 1;
    for (int i = 0; i < n; ++i) {
        for (int j = sum; j >= nums[i]; --j) {
            dp[j] = dp[j] || dp[j - nums[i]];
        }
    }
    cout << (dp[sum] ? "true" : "false") << endl;
    return 0;
}
```

### Python
``` python
import sys

def main():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    n = int(input_data[0])
    nums = list(map(int, input_data[1:1+n]))
    
    total_sum = sum(nums)
    
    # 如果总和是奇数，不可能分成相等的两部分
    if total_sum % 2 != 0:
        print("false")
        return
    
    target = total_sum // 2
    
    # 使用bitset优化（Python的int可以当作bitset用）
    dp = 1 << 0  # 第0位设为1，表示和为0是可达的
    
    for num in nums:
        # 左移num位然后与原来的dp进行或操作
        dp = dp | (dp << num)
    
    # 检查target位是否为1
    if (dp >> target) & 1:
        print("true")
    else:
        print("false")

if __name__ == "__main__":
    main()
```
### Java
``` java
import java.util.*;
import java.io.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        PrintWriter pw = new PrintWriter(new BufferedWriter(new OutputStreamWriter(System.out)));
        
        int n = Integer.parseInt(br.readLine().trim());
        int[] nums = new int[n];
        
        StringTokenizer st = new StringTokenizer(br.readLine());
        int sum = 0;
        for (int i = 0; i < n; i++) {
            nums[i] = Integer.parseInt(st.nextToken());
            sum += nums[i];
        }
        
        // 如果总和是奇数，不可能分成相等的两部分
        if (sum % 2 != 0) {
            pw.println("false");
            pw.flush();
            return;
        }
        
        int target = sum / 2;
        boolean[] dp = new boolean[target + 1];
        dp[0] = true;
        
        // 0-1背包问题
        for (int i = 0; i < n; i++) {
            // 逆序更新，确保每个数字只用一次
            for (int j = target; j >= nums[i]; j--) {
                dp[j] = dp[j] || dp[j - nums[i]];
            }
        }
        
        pw.println(dp[target] ? "true" : "false");
        pw.flush();
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
)

func main() {
	in := bufio.NewReader(os.Stdin)
	out := bufio.NewWriter(os.Stdout)
	defer out.Flush()
	
	var n int
	fmt.Fscan(in, &n)
	
	nums := make([]int, n)
	sum := 0
	
	for i := 0; i < n; i++ {
		fmt.Fscan(in, &nums[i])
		sum += nums[i]
	}
	
	// 如果总和是奇数，不可能分成相等的两部分
	if sum%2 != 0 {
		fmt.Fprintln(out, "false")
		return
	}
	
	target := sum / 2
	
	// 使用bitset优化，大幅减少内存使用
	dp := make([]bool, target+1)
	dp[0] = true
	
	// 0-1背包问题
	for i := 0; i < n; i++ {
		num := nums[i]
		// 逆序更新，确保每个数字只用一次
		for j := target; j >= num; j-- {
			if dp[j-num] {
				dp[j] = true
			}
		}
	}
	
	if dp[target] {
		fmt.Fprintln(out, "true")
	} else {
		fmt.Fprintln(out, "false")
	}
}
```
