# 组合数学
[题目链接](https://kamacoder.com/problempage.php?pid=1454)

## 思路分析
这道题目要求我们找回丢失的数据 `0` 并且保持整个数组是非严格递增的。
题目中保证了数组的首尾 $a_1$ 和 $a_n$ 不为 0。所以每遇到一段连续丢失的数据（连续的 `0`），它们必然处在两个非 `0` 的数字之间。假设有一段连续的 `0`，它的前一个非零数字是 $a_{l-1}$，后一个非零数字是 $a_{r+1}$，中间共有 $k$ 个 `0`。

为了使填充后的数组保持非严格递增，我们需要在 $[a_{l-1}, a_{r+1}]$ 的范围内选出 $k$ 个数字（允许重复），并保证它们是非严格递增排列。
这里可以用到组合数学中的**隔板法（Stars and Bars）**。其等价于从集合的数字中无序选取 $k$ 个允许重复的元素。
可用数字的种类数为 $m = a_{r+1} - a_{l-1} + 1$。
因此，方案数为组合数：
$$ C(m + k - 1, k) = \frac{(m+k-1)(m+k-2)\dots(m)}{k!} $$

每一段连续的 `0` 所得出的方案数是相互独立的，根据乘法原理，将每一段的组合数相乘，并对 $10^9+7$ 取模，就是最终的总方案数。

为了提高计算效率，分母 $k!$ 我们可以预处理得到阶乘的模逆元；由于每一段的 $k$ 的总和不超过 $N$，分子直接用 $k$ 次循环累乘即可完成计算。

## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(n)。

## 代码实现

### Cpp
```cpp
#include <iostream>
#include <vector>

using namespace std;

const int MOD = 1e9 + 7;

// 快速幂计算逆元
long long power(long long base, long long exp) {
    long long res = 1;
    base %= MOD;
    while (exp > 0) {
        if (exp % 2 == 1) res = (res * base) % MOD;
        base = (base * base) % MOD;
        exp /= 2;
    }
    return res;
}

int main() {
    // 优化输入输出流
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;

    vector<long long> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    // 特判: 如果给出的非零元素已经违反了非严格递增规律，直接输出 0
    long long last_val = -1;
    for(int i = 0; i < n; i++){
        if(a[i] != 0){
            if(last_val != -1 && a[i] < last_val){
                cout << 0 << "\n";
                return 0;
            }
            last_val = a[i];
        }
    }

    // 预处理阶乘逆元
    vector<long long> invFact(n + 1);
    long long fact = 1;
    for (int i = 1; i <= n; i++) {
        fact = (fact * i) % MOD;
    }
    invFact[n] = power(fact, MOD - 2);
    for (int i = n - 1; i >= 0; i--) {
        invFact[i] = (invFact[i + 1] * (i + 1)) % MOD;
    }
    invFact[0] = 1;

    long long total_ans = 1;
    int i = 0;
    while (i < n) {
        if (a[i] == 0) {
            int j = i;
            while (j < n && a[j] == 0) {
                j++;
            }
            // 取前后确定的非零元素进行组合数学计算
            long long prev = a[i - 1];
            long long next_val = a[j];
            int k = j - i;
            long long m = next_val - prev + 1;
            
            // 计算 C(m + k - 1, k)
            long long start = (m + k - 1) % MOD;
            long long num = 1; // 分子
            for (int step = 0; step < k; step++) {
                num = (num * start) % MOD;
                start = (start - 1 + MOD) % MOD; // 防止负数
            }
            long long combinations = (num * invFact[k]) % MOD;
            total_ans = (total_ans * combinations) % MOD;
            
            i = j;
        } else {
            i++;
        }
    }

    cout << total_ans << "\n";
    return 0;
}
```

```java
    static long power(long base, long exp) {
        long res = 1;

        while (exp > 0) {
            if (exp % 2 == 1) res = (res * base) % MOD;
            base = (base * base) % MOD;
            exp /= 2;

        return res;

    static class FastReader {
        BufferedReader br;
        StringTokenizer st;

        public FastReader() {
            br = new BufferedReader(new InputStreamReader(System.in));
        }

        String next() {
            while (st == null || !st.hasMoreElements()) {
                try {
                    String line = br.readLine();
                    if(line == null) return null;
                    st = new StringTokenizer(line);
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            return st.nextToken();
        }

        int nextInt() {
            return Integer.parseInt(next());
        }
        
        long nextLong() {
            return Long.parseLong(next());
        }
    }

    public static void main(String[] args) {
```

```go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
)

const MOD = 1000000007

func power(base, exp int64) int64 {
	var res int64 = 1
	base %= MOD
	for exp > 0 {
		if exp%2 == 1 {
			res = (res * base) % MOD
		}
		base = (base * base) % MOD
		exp /= 2
	}
	return res
}

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	const maxCapacity = 1024 * 1024 * 10 // 10MB Token Buffer
	buf := make([]byte, maxCapacity)
	scanner.Buffer(buf, maxCapacity)
	scanner.Split(bufio.ScanWords)

	if !scanner.Scan() {
		return
	}
	n, _ := strconv.Atoi(scanner.Text())

	a := make([]int64, n)
	for i := 0; i < n; i++ {
		scanner.Scan()
		a[i], _ = strconv.ParseInt(scanner.Text(), 10, 64)
	}

	var last_val int64 = -1
	for i := 0; i < n; i++ {
		if a[i] != 0 {
			if last_val != -1 && a[i] < last_val {
				fmt.Println(0)
				return
			}
			last_val = a[i]
		}
	}

	invFact := make([]int64, n+1)
	var fact int64 = 1
	for i := 1; i <= n; i++ {
		fact = (fact * int64(i)) % MOD
	}
	invFact[n] = power(fact, MOD-2)
	for i := n - 1; i >= 0; i-- {
		invFact[i] = (invFact[i+1] * int64(i+1)) % MOD
	}
	invFact[0] = 1

	var total_ans int64 = 1
	i := 0
	for i < n {
		if a[i] == 0 {
			j := i
			for j < n && a[j] == 0 {
				j++
			}
			prev := a[i-1]
			next_val := a[j]
			k := j - i
			m := next_val - prev + 1

			start := (m + int64(k) - 1) % MOD
			var num int64 = 1
			for step := 0; step < k; step++ {
				num = (num * start) % MOD
				start = (start - 1 + MOD) % MOD
			}

			combinations := (num * invFact[k]) % MOD
			total_ans = (total_ans * combinations) % MOD

			i = j
		} else {
			i++
		}
	}

	fmt.Println(total_ans)
}
```
