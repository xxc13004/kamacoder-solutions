# 模拟
[题目链接](https://kamacoder.com/problempage.php?pid=1357)
## 思路分析
我们只要知道每两个相邻的时间之间经过了多少秒即可计算答案，为了方便计算每个时间的之间经过的秒数，我们只需要将每个时间转换为距00:00:00这个时间点经过的秒数就可以了。

1. 如果下一个时间的秒数大于当前的相对秒数，那么直接用下一个秒数减掉当前的秒数即可；
2. 如果下一个时间的秒数小于当前的相对秒数，那么就需要先用一整天时间的秒数减去这两个时间的相对秒数差值的绝对值就行。

## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(1)。

``` cpp
#include<bits/stdc++.h>
#define endl '\n'
const int N = 100005, MOD = 1000000007;
using namespace std;
using LL = long long;
using PIL = pair<int, LL>;
using PII = pair<int, int>;
using T3I = tuple<int, int, int>;

constexpr int day = 24 * 3600;
int main() {
    cin.tie(nullptr) -> sync_with_stdio(false);
    int n;
    cin >> n;
    vector<int> times(n);
    // 将每个时间解析成秒来表示
    for (int i = 0; i < n; ++i) {
        string s, time;
        cin >> s;
        stringstream ss(s);
        int second = 0, base = 3600;
        while (getline(ss, time, ':')) {
            second += stoi(time) * base;
            base /= 60;
        }
        times[i] = second;
    }
    for (int i = 1; i < n; ++i) {
        int t = times[i] - times[i - 1];
        if (t >= 0) {
            printf("%.2f ", t / 60.0);
        } else {
            printf("%.2f ", (day + t) / 60.0);
        }
    }
    return 0;
}
```

### Python
``` python
import sys

DAY = 24 * 3600

def main():
    input = sys.stdin.read().split()
    n = int(input[0])
    time_strings = input[1:n+1]
    times = []
    
    for ts in time_strings:
        h, m, s = map(int, ts.split(':'))
        times.append(h * 3600 + m * 60 + s)
    
    output = []
    for i in range(1, n):
        t = times[i] - times[i-1]
        if t >= 0:
            output.append("%.2f" % (t / 60))
        else:
            output.append("%.2f" % ((DAY + t) / 60))
    
    print(' '.join(output))

if __name__ == "__main__":
    main()
```

### Java
``` java
import java.io.*;
import java.util.*;

public class Main {
    static final int DAY = 24 * 3600;
    
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int n = Integer.parseInt(br.readLine());
        String[] timeStrings = br.readLine().trim().split(" ");
        int[] times = new int[n];
        
        for (int i = 0; i < n; i++) {
            String[] parts = timeStrings[i].split(":");
            int seconds = Integer.parseInt(parts[0]) * 3600;
            seconds += Integer.parseInt(parts[1]) * 60;
            seconds += Integer.parseInt(parts[2]);
            times[i] = seconds;
        }
        
        StringBuilder sb = new StringBuilder();
        for (int i = 1; i < n; i++) {
            int t = times[i] - times[i-1];
            if (t >= 0) {
                sb.append(String.format("%.2f ", t / 60.0));
            } else {
                sb.append(String.format("%.2f ", (DAY + t) / 60.0));
            }
        }
        System.out.println(sb.toString().trim());
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"bytes"
	"fmt"
	"os"
	"strconv"
	"strings"
)

const day = 24 * 3600

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)

	// Read n
	scanner.Scan()
	n, _ := strconv.Atoi(scanner.Text())

	// Read time strings
	timeStrings := make([]string, n)
	for i := 0; i < n; i++ {
		scanner.Scan()
		timeStrings[i] = scanner.Text()
	}

	// Parse times
	times := make([]int, n)
	for i, ts := range timeStrings {
		parts := strings.Split(ts, ":")
		h, _ := strconv.Atoi(parts[0])
		m, _ := strconv.Atoi(parts[1])
		s, _ := strconv.Atoi(parts[2])
		times[i] = h*3600 + m*60 + s
	}

	// Calculate and print differences
	var buf bytes.Buffer
	for i := 1; i < n; i++ {
		t := times[i] - times[i-1]
		if t >= 0 {
			fmt.Fprintf(&buf, "%.2f ", float64(t)/60.0)
		} else {
			fmt.Fprintf(&buf, "%.2f ", float64(day+t)/60.0)
		}
	}
	result := buf.String()
	fmt.Println(result[:len(result)-1]) // Remove trailing space
}
```
