## 思路分析
按照题意模拟即可。注意c++读取字符串的方式，用getline或者字符数组配合getchar使用需要手动过滤掉回车。
## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(1)。

### Python
``` python
import sys

def upper_char(x):
    if 'A' <= x <= 'Z':
        return x
    return chr(ord(x) - 32)

def lower_char(x):
    if 'a' <= x <= 'z':
        return x
    return chr(ord(x) + 32)

def main():
    n = int(sys.stdin.readline())
    s = sys.stdin.readline().rstrip('\n\r')
    t = sys.stdin.readline().rstrip('\n\r')
    
    result = []
    for i in range(n):
        if 'A' <= s[i] <= 'Z':
            result.append(upper_char(t[i]))
        elif 'a' <= s[i] <= 'z':
            result.append(lower_char(t[i]))
        elif '0' <= s[i] <= '9':
            result.append(str(ord(t[i])))
        else:
            result.append('_')
    
    print(''.join(result))

if __name__ == "__main__":
    main()
```

### Java
``` java
import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        
        int n = Integer.parseInt(br.readLine());
        String s = br.readLine();
        String t = br.readLine();
        
        StringBuilder result = new StringBuilder();
        for (int i = 0; i < n; i++) {
            char sc = s.charAt(i);
            char tc = t.charAt(i);
            
            if (sc >= 'A' && sc <= 'Z') {
                result.append(upperChar(tc));
            } else if (sc >= 'a' && sc <= 'z') {
                result.append(lowerChar(tc));
            } else if (sc >= '0' && sc <= '9') {
                result.append((int) tc);
            } else {
                result.append('_');
            }
        }
        
        System.out.println(result.toString());
    }
    
    private static char upperChar(char x) {
        if (x >= 'A' && x <= 'Z') return x;
        return (char) (x - 32);
    }
    
    private static char lowerChar(char x) {
        if (x >= 'a' && x <= 'z') return x;
        return (char) (x + 32);
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
)

func main() {
	const bufferSize = 1 << 20 // 1MB缓冲区
	reader := bufio.NewReaderSize(os.Stdin, bufferSize)
	writer := bufio.NewWriterSize(os.Stdout, bufferSize)
	defer writer.Flush()

	var n int
	fmt.Fscan(reader, &n)
	
	// 跳过换行符
	reader.ReadByte()
	
	// 直接读取两个字符串，避免中间转换
	s, _ := reader.ReadString('\n')
	t, _ := reader.ReadString('\n')
	
	// 流式处理，避免大内存分配
	for i := 0; i < n; i++ {
		sc := s[i]
		tc := t[i]
		
		switch {
		case sc >= 'A' && sc <= 'Z':
			if tc >= 'a' && tc <= 'z' {
				writer.WriteByte(tc - 32)
			} else {
				writer.WriteByte(tc)
			}
		case sc >= 'a' && sc <= 'z':
			if tc >= 'A' && tc <= 'Z' {
				writer.WriteByte(tc + 32)
			} else {
				writer.WriteByte(tc)
			}
		case sc >= '0' && sc <= '9':
			// 直接写入数字，避免字符串转换
			num := strconv.Itoa(int(tc))
			writer.WriteString(num)
		default:
			writer.WriteByte('_')
		}
	}
}
```
