# 枚举 哈希表
[题目链接](https://kamacoder.com/problempage.php?pid=1427)
## 思路分析

由于数组内的元素和查询值都在[1, 1e6]的范围内，所以可以直接枚举每个查询值x的所有因子对(a, b)是否在数组中。

用一个哈希表cnt统计数组中每个数字出现的次数。

1. 对于因子对(a, b)，即a * b = x；
2. 如果a == b，那么需要cnt[a] >= 2；
3. 对于a != b，那么需要cnt[a] >= 1 && cnt[b] >= 1。

对于每个x，只需要枚举到sqrt(x)即可。另外可以用数组替代哈希表，提高访问速度。

## 复杂度分析
1. 时间复杂度：O(n * sqrt(M)), M最大取值为1e6；
2. 空间复杂度：O(n)。
## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
using namespace std;

const int MAXN = 1000005;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    cin >> n;
    
    vector<int> cnt(MAXN, 0);
    vector<int> arr(n);
    
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
        cnt[arr[i]]++;
    }
    
    // 预处理：去重排序，便于后续查找
    sort(arr.begin(), arr.end());
    arr.erase(unique(arr.begin(), arr.end()), arr.end());
    
    int q;
    cin >> q;
    
    while (q--) {
        int x;
        cin >> x;
        
        bool found = false;
        
        // 枚举x的所有因子
        for (int a = 1; a * a <= x; a++) {
            if (x % a == 0) {
                int b = x / a;
                
                // 确保因子在有效范围内
                if (a > 1000000 || b > 1000000) {
                    continue;
                }
                
                // 检查是否存在这样的因子对
                if (a == b) {
                    // 需要至少出现两次
                    if (cnt[a] >= 2) {
                        found = true;
                        break;
                    }
                } else {
                    // 两个不同的数，都需要存在
                    if (cnt[a] > 0 && cnt[b] > 0) {
                        found = true;
                        break;
                    }
                }
            }
        }
        
        cout << (found ? "Yes" : "No") << "\n";
    }
    
    return 0;
}
```

### Python
``` python
import sys
import math

def main():
    input_data = sys.stdin.read().split()
    idx = 0
    
    n = int(input_data[idx]); idx += 1
    cnt = [0] * 1000005
    arr = []
    
    for i in range(n):
        num = int(input_data[idx]); idx += 1
        cnt[num] += 1
        arr.append(num)
    
    # 去重排序
    arr = sorted(set(arr))
    
    q = int(input_data[idx]); idx += 1
    
    out_lines = []
    for _ in range(q):
        x = int(input_data[idx]); idx += 1
        found = False
        
        # 枚举x的所有因子
        limit = int(math.sqrt(x))
        for a in range(1, limit + 1):
            if x % a == 0:
                b = x // a
                
                # 确保因子在有效范围内
                if a > 1000000 or b > 1000000:
                    continue
                
                # 检查是否存在这样的因子对
                if a == b:
                    # 需要至少出现两次
                    if cnt[a] >= 2:
                        found = True
                        break
                else:
                    # 两个不同的数，都需要存在
                    if cnt[a] > 0 and cnt[b] > 0:
                        found = True
                        break
        
        out_lines.append("Yes" if found else "No")
    
    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    main()
```

``` java
import java.util.*;
import java.io.*;

public class Main {
    static final int MAXN = 1000005;
    
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        PrintWriter pw = new PrintWriter(new BufferedWriter(new OutputStreamWriter(System.out)));
        
        int n = Integer.parseInt(br.readLine().trim());
        int[] cnt = new int[MAXN];
        int[] arr = new int[n];
        
        StringTokenizer st = new StringTokenizer(br.readLine());
        for (int i = 0; i < n; i++) {
            arr[i] = Integer.parseInt(st.nextToken());
            cnt[arr[i]]++;
        }
        
        // 去重排序
        Arrays.sort(arr);
        int uniqueSize = 0;
        for (int i = 0; i < n; i++) {
            if (i == 0 || arr[i] != arr[i-1]) {
                arr[uniqueSize++] = arr[i];
            }
        }
        
        int q = Integer.parseInt(br.readLine().trim());
        
        while (q-- > 0) {
            int x = Integer.parseInt(br.readLine().trim());
            boolean found = false;
            
            // 枚举x的所有因子
            for (int a = 1; a * a <= x; a++) {
                if (x % a == 0) {
                    int b = x / a;
                    
                    // 确保因子在有效范围内
                    if (a > 1000000 || b > 1000000) {
                        continue;
                    }
                    
                    // 检查是否存在这样的因子对
                    if (a == b) {
                        // 需要至少出现两次
                        if (cnt[a] >= 2) {
                            found = true;
                            break;
                        }
                    } else {
                        // 两个不同的数，都需要存在
                        if (cnt[a] > 0 && cnt[b] > 0) {
                            found = true;
                            break;
                        }
                    }
                }
            }
            
            pw.println(found ? "Yes" : "No");
        }
        
        pw.flush();
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"os"
)

const (
	MAXN   = 1000005
	BUFSIZ = 1 << 20 // 1MB
)

func main() {
	// 设置大缓冲区
	in := bufio.NewReaderSize(os.Stdin, BUFSIZ)
	out := bufio.NewWriterSize(os.Stdout, BUFSIZ)
	defer out.Flush()
	
	// 快速读取函数
	readInt := func() int {
		var num int
		var b byte
		var neg bool
		
		// 跳过空白字符
		for {
			b, _ = in.ReadByte()
			if b >= '0' && b <= '9' || b == '-' {
				break
			}
		}
		
		// 处理负号
		if b == '-' {
			neg = true
			b, _ = in.ReadByte()
		}
		
		// 读取数字
		for ; b >= '0' && b <= '9'; b, _ = in.ReadByte() {
			num = num*10 + int(b-'0')
		}
		
		if neg {
			num = -num
		}
		return num
	}
	
	// 读取n
	n := readInt()
	
	cnt := make([]int, MAXN)
	
	// 批量读取并统计
	for i := 0; i < n; i++ {
		num := readInt()
		if num < MAXN {
			cnt[num]++
		}
	}
	
	// 读取q
	q := readInt()
	
	// 处理查询
	for i := 0; i < q; i++ {
		x := readInt()
		found := false
		
		if x == 0 {
			// 处理0的特殊情况
			if cnt[0] > 0 {
				found = true
			}
		} else {
			// 优化：如果x很小，直接枚举所有可能因子
			if x <= 1000 {
				for a := 1; a*a <= x; a++ {
					if x%a == 0 {
						b := x / a
						if a < MAXN && b < MAXN {
							if a == b {
								if cnt[a] >= 2 {
									found = true
									break
								}
							} else if cnt[a] > 0 && cnt[b] > 0 {
								found = true
								break
							}
						}
					}
				}
			} else {
				// 对于大数，使用优化的因子枚举
				limit := intSqrt(x)
				for a := 1; a <= limit; a++ {
					if x%a == 0 {
						b := x / a
						if a < MAXN && b < MAXN {
							if a == b {
								if cnt[a] >= 2 {
									found = true
									break
								}
							} else if cnt[a] > 0 && cnt[b] > 0 {
								found = true
								break
							}
						}
					}
				}
			}
		}
		
		if found {
			out.WriteString("Yes\n")
		} else {
			out.WriteString("No\n")
		}
	}
}

// 优化的整数平方根函数
func intSqrt(n int) int {
	if n <= 1 {
		return n
	}
	
	// 使用位运算加速
	x := n
	y := (x + 1) >> 1
	for y < x {
		x = y
		y = (x + n/x) >> 1
	}
	return x
}
```
