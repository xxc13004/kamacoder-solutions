# 队列模拟 / 贪心分配
[题目链接](https://kamacoder.com/problempage.php?pid=1459)
## 思路分析
这道题的核心是利用**队列结构**来模拟副将的轮转，并且对每辆粮车使用**贪心策略**来尽可能多地满足连续的副将。

1. **若当前粮车 K ≥ 队首副将的需求 L1**：
   - 粮车足够满足队首，我们开始贪心地往后满足。
   - 记录当前已分配的粮食总量，不断弹出队首的副将，直到“已分配的粮食 + 下一个队首副将的需求量 > K”为止。
   - 分配完毕后，这辆车离开（下标 `car_idx + 1`），记录轮转次数的变量重置为 0。
2. **若当前粮车 K < 队首副将的需求 L1**：
   - 队首副将拿不到粮食，去队尾重新排队（出队并立即入队）。
   - 记录轮转的次数 `cycle_count` 加 1。
   - 如果 `cycle_count == 当前队列的长度`，说明排队的所有副将都看过了这辆粮车，且需求都大于粮车容量。此时没人能要这车粮食，这辆车直接被丢弃（离开营地），我们换下一辆粮车进来（下标 `car_idx + 1`，并将 `cycle_count` 重置为 0）。

## 复杂度分析
1. 时间复杂度：O(N * M)；其中 N 为粮车数量，M 为副将数量。最坏情况下，每辆粮车都可能被队列中现存的所有副将拒绝一圈才被丢弃，轮询耗时 O(M)。
2. 空间复杂度：O(M)。

## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <string>
#include <sstream>
#include <deque>

using namespace std;

int main() {
    // 优化输入输出
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    string line1, line2;
    if (!getline(cin, line1)) return 0;
    if (!getline(cin, line2)) return 0;

    vector<int> cars;
    deque<int> gens;
    int val;

    stringstream ss1(line1);
    while (ss1 >> val) {
        cars.push_back(val);
    }

    stringstream ss2(line2);
    while (ss2 >> val) {
        gens.push_back(val);
    }

    int car_idx = 0;
    int cycle_count = 0;
    
    while (car_idx < cars.size() && !gens.empty()) {
        int k = cars[car_idx];
        
        if (k >= gens.front()) {
            int current_sum = 0;
            // 贪心满足连续副将
            while (!gens.empty() && current_sum + gens.front() <= k) {
                current_sum += gens.front();
                gens.pop_front();
            }
            car_idx++;
            cycle_count = 0; // 换新车，循环次数重置
        } else {
            // 当前副将不能满足，去队尾排队
            gens.push_back(gens.front());
            gens.pop_front();
            cycle_count++;
            
            // 如果这辆车被所有剩下的人都拒绝了，直接丢弃这辆车
            if (cycle_count == gens.size()) {
                car_idx++;
                cycle_count = 0;
            }
        }
    }
    
    cout << gens.size() << "\n";
    return 0;
}
```

### Python
``` python
import sys
from collections import deque

def solve():
    lines = sys.stdin.read().splitlines()
    if len(lines) < 2:
        return
        
    cars = list(map(int, lines[0].split()))
    gens = deque(map(int, lines[1].split()))

    car_idx = 0
    cycle_count = 0
    
    while car_idx < len(cars) and gens:
        k = cars[car_idx]
        
        if k >= gens[0]:
            current_sum = 0
            # 贪心吃掉连续的副将
            while gens and current_sum + gens[0] <= k:
                current_sum += gens.popleft()
            car_idx += 1
            cycle_count = 0
        else:
            # 去队尾排队
            gens.append(gens.popleft())
            cycle_count += 1
            
            # 如果转了一整圈都没人要这车粮，直接淘汰此车
            if cycle_count == len(gens):
                car_idx += 1
                cycle_count = 0
                
    print(len(gens))

if __name__ == '__main__':
    solve()
```

### Java
``` java
import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        String line1 = br.readLine();
        String line2 = br.readLine();
        if (line1 == null || line2 == null) return;
        
        String[] carStrs = line1.trim().split("\\s+");
        String[] genStrs = line2.trim().split("\\s+");
        
        int[] cars = new int[carStrs.length];
        for (int i = 0; i < carStrs.length; i++) {
            cars[i] = Integer.parseInt(carStrs[i]);
        }
        
        ArrayDeque<Integer> gens = new ArrayDeque<>();
        for (String s : genStrs) {
            gens.offer(Integer.parseInt(s));
        }

        int carIdx = 0;
        int cycleCount = 0;
        
        while (carIdx < cars.length && !gens.isEmpty()) {
            int k = cars[carIdx];
            
            if (k >= gens.peek()) {
                int currentSum = 0;
                while (!gens.isEmpty() && currentSum + gens.peek() <= k) {
                    currentSum += gens.poll();
                }
                carIdx++;
                cycleCount = 0;
            } else {
                gens.offer(gens.poll());
                cycleCount++;
                
                if (cycleCount == gens.size()) {
                    carIdx++;
                    cycleCount = 0;
                }
            }
        }
        
        System.out.println(gens.size());
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	
	if !scanner.Scan() { return }
	carStrs := strings.Fields(scanner.Text())
	
	if !scanner.Scan() { return }
	genStrs := strings.Fields(scanner.Text())
	
	var cars []int
	for _, s := range carStrs {
		val, _ := strconv.Atoi(s)
		cars = append(cars, val)
	}
	
	var gens []int
	for _, s := range genStrs {
		val, _ := strconv.Atoi(s)
		gens = append(gens, val)
	}
	
	carIdx := 0
	cycleCount := 0
	
	for carIdx < len(cars) && len(gens) > 0 {
		k := cars[carIdx]
		
		if k >= gens[0] {
			currentSum := 0
			for len(gens) > 0 && currentSum + gens[0] <= k {
				currentSum += gens[0]
				gens = gens[1:] // 弹出队首
			}
			carIdx++
			cycleCount = 0
		} else {
			first := gens[0]
			gens = gens[1:]
			gens = append(gens, first)
			cycleCount++
			
			if cycleCount == len(gens) {
				carIdx++
				cycleCount = 0
			}
		}
	}
	
	fmt.Println(len(gens))
}
```
