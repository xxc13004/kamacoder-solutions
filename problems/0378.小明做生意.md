# 贪心 + 优先队列
[题目链接](https://kamacoder.com/problempage.php?pid=1458)
## 思路分析
这道题的核心在于：我们在有限的 k 天内，每天只能买一件商品，并且希望总利润最大。由于每次交易后，我们的本金都会增加（增加的金额即为该商品的利润），所以只要是我们当前买得起的商品，买了就一定能增加本金。本金越多，后续就能买得起成本更高的商品。

要让利润最大化，我们可以采用**贪心策略**：
1. **优先考虑当前能买得起的、利润最大的商品**。
2. 为了快速找到当前本金能买得起的商品，我们可以先将所有商品按照**成本价（cost）升序排序**。
3. 维护一个**大顶堆（优先队列）**，用于存储所有当前本金可以买得起的商品的利润。
4. 在每一天（最多 k 天）：
   - 遍历排序好的商品列表，将所有 `cost <= 当前本金` 的商品利润放入大顶堆中。
   - 如果大顶堆为空，说明剩下的商品都买不起了，直接结束。
   - 如果大顶堆不为空，弹出堆顶元素（即当前能买到的最大利润），将其加入总利润和当前本金中。

## 复杂度分析
1. 时间复杂度：O(N log N + K log N)；其中 N 为商品的数量，K 为天数。排序需要 O(N log N)，在最多 K 次循环中，每个商品最多入堆出堆一次，堆操作耗时 O(N log N)，每次弹出最大值耗时 O(K log N)。
2. 空间复杂度：O(N)；需要额外的数组来存储商品信息以及使用优先队列维护利润，最大占用 O(N) 的空间。

## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <sstream>
#include <string>

using namespace std;

int main() {
    // 优化输入输出
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    string line;
    if (!getline(cin, line)) return 0;
    int k = stoi(line);
    
    getline(cin, line);
    int m = stoi(line);
    
    getline(cin, line);
    stringstream ss1(line);
    vector<int> cost;
    int temp;
    while (ss1 >> temp) {
        cost.push_back(temp);
    }
    
    getline(cin, line);
    stringstream ss2(line);
    vector<int> profit;
    while (ss2 >> temp) {
        profit.push_back(temp);
    }
    
    int n = cost.size();
    vector<pair<int, int>> items(n);
    for (int i = 0; i < n; ++i) {
        items[i] = {cost[i], profit[i]};
    }
    
    // 按成本价从小到大排序
    sort(items.begin(), items.end());
    
    // 大顶堆，存储利润
    priority_queue<int> pq;
    int current_money = m;
    int total_profit = 0;
    int idx = 0;
    
    for (int i = 0; i < k; ++i) {
        // 将所有当前买得起的商品利润加入大顶堆
        while (idx < n && items[idx].first <= current_money) {
            pq.push(items[idx].second);
            idx++;
        }
        
        // 如果买不起任何新商品，直接结束
        if (pq.empty()) {
            break;
        }
        
        // 取出能获得的最大利润
        int max_profit = pq.top();
        pq.pop();
        current_money += max_profit;
        total_profit += max_profit;
    }
    
    cout << total_profit << "\n";
    return 0;
}
```

### Python
``` python
import sys
import heapq

def solve():
    # 优化输入，一次性读取所有行
    input_data = sys.stdin.read().splitlines()
    if not input_data:
        return
    
    k = int(input_data[0].strip())
    m = int(input_data[1].strip())
    cost = list(map(int, input_data[2].split()))
    profit = list(map(int, input_data[3].split()))
    
    # 将成本和利润组合，并按成本升序排序
    items = sorted(zip(cost, profit))
    
    max_heap = []
    current_money = m
    total_profit = 0
    idx = 0
    n = len(items)
    
    for _ in range(k):
        # 把当前买得起的商品利润放入大顶堆
        # Python内置的是小顶堆，所以存入利润的负数
        while idx < n and items[idx][0] <= current_money:
            heapq.heappush(max_heap, -items[idx][1])
            idx += 1
        
        # 如果堆为空，说明买不起任何商品了
        if not max_heap:
            break
            
        # 贪心：选取当前能获取的最大利润
        max_profit = -heapq.heappop(max_heap)
        current_money += max_profit
        total_profit += max_profit
        
    print(total_profit)

if __name__ == '__main__':
    solve()
```

### Java
``` java
import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        // 使用 BufferedReader 优化输入
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        String line = br.readLine();
        if (line == null) return;
        
        int k = Integer.parseInt(line.trim());
        int m = Integer.parseInt(br.readLine().trim());
        
        String[] costStrs = br.readLine().trim().split("\\s+");
        String[] profitStrs = br.readLine().trim().split("\\s+");
        
        int n = costStrs.length;
        int[][] items = new int[n][2];
        for (int i = 0; i < n; i++) {
            items[i][0] = Integer.parseInt(costStrs[i]);
            items[i][1] = Integer.parseInt(profitStrs[i]);
        }
        
        // 按成本价从小到大排序
        Arrays.sort(items, (a, b) -> Integer.compare(a[0], b[0]));
        
        // 大顶堆，存放利润
        PriorityQueue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder());
        int currentMoney = m;
        int totalProfit = 0;
        int idx = 0;
        
        for (int i = 0; i < k; i++) {
            // 将买得起的商品利润入堆
            while (idx < n && items[idx][0] <= currentMoney) {
                pq.offer(items[idx][1]);
                idx++;
            }
            
            // 无商品可买时退出
            if (pq.isEmpty()) {
                break;
            }
            
            // 获得最大利润
            int maxProfit = pq.poll();
            currentMoney += maxProfit;
            totalProfit += maxProfit;
        }
        
        System.out.println(totalProfit);
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"container/heap"
	"fmt"
	"os"
	"sort"
	"strconv"
	"strings"
)

type Item struct {
	cost   int
	profit int
}

// 定义大顶堆
type MaxHeap []int
func (h MaxHeap) Len() int           { return len(h) }
func (h MaxHeap) Less(i, j int) bool { return h[i] > h[j] } // 大于号实现大顶堆
func (h MaxHeap) Swap(i, j int)      { h[i], h[j] = h[j], h[i] }
func (h *MaxHeap) Push(x interface{}) {
	*h = append(*h, x.(int))
}
func (h *MaxHeap) Pop() interface{} {
	old := *h
	n := len(old)
	x := old[n-1]
	*h = old[0 : n-1]
	return x
}

func main() {
    // 优化输入
	scanner := bufio.NewScanner(os.Stdin)
	
	scanner.Scan()
	k, _ := strconv.Atoi(strings.TrimSpace(scanner.Text()))
	
	scanner.Scan()
	m, _ := strconv.Atoi(strings.TrimSpace(scanner.Text()))
	
	scanner.Scan()
	costStrs := strings.Fields(scanner.Text())
	scanner.Scan()
	profitStrs := strings.Fields(scanner.Text())
	
	n := len(costStrs)
	items := make([]Item, n)
	for i := 0; i < n; i++ {
		items[i].cost, _ = strconv.Atoi(costStrs[i])
		items[i].profit, _ = strconv.Atoi(profitStrs[i])
	}
	
	// 按成本升序排列
	sort.Slice(items, func(i, j int) bool {
		return items[i].cost < items[j].cost
	})
	
	h := &MaxHeap{}
	heap.Init(h)
	
	currentMoney := m
	totalProfit := 0
	idx := 0
	
	for i := 0; i < k; i++ {
        // 当前本金能买的起的所有商品利润入堆
		for idx < n && items[idx].cost <= currentMoney {
			heap.Push(h, items[idx].profit)
			idx++
		}
		
		if h.Len() == 0 {
			break
		}
		
        // 弹出当前能获得的最高利润
		maxProfit := heap.Pop(h).(int)
		currentMoney += maxProfit
		totalProfit += maxProfit
	}
	
	fmt.Println(totalProfit)
}
```
