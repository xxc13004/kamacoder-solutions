# 模拟 + 双指针
[题目链接](https://kamacoder.com/problempage.php?pid=1439)
## 思路分析

这道题其实就是统计从第一次释放技能开始，到最后结束，中间可能得最大的击杀数。

由于两个数组都是升序的，我们依次遍历技能释放的时间点，将这个时间点t1作为左指针，右指针一直往后走，一直到t1 + t >= b[r]为止；t，b[r]分别为技能无敌时间，b为敌人出现的时间。

此时，r - l就是本次释放技能击杀的敌人的数量，取所有时间段的最大值即可。

注意，在遍历技能释放时间点的时候，要判断本次释放技能是否处于冷却期，处于冷却期则跳过本次释放技能。

## 复杂度分析
1. 时间复杂度：O(m + n)；
2. 空间复杂度：O(1)。

### Cpp
``` cpp
#include <bits/stdc++.h>
#define endl '\n'
const int N = 500000, MOD = 1000000007;
using namespace std;
using LL = long long;
using PIL = pair<int, LL>;
using PII = pair<int, int>;
using T3I = tuple<int, int, int>;
int T = 0, dir[]{1, 0, -1, 0, 1};
int main() {
    cin.tie(nullptr) -> sync_with_stdio(false);
    int p, t, c, n, m;
    cin >> p;
    while (p--) {
        int ans = 0;
        cin >> t >> c >> n >> m;
        vector<int> a(n), b(m);
        for (int i = 0; i < n; ++i) cin >> a[i];  // 表示技能释放时间点
        for (int i = 0; i < m; ++i) cin >> b[i];  // 表示敌人出现时间点
        b.push_back(MOD);  // 增加哨兵
        int l = 0, r = 0;  // 双指针
        int next = 0;
        for (int i = 0; i < n; ++i) {
            if (next > a[i]) continue;
            next = a[i] + t + c;
            while (l <= m && b[l] < a[i]) ++l;  // 此时表示选择当前时间点，无法集中任何敌人，将左指针指向第一个大于此时间段被击杀的敌人
            if (r < l) r = l;
            while (r <= m && b[r] < a[i] + t) ++r;
            ans = max(ans, r - l);
        }
        cout << ans << endl;
    }
    return 0;
}
```

### Python
``` python
import sys

def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    
    idx = 0
    p = int(data[idx]); idx += 1
    
    results = []
    MOD = 1000000007  # 用于哨兵值
    
    for _ in range(p):
        t = int(data[idx]); idx += 1
        c = int(data[idx]); idx += 1
        n = int(data[idx]); idx += 1
        m = int(data[idx]); idx += 1
        
        a = []
        for i in range(n):
            a.append(int(data[idx])); idx += 1
        
        b = []
        for i in range(m):
            b.append(int(data[idx])); idx += 1
        
        b.append(MOD)  # 增加哨兵
        
        l = r = 0
        next_time = 0
        ans = 0
        
        for i in range(n):
            if next_time > a[i]:
                continue
            next_time = a[i] + t + c
            
            # 将左指针指向第一个大于此时间段被击杀的敌人
            while l <= m and b[l] < a[i]:
                l += 1
            
            if r < l:
                r = l
            
            while r <= m and b[r] < a[i] + t:
                r += 1
            
            ans = max(ans, r - l)
        
        results.append(str(ans))
    
    print("\n".join(results))

if __name__ == "__main__":
    main()
```

``` java
import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        String line = br.readLine();
        if (line == null || line.isEmpty()) return;
        
        int p = Integer.parseInt(line.trim());
        List<Integer> results = new ArrayList<>();
        final int MOD = 1000000007;
        
        for (int test = 0; test < p; test++) {
            String[] parts = br.readLine().trim().split("\\s+");
            int t = Integer.parseInt(parts[0]);
            int c = Integer.parseInt(parts[1]);
            int n = Integer.parseInt(parts[2]);
            int m = Integer.parseInt(parts[3]);
            
            int[] a = new int[n];
            parts = br.readLine().trim().split("\\s+");
            for (int i = 0; i < n; i++) {
                a[i] = Integer.parseInt(parts[i]);
            }
            
            int[] b = new int[m + 1];
            parts = br.readLine().trim().split("\\s+");
            for (int i = 0; i < m; i++) {
                b[i] = Integer.parseInt(parts[i]);
            }
            b[m] = MOD; // 增加哨兵
            
            int l = 0, r = 0;
            int nextTime = 0;
            int ans = 0;
            
            for (int i = 0; i < n; i++) {
                if (nextTime > a[i]) {
                    continue;
                }
                nextTime = a[i] + t + c;
                
                while (l <= m && b[l] < a[i]) {
                    l++;
                }
                
                if (r < l) {
                    r = l;
                }
                
                while (r <= m && b[r] < a[i] + t) {
                    r++;
                }
                
                ans = Math.max(ans, r - l);
            }
            
            results.add(ans);
        }
        
        for (int res : results) {
            System.out.println(res);
        }
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
)

func main() {
	// 使用 bufio 的 Scanner 并设置较大的缓冲区以支持大量输入
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)
	
	// 预分配一个较大的缓冲区（1MB）
	buf := make([]byte, 0, 1024*1024)
	scanner.Buffer(buf, 10*1024*1024) // 最大10MB缓冲区
	
	// 读取p
	scanner.Scan()
	p, _ := strconv.Atoi(scanner.Text())
	
	results := make([]int, 0, p) // 预分配结果切片
	const MOD = 1000000007
	
	for test := 0; test < p; test++ {
		// 读取 t, c, n, m
		scanner.Scan()
		t, _ := strconv.Atoi(scanner.Text())
		scanner.Scan()
		c, _ := strconv.Atoi(scanner.Text())
		scanner.Scan()
		n, _ := strconv.Atoi(scanner.Text())
		scanner.Scan()
		m, _ := strconv.Atoi(scanner.Text())
		
		// 预分配切片
		a := make([]int, n)
		for i := 0; i < n; i++ {
			scanner.Scan()
			a[i], _ = strconv.Atoi(scanner.Text())
		}
		
		b := make([]int, m+1)
		for i := 0; i < m; i++ {
			scanner.Scan()
			b[i], _ = strconv.Atoi(scanner.Text())
		}
		b[m] = MOD // 增加哨兵
		
		l, r := 0, 0
		nextTime := 0
		ans := 0
		
		for i := 0; i < n; i++ {
			if nextTime > a[i] {
				continue
			}
			nextTime = a[i] + t + c
			
			// 将左指针指向第一个大于此时间段被击杀的敌人
			for l <= m && b[l] < a[i] {
				l++
			}
			
			if r < l {
				r = l
			}
			
			for r <= m && b[r] < a[i]+t {
				r++
			}
			
			if r-l > ans {
				ans = r - l
			}
		}
		
		results = append(results, ans)
	}
	
	// 使用 strings.Builder 或直接写入输出缓冲区
	writer := bufio.NewWriter(os.Stdout)
	defer writer.Flush()
	
	for _, res := range results {
		fmt.Fprintln(writer, res)
	}
}
```
