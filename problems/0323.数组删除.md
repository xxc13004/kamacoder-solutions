# 哈希表
[题目链接](https://kamacoder.com/problempage.php?pid=1403)
## 思路分析
用哈希表统计每个数字出现的次数，然后遍历哈希表，将出现次数不在[l, r]的数字全部取出来，将其添加到集合中，最后遍历数组，将在集合中的数字都保存到ans数组中。

注意这里输出的大小不是集合的大小，而是最后的数组的大小，所以需要将答案先保存到一个数组中。
## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(n)。

``` cpp
#include <bits/stdc++.h>
#define endl '\n'
const int N = 100005, MOD = 1000000007;
using namespace std;
using LL = long long;
using PIL = pair<int, LL>;
using PII = pair<int, int>;
using T3I = tuple<int, int, int>;
int t = 0, dir[]{1, 0, -1, 0, 1};
int main() {
    cin.tie(nullptr)->sync_with_stdio(false);
    cin >> t;
    while (t--) {
        int n, l, r;
        cin >> n >> l >> r;
        unordered_map<int, int> ump;
        vector<int> arr(n);
        for (int i = 0; i < n; i++) {
            int x;
            cin >> x;
            arr[i] = x;
            ump[x]++;
        }
        unordered_set<int> s;
        for (auto [x, y] : ump) {
            if (y >= l && y <= r) {
                s.insert(x);
            }
        }
        vector<int> ans;
        for (int i = 0; i < n; i++) {
            if (!s.count(arr[i])) {
                ans.push_back(arr[i]);
            }
        }
        cout << ans.size() << endl;
        for (int x : ans) {
            cout << x << ' ';
        }
        cout << endl;
    }
    return 0;
}
```

### Python
``` python
import sys
from collections import defaultdict

def main():
    input = sys.stdin.readline
    t = int(input().strip())
    for _ in range(t):
        _, l, r = map(int, input().split())
        arr = list(map(int, input().split()))
        
        ump = defaultdict(int)
        for x in arr:
            ump[x] += 1
        
        s = set()
        for x, count in ump.items():
            if l <= count <= r:
                s.add(x)
        
        result = []
        for x in arr:
            if x not in s:
                result.append(str(x))
        print(len(result))
        print(' '.join(result))

if __name__ == "__main__":
    main()
```

### Java
``` java
import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        PrintWriter pw = new PrintWriter(new BufferedWriter(new OutputStreamWriter(System.out)));
        
        int t = Integer.parseInt(br.readLine().trim());
        while (t-- > 0) {
            StringTokenizer st = new StringTokenizer(br.readLine());
            int n = Integer.parseInt(st.nextToken());
            int l = Integer.parseInt(st.nextToken());
            int r = Integer.parseInt(st.nextToken());
            
            int[] arr = new int[n];
            Map<Integer, Integer> ump = new HashMap<>();
            
            st = new StringTokenizer(br.readLine());
            for (int i = 0; i < n; i++) {
                int x = Integer.parseInt(st.nextToken());
                arr[i] = x;
                ump.put(x, ump.getOrDefault(x, 0) + 1);
            }
            
            Set<Integer> s = new HashSet<>();
            for (Map.Entry<Integer, Integer> entry : ump.entrySet()) {
                int count = entry.getValue();
                if (count >= l && count <= r) {
                    s.add(entry.getKey());
                }
            }
            
            List<Integer> ans = new ArrayList<>();
            for (int x : arr) {
                if (!s.contains(x)) {
                    ans.add(x);
                }
            }
            
            pw.println(ans.size());
            StringBuilder sb = new StringBuilder();
            for (int num : ans) {
                sb.append(num).append(" ");
            }
            if (sb.length() > 0) {
                sb.setLength(sb.length() - 1);
            }
            pw.println(sb);
        }
        
        pw.flush();
        pw.close();
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	writer := bufio.NewWriter(os.Stdout)
	defer writer.Flush()

	tStr, _ := reader.ReadString('\n')
	t, _ := strconv.Atoi(strings.TrimSpace(tStr))

	for i := 0; i < t; i++ {
		line, _ := reader.ReadString('\n')
		fields := strings.Fields(line)
		n, _ := strconv.Atoi(fields[0])
		l, _ := strconv.Atoi(fields[1])
		r, _ := strconv.Atoi(fields[2])

		arrLine, _ := reader.ReadString('\n')
		arrStr := strings.Fields(arrLine)
		arr := make([]int, n)
		ump := make(map[int]int)

		for j := 0; j < n; j++ {
			x, _ := strconv.Atoi(arrStr[j])
			arr[j] = x
			ump[x]++
		}

		s := make(map[int]bool)
		for x, count := range ump {
			if count >= l && count <= r {
				s[x] = true
			}
		}

		ans := make([]int, 0)
		for _, x := range arr {
			if !s[x] {
				ans = append(ans, x)
			}
		}

		fmt.Fprintln(writer, len(ans))
		if len(ans) > 0 {
			result := make([]string, len(ans))
			for i, num := range ans {
				result[i] = strconv.Itoa(num)
			}
			fmt.Fprintln(writer, strings.Join(result, " "))
		} else {
			fmt.Fprintln(writer)
		}
	}
}
```
