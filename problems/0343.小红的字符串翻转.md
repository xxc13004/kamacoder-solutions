# 脑筋急转弯
[题目链接](https://kamacoder.com/problempage.php?pid=1423)
## 思路分析

分析题目可知，每次翻转都会使一个字符的位置固定。这个字符是字符串的后缀n-k+1位中的每个字符。如果字符串为abcdefg, k, n分别为3，7。那么大致可以将字符串分为两个部分ab + cdefg。cdefg是7-3+1=5个后缀。每次翻转都会确定一个后缀串中的字符的位置，且接下来它不会再变动。所以最终的串一定是cdefg开头的。对于前缀串，如果n-k+1是偶数，说明它翻转了偶数次，则不变；如果是奇数，那么它需要翻转。所以最终的串为cdefg + ba = cdefgba。

## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(n)。

``` python
import sys

def main():
    input_data = sys.stdin.read().split()
    n, k = map(int, input_data[:2])
    s = input_data[2]
    
    # 根据条件构造结果
    if (n - k + 1) % 2 == 0:
        # 偶数情况：s[k-1:] + s[:k-1]
        result = s[k-1:] + s[:k-1]
    else:
        # 奇数情况：s[k-1:] + reverse(s[:k-1])
        result = s[k-1:] + s[:k-1][::-1]
    
    sys.stdout.write(result)

if __name__ == "__main__":
    main()
```

### Java
``` java
import java.util.*;
import java.io.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        PrintWriter pw = new PrintWriter(new BufferedWriter(new OutputStreamWriter(System.out)));
        
        // 读取n, k和字符串s
        String[] firstLine = br.readLine().split(" ");
        int n = Integer.parseInt(firstLine[0]);
        int k = Integer.parseInt(firstLine[1]);
        String s = br.readLine();
        
        // 根据条件构造结果
        if ((n - k + 1) % 2 == 0) {
            // 偶数情况：s[k-1:] + s[0:k-1]
            pw.println(s.substring(k - 1) + s.substring(0, k - 1));
        } else {
            // 奇数情况：s[k-1:] + reverse(s[0:k-1])
            String prefix = s.substring(0, k - 1);
            StringBuilder rs = new StringBuilder(prefix);
            rs.reverse();
            pw.println(s.substring(k - 1) + rs.toString());
        }
        
        pw.flush();
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"os"
	"strconv"
)

func main() {
	// 方法1：使用Scanner读取
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)
	
	// 读取n和k
	scanner.Scan()
	n, _ := strconv.Atoi(scanner.Text())
	scanner.Scan()
	k, _ := strconv.Atoi(scanner.Text())
	
	// 读取字符串s
	scanner.Scan()
	s := scanner.Text()
	
	// 准备输出
	out := bufio.NewWriter(os.Stdout)
	defer out.Flush()
	
	if (n-k+1)%2 == 0 {
		// 偶数情况
		out.WriteString(s[k-1:])
		out.WriteString(s[:k-1])
	} else {
		// 奇数情况
		out.WriteString(s[k-1:])
		
		// 反转前缀
		prefix := s[:k-1]
		for i := len(prefix) - 1; i >= 0; i-- {
			out.WriteByte(prefix[i])
		}
	}
	
	out.WriteByte('\n')
}
```
