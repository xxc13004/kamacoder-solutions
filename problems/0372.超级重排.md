# 贪心
[题目链接](https://kamacoder.com/problempage.php?pid=1452)
## 思路分析

根据题意，每个原数字 $ai$ 的长度固定，假设某一个数字的长度为 $L$，那么它需要 $10^0, 10^1, 10^2, \ldots, 10^{L-1}$ 这些“数位权重”来填补。
也就是说，数字的重排不仅受限于数字池里的具体字符，也受限于我们要构造的各个数字所需的所有“位置”。

我们可以将问题理解为：
1. **收集所有权重**：如果我们要构造一个长度为 $L$ 的数字，那就需要一个个位（权值为 $10^0$），一个十位（权值为 $10^1$）... 一直达到 $10^{L-1}$。遍历所有的原数字，统计我们在所有数字中总共需要多少个 $10^0$，多少个 $10^1$，等等。
2. **收集所有数字**：把原数组中所有的数字拆分成单个字符，统计 `1~9` 各自出现的频率。
3. **贪心匹配**：权值越大的位置，分配越大的数字，对总和的贡献就越大。因此，我们将所有收集到的位置按权值从大到小排列（即从最高位开始分配），再将可用的数字字符从 `9` 到 `1` 排列。每次从最高位的需求出发，尽可能多地填入当前最大的数字。

由于题目中数字最大为 $10^9$，即长度最大不超过 10，我们只需用大小为 15 的数组即可完全统计所有位置频率，并且只需用大小为 10 的数组统计 `1~9` 的频率。

## 复杂度分析
1. 时间复杂度：O(ΣLi)，其中，ΣLi是所有数字长度的总和；
2. 空间复杂度：O(1)。我们只需维护几个长度不超过 15 的数组来统计频次和权重，这属于常数级别的空间。

## 代码实现

### Cpp
```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    // 优化输入输出流
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;

    vector<long long> freq(10, 0);       // 统计 '1'-'9' 的个数
    vector<long long> pos_count(15, 0);  // 统计各个权值位置(10^p)需要的数量

    string s;
    for (int i = 0; i < n; ++i) {
        cin >> s;
        int len = s.length();
        for (int j = 0; j < len; ++j) {
            pos_count[j]++;              // 长度为len的数贡献了 10^0, ..., 10^(len-1) 
            freq[s[j] - '0']++;          // 统计字符出现次数
        }
    }

    // 预处理 10 的各次幂
    vector<long long> pow10(15, 1);
    for (int i = 1; i < 15; ++i) {
        pow10[i] = pow10[i - 1] * 10;
    }

    long long total_sum = 0;
    int current_digit = 9;

    // 从最高位权重开始贪心匹配最大的数字
    for (int p = 14; p >= 0; --p) {
        while (pos_count[p] > 0) {
            // 找到当前可用的最大数字
            while (current_digit > 0 && freq[current_digit] == 0) {
                current_digit--;
            }
            if (current_digit == 0) break;

            // 批量分配，取该位置需求和该数字剩余量的最小值
            long long take = min(pos_count[p], freq[current_digit]);
            total_sum += take * current_digit * pow10[p];
            
            // 扣除消耗
            pos_count[p] -= take;
            freq[current_digit] -= take;
        }
    }

    cout << total_sum << "\n";
    return 0;
}
```

### Python
```python
import sys

def main():
    # 优化读取：一次性读取所有输入切分为Token
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    n = int(input_data[0])
    freq = [0] * 10
    pos_count = [0] * 15
    
    # 统计位权和数字频率
    for i in range(1, n + 1):
        s = input_data[i]
        length = len(s)
        for j in range(length):
            pos_count[j] += 1
            freq[int(s[j])] += 1
            
    total_sum = 0
    current_digit = 9
    
    # 贪心计算最大和：大位置配大数字
    for p in range(14, -1, -1):
        while pos_count[p] > 0:
            while current_digit > 0 and freq[current_digit] == 0:
                current_digit -= 1
            if current_digit == 0:
                break
            
            take = min(pos_count[p], freq[current_digit])
            total_sum += take * current_digit * (10 ** p)
            pos_count[p] -= take
            freq[current_digit] -= take
            
    print(total_sum)

if __name__ == '__main__':
    main()
```

### Java
```java
import java.io.*;
import java.util.*;

public class Main {
    static class FastReader {
        BufferedReader br;
        StringTokenizer st;

        public FastReader() {
            br = new BufferedReader(new InputStreamReader(System.in));
        }

        String next() {
            while (st == null || !st.hasMoreElements()) {
                try {
                    st = new StringTokenizer(br.readLine());
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            return st.nextToken();
        }

        int nextInt() {
            return Integer.parseInt(next());
        }
    }

    public static void main(String[] args) {
        FastReader in = new FastReader();
        int n = in.nextInt();
        
        long[] freq = new long[10];
        long[] posCount = new long[15];
        
        for (int i = 0; i < n; i++) {
            String s = in.next();
            int len = s.length();
            for (int j = 0; j < len; j++) {
                posCount[j]++;
                freq[s.charAt(j) - '0']++;
            }
        }
        
        long[] pow10 = new long[15];
        pow10[0] = 1;
        for (int i = 1; i < 15; i++) {
            pow10[i] = pow10[i - 1] * 10L;
        }
        
        long totalSum = 0;
        int currentDigit = 9;
        
        for (int p = 14; p >= 0; p--) {
            while (posCount[p] > 0) {
                while (currentDigit > 0 && freq[currentDigit] == 0) {
                    currentDigit--;
                }
                if (currentDigit == 0) break;
                
                long take = Math.min(posCount[p], freq[currentDigit]);
                totalSum += take * currentDigit * pow10[p];
                
                posCount[p] -= take;
                freq[currentDigit] -= take;
            }
        }
        
        System.out.println(totalSum);
    }
}
```

### Go
```go
package main

import (
	"bufio"
	"fmt"
	"os"
)

func main() {
    // 采用自定义大缓冲的 Scanner 来提速
	scanner := bufio.NewScanner(os.Stdin)
	// 调整 Scanner 允许的最大 token 大小
	const maxCapacity = 1024 * 1024
	buf := make([]byte, maxCapacity)
	scanner.Buffer(buf, maxCapacity)
	scanner.Split(bufio.ScanWords)

	if !scanner.Scan() {
		return
	}

	freq := make([]int64, 10)
	pos_count := make([]int64, 15)

	for scanner.Scan() {
		s := scanner.Text()
		length := len(s)
		for i := 0; i < length; i++ {
			pos_count[i]++
			freq[s[i]-'0']++
		}
	}

	pow10 := make([]int64, 15)
	pow10[0] = 1
	for i := 1; i < 15; i++ {
		pow10[i] = pow10[i-1] * 10
	}

	var total_sum int64 = 0
	current_digit := 9

	for p := 14; p >= 0; p-- {
		for pos_count[p] > 0 {
			for current_digit > 0 && freq[current_digit] == 0 {
				current_digit--
			}
			if current_digit == 0 {
				break
			}

			take := pos_count[p]
			if freq[current_digit] < take {
				take = freq[current_digit]
			}

			total_sum += take * int64(current_digit) * pow10[p]
			pos_count[p] -= take
			freq[current_digit] -= take
		}
	}

	fmt.Println(total_sum)
}
```
