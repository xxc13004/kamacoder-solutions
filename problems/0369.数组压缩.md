# 模拟
[题目链接](https://kamacoder.com/problempage.php?pid=1449)
## 思路分析

1. **输入处理**：由于题目给定的输入包含了中括号 `[]` 和逗号 `,`，我们需要先将首尾的中括号去除，然后通过逗号分割成字符串数组，再转化为整数数组。
2. **寻找连续区间**：
   - 遍历整数数组，用一个指针 `i` 记录当前区间的起点 `start = nums[i]`。
   - 使用一个 `while` 循环向后探查，如果下一个元素满足 `nums[i + 1] == nums[i] + 1`，说明属于连续区间，指针 `i` 继续后移。
   - 当遇到不连续的元素或到达数组末尾时，探查结束，当前 `i` 指向的就是区间的终点 `end = nums[i]`。
   - 如果起点和终点相同 (`start == end`)，说明这是一个独立的数字，将其格式化为 `"start"`。
   - 如果起点和终点不同 (`start != end`)，说明这是一个连续区间，将其格式化为 `"start->end"`。

## 复杂度分析
1. 时间复杂度：O(N)；
2. 空间复杂度：O(N)。

## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <string>
#include <sstream>

using namespace std;

int main() {
    string s;
    if (!(cin >> s)) return 0;
    
    // 去除前后的中括号
    s = s.substr(1, s.length() - 2); 
    vector<int> nums;
    
    // 解析逗号分隔的数字
    if (!s.empty()) {
        stringstream ss(s);
        string item;
        while (getline(ss, item, ',')) {
            nums.push_back(stoi(item));
        }
    }

    vector<string> res;
    int n = nums.size();
    
    // 核心逻辑
    for (int i = 0; i < n; i++) {
        int start = nums[i];
        // 寻找连续区间的终点
        while (i + 1 < n && nums[i + 1] == nums[i] + 1) {
            i++;
        }
        // 格式化当前区间
        if (start != nums[i]) {
            res.push_back(to_string(start) + "->" + to_string(nums[i]));
        } else {
            res.push_back(to_string(start));
        }
    }

    // 格式化输出
    cout << "[";
    for (int i = 0; i < res.size(); i++) {
        cout << res[i] << (i == res.size() - 1 ? "" : ",");
    }
    cout << "]" << endl;
    
    return 0;
}
```

### Python
``` python
import sys

def main():
    # 读取输入并去除首尾空白字符
    line = sys.stdin.read().strip()
    if not line:
        return
    
    # 去除两边的中括号
    line = line[1:-1]
    if not line:
        print("[]")
        return
        
    # 解析成整数列表
    nums = list(map(int, line.split(',')))
    
    res = []
    n = len(nums)
    i = 0
    
    # 核心逻辑
    while i < n:
        start = nums[i]
        # 寻找连续区间的终点
        while i + 1 < n and nums[i + 1] == nums[i] + 1:
            i += 1
        
        # 格式化当前区间
        if start != nums[i]:
            res.append(f"{start}->{nums[i]}")
        else:
            res.append(str(start))
        
        i += 1
        
    # 格式化输出
    print("[" + ",".join(res) + "]")

if __name__ == "__main__":
    main()
```

### Java
``` java
import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        if (!scanner.hasNextLine()) return;
        String s = scanner.nextLine().trim();
        
        // 去除中括号
        s = s.substring(1, s.length() - 1);
        if (s.isEmpty()) {
            System.out.println("[]");
            return;
        }
        
        // 解析数组
        String[] parts = s.split(",");
        int[] nums = new int[parts.length];
        for (int i = 0; i < parts.length; i++) {
            nums[i] = Integer.parseInt(parts[i]);
        }
        
        List<String> res = new ArrayList<>();
        int n = nums.length;
        
        // 核心逻辑
        for (int i = 0; i < n; i++) {
            int start = nums[i];
            // 探查连续段
            while (i + 1 < n && nums[i + 1] == nums[i] + 1) {
                i++;
            }
            if (start != nums[i]) {
                res.add(start + "->" + nums[i]);
            } else {
                res.add(String.valueOf(start));
            }
        }
        
        // 格式化输出
        System.out.println("[" + String.join(",", res) + "]");
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	line, _ := reader.ReadString('\n')
	line = strings.TrimSpace(line)

	if len(line) == 0 {
		return
	}

	// 去除中括号 []
	line = line[1 : len(line)-1]
	if line == "" {
		fmt.Println("[]")
		return
	}

	// 解析逗号分隔的字符串
	parts := strings.Split(line, ",")
	nums := make([]int, len(parts))
	for i, part := range parts {
		nums[i], _ = strconv.Atoi(part)
	}

	var res []string
	n := len(nums)
    
	// 核心逻辑
	for i := 0; i < n; i++ {
		start := nums[i]
		// 寻找连续的结尾
		for i+1 < n && nums[i+1] == nums[i]+1 {
			i++
		}
		
		if start != nums[i] {
			res = append(res, fmt.Sprintf("%d->%d", start, nums[i]))
		} else {
			res = append(res, fmt.Sprintf("%d", start))
		}
	}

	// 格式化输出
	fmt.Printf("[%s]\n", strings.Join(res, ","))
}
```
