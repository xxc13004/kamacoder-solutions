# 排序 + 多指针
[题目链接](https://kamacoder.com/problempage.php?pid=1461)
## 思路分析
这道题的核心需求是：领养人每次从符合其偏好毛色（第一种或第二种毛色之一）的宠物中，选择“领养编号”最小的一只。如果直接在每次领养时遍历寻找，时间复杂度会非常高导致超时。

我们可以采用“排序 + 多指针”的思想：
1. 将所有宠物按照“领养编号”（即优先级，数值越小优先级越高）进行升序排序。
2. 排序后，我们按从左到右的顺序，分别记录拥有毛色 1、毛色 2、毛色 3 的宠物在排序后数组中的下标位置。这样就能得到三个列表，每个列表里的宠物下标对应的领养编号天然就是从小到大递增的。
3. 维护一个布尔数组 adopted，记录每个下标对应的宠物是否已经被领养。
4. 为三种毛色各维护一个遍历指针，初始均指向 0。
5. 当来一个偏好某种毛色的领养人时，我们只需取出对应毛色列表的指针，不断向后移动指针，跳过那些已经被领养的宠物。遇到第一个未被领养的宠物时，将其标记为已领养，输出其编号并结束当前领养人的匹配。如果指针越界，说明没有满足条件的宠物，输出 -1。

## 复杂度分析
1. 时间复杂度：O(n log n + m)；
2. 空间复杂度：O(n + m)。

## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Pet {
    int x, a, b;
    bool operator<(const Pet& other) const {
        return x < other.x;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;

    vector<Pet> pets(n);
    for (int i = 0; i < n; ++i) cin >> pets[i].x;
    for (int i = 0; i < n; ++i) cin >> pets[i].a;
    for (int i = 0; i < n; ++i) cin >> pets[i].b;

    int m;
    cin >> m;
    vector<int> c(m);
    for (int i = 0; i < m; ++i) cin >> c[i];

    sort(pets.begin(), pets.end());

    vector<int> lists[4];
    for (int i = 0; i < n; ++i) {
        lists[pets[i].a].push_back(i);
        if (pets[i].a != pets[i].b) {
            lists[pets[i].b].push_back(i);
        }
    }

    int ptrs[4] = {0, 0, 0, 0};
    vector<bool> adopted(n, false);

    for (int i = 0; i < m; ++i) {
        int pref = c[i];
        bool found = false;
        while (ptrs[pref] < lists[pref].size()) {
            int idx = lists[pref][ptrs[pref]];
            ptrs[pref]++;
            if (!adopted[idx]) {
                adopted[idx] = true;
                cout << pets[idx].x << (i == m - 1 ? "" : " ");
                found = true;
                break;
            }
        }
        if (!found) {
            cout << -1 << (i == m - 1 ? "" : " ");
        }
    }
    cout << "\n";

    return 0;
}
```

### Python
``` python
import sys

def main():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    n = int(input_data[0])
    x = [int(i) for i in input_data[1:n+1]]
    a = [int(i) for i in input_data[n+1:2*n+1]]
    b = [int(i) for i in input_data[2*n+1:3*n+1]]
    m = int(input_data[3*n+1])
    c = [int(i) for i in input_data[3*n+2:3*n+2+m]]
    
    pets = [(x[i], a[i], b[i]) for i in range(n)]
    pets.sort(key=lambda p: p[0])
    
    lists = {1: [], 2: [], 3: []}
    for i in range(n):
        lists[pets[i][1]].append(i)
        if pets[i][1] != pets[i][2]:
            lists[pets[i][2]].append(i)
            
    ptrs = {1: 0, 2: 0, 3: 0}
    adopted = [False] * n
    
    ans = []
    for pref in c:
        found = False
        while ptrs[pref] < len(lists[pref]):
            idx = lists[pref][ptrs[pref]]
            ptrs[pref] += 1
            if not adopted[idx]:
                adopted[idx] = True
                ans.append(str(pets[idx][0]))
                found = True
                break
        if not found:
            ans.append("-1")
            
    print(" ".join(ans))

if __name__ == '__main__':
    main()
```

### Java
``` java
import java.io.*;
import java.util.*;

public class Main {
    static class Pet implements Comparable<Pet> {
        int x, a, b;
        public Pet(int x, int a, int b) {
            this.x = x;
            this.a = a;
            this.b = b;
        }
        @Override
        public int compareTo(Pet other) {
            return Integer.compare(this.x, other.x);
        }
    }

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());
        
        if (!st.hasMoreTokens()) return;
        int n = Integer.parseInt(st.nextToken());
        
        int[] x = new int[n];
        int[] a = new int[n];
        int[] b = new int[n];
        
        readArray(br, st, x, n);
        readArray(br, st, a, n);
        readArray(br, st, b, n);
        
        st = new StringTokenizer(br.readLine());
        int m = Integer.parseInt(st.nextToken());
        int[] c = new int[m];
        readArray(br, st, c, m);
        
        Pet[] pets = new Pet[n];
        for (int i = 0; i < n; i++) {
            pets[i] = new Pet(x[i], a[i], b[i]);
        }
        
        Arrays.sort(pets);
        
        ArrayList<Integer>[] lists = new ArrayList[4];
        for (int i = 1; i <= 3; i++) {
            lists[i] = new ArrayList<>();
        }
        
        for (int i = 0; i < n; i++) {
            lists[pets[i].a].add(i);
            if (pets[i].a != pets[i].b) {
                lists[pets[i].b].add(i);
            }
        }
        
        int[] ptrs = new int[4];
        boolean[] adopted = new boolean[n];
        StringBuilder sb = new StringBuilder();
        
        for (int i = 0; i < m; i++) {
            int pref = c[i];
            boolean found = false;
            while (ptrs[pref] < lists[pref].size()) {
                int idx = lists[pref].get(ptrs[pref]);
                ptrs[pref]++;
                if (!adopted[idx]) {
                    adopted[idx] = true;
                    sb.append(pets[idx].x).append(i == m - 1 ? "" : " ");
                    found = true;
                    break;
                }
            }
            if (!found) {
                sb.append("-1").append(i == m - 1 ? "" : " ");
            }
        }
        System.out.println(sb.toString());
    }

    private static void readArray(BufferedReader br, StringTokenizer st, int[] arr, int len) throws IOException {
        int idx = 0;
        while (idx < len) {
            if (!st.hasMoreTokens()) {
                String line = br.readLine();
                if (line == null) break;
                st = new StringTokenizer(line);
            }
            while (st.hasMoreTokens() && idx < len) {
                arr[idx++] = Integer.parseInt(st.nextToken());
            }
        }
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"sort"
	"strconv"
	"strings"
)

type Pet struct {
	x, a, b int
}

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)
	
	// Increase buffer size to handle potentially large inputs smoothly
	buf := make([]byte, 1024*1024)
	scanner.Buffer(buf, 1024*1024*10)

	if !scanner.Scan() {
		return
	}
	n, _ := strconv.Atoi(scanner.Text())

	x := make([]int, n)
	for i := 0; i < n; i++ {
		scanner.Scan()
		x[i], _ = strconv.Atoi(scanner.Text())
	}

	a := make([]int, n)
	for i := 0; i < n; i++ {
		scanner.Scan()
		a[i], _ = strconv.Atoi(scanner.Text())
	}

	b := make([]int, n)
	for i := 0; i < n; i++ {
		scanner.Scan()
		b[i], _ = strconv.Atoi(scanner.Text())
	}

	scanner.Scan()
	m, _ := strconv.Atoi(scanner.Text())

	c := make([]int, m)
	for i := 0; i < m; i++ {
		scanner.Scan()
		c[i], _ = strconv.Atoi(scanner.Text())
	}

	pets := make([]Pet, n)
	for i := 0; i < n; i++ {
		pets[i] = Pet{x: x[i], a: a[i], b: b[i]}
	}

	sort.Slice(pets, func(i, j int) bool {
		return pets[i].x < pets[j].x
	})

	lists := make([][]int, 4)
	for i := range lists {
		lists[i] = make([]int, 0)
	}

	for i := 0; i < n; i++ {
		lists[pets[i].a] = append(lists[pets[i].a], i)
		if pets[i].a != pets[i].b {
			lists[pets[i].b] = append(lists[pets[i].b], i)
		}
	}

	ptrs := make([]int, 4)
	adopted := make([]bool, n)
	
	var outBuilder strings.Builder

	for i := 0; i < m; i++ {
		pref := c[i]
		found := false
		for ptrs[pref] < len(lists[pref]) {
			idx := lists[pref][ptrs[pref]]
			ptrs[pref]++
			if !adopted[idx] {
				adopted[idx] = true
				outBuilder.WriteString(strconv.Itoa(pets[idx].x))
				found = true
				break
			}
		}
		if !found {
			outBuilder.WriteString("-1")
		}
		if i != m-1 {
			outBuilder.WriteString(" ")
		}
	}
	fmt.Println(outBuilder.String())
}
```
