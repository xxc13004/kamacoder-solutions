[题目链接](https://kamacoder.com/problempage.php?pid=1405)
## 思路分析

按照题意模拟，用两个指针，每次读取一个字符后，下一次移动的步长+1。

## 复杂度分析

1. 时间复杂度：O(n)；
2. 空间复杂度：O(n)。

### Python
``` python
import sys

def main():
    # 优化输入处理，支持大量输入
    data = sys.stdin.read().split()
    if not data:
        return
    
    s = data[0]
    n = len(s)
    ans = []
    
    # 使用与C++相同的逻辑
    i, j = 1, 0
    while j < n:
        ans.append(s[j])
        j += i
        i += 1
    
    print(''.join(ans))

if __name__ == "__main__":
    main()
```

### Go
``` go
package main

import (
    "bufio"
    "fmt"
    "os"
)

func main() {
    // 优化输入处理，支持大量输入
    reader := bufio.NewReader(os.Stdin)
    s, _ := reader.ReadString('\n')
    // 去除换行符
    if len(s) > 0 && s[len(s)-1] == '\n' {
        s = s[:len(s)-1]
    }
    
    n := len(s)
    var ans []byte
    
    // 使用与C++相同的逻辑
    i := 1
    j := 0
    for j < n {
        ans = append(ans, s[j])
        j += i
        i++
    }
    
    fmt.Println(string(ans))
}
```
