# 组合数学
[题目链接](https://kamacoder.com/problempage.php?pid=1424)
## 思路分析

在不考虑题目所给的限制时，一共有几种组合，答案是``` total = C(n, 3) = n * (n - 1) * (n - 2) / 6 ```。但是实际上我们需要去除掉下面两种情况。

1. a的值有重复（三个二元组里面至少有两个a的值重复）；
2. b的值有重复（三个二元组里面至少有两个b的值重复）。
3. 由于还存在另一种情况，即第三个二元组的a的值与某个二元组的a重复，b的值与另一个二元组的b重复。如``` (1, 2), (2, 1), (2, 2) ```。所以需要利用容斥原理减掉这种情况。

计算方法如下，设第一种情况为A，第二种为B，第三种为AB。

1. 统计每个a出现的数量，设为x，那么至少有两个a值重复的情况就可以分解为两个二元组a重复和3个a都重复。根据组合数学原理得``` C(x, 2) * (n - x) + C(x, 3) ```。前面的部分表示，先从x个数里面选两个，再从剩下的数里面选一个；后面的部分表示直接从x个数里面选三个；
2. 第二种情况同理；
3. 第三种情况：如果我们已经确定了两个三元组的话，在满足第三个条件的情况下，第三个三元组是唯一确定的。那就是取第一个三元组的a和取第二个三元组的b构成。所以他的计算方式为``` (x - 1) * (y - 1) ```, 其中x, y分别为每个a和b的数量。
4. A, B, AB就是将上面计算的结果累加。
5. 最终结果为``` total - A - B + AB ```。

## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(n)。
## 代码实现
### Cpp
``` cpp
#include <bits/stdc++.h>
#define endl '\n'
const int N = 500000, MOD = 1000000007;
using namespace std;
using LL = long long;
using PIL = pair<int, LL>;
using PII = pair<int, int>;
using T3I = tuple<int, int, int>;
int t = 0, dir[]{1, 0, -1, 0, 1};
int main() {
    cin.tie(nullptr) -> sync_with_stdio(false);
    int n; 
    cin >> n;
    vector<PII> vp;
    vector<LL> cnt_a(n + 1), cnt_b(n + 1);
    for (int i = 0, a, b; i < n; ++i) {
        cin >> a >> b;
        vp.emplace_back(a, b);
        ++cnt_a[a], ++cnt_b[b];
    }
    LL total = 1LL * n * (n - 1) * (n - 2) / 6;  // 总数量
    LL A = 0, B = 0;  // 记录A与B类型的数量
    for (LL x : cnt_a) {
        if (x >= 2) A += x * (x - 1) / 2 * (n - x);
        if (x >= 3) A += x * (x - 1) * (x - 2) / 6;
    }
    for (LL x : cnt_b) {
        if (x >= 2) B += x * (x - 1) / 2 * (n - x);
        if (x >= 3) B += x * (x - 1) * (x - 2) / 6;
    }
    LL AB = 0;  // 记录AB类型的数量
    for (auto [a, b] : vp) {
        AB += 1LL * (cnt_a[a] - 1) * (cnt_b[b] - 1);
    }
    LL ans = total - A - B + AB;
    cout << ans << endl;
    return 0;
}
```
### Python
``` python
import sys

data = sys.stdin.read().split()
it = iter(data)
n = int(next(it))
cnt_a = [0] * (n + 1)
cnt_b = [0] * (n + 1)
edges = []
for _ in range(n):
    a = int(next(it))
    b = int(next(it))
    edges.append((a, b))
    cnt_a[a] += 1
    cnt_b[b] += 1

total_triples = n * (n - 1) * (n - 2) // 6

A = 0
for d in cnt_a:
    if d >= 2:
        A += d * (d - 1) // 2 * (n - d)
    if d >= 3:
        A += d * (d - 1) * (d - 2) // 6

B = 0
for d in cnt_b:
    if d >= 2:
        B += d * (d - 1) // 2 * (n - d)
    if d >= 3:
        B += d * (d - 1) * (d - 2) // 6

AB = 0
for a, b in edges:
    AB += (cnt_a[a] - 1) * (cnt_b[b] - 1)

ans = total_triples - A - B + AB
print(ans)
```
### Java
``` java
import java.util.*;
import java.io.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        PrintWriter pw = new PrintWriter(new BufferedWriter(new OutputStreamWriter(System.out)));
        
        int n = Integer.parseInt(br.readLine().trim());
        List<int[]> vp = new ArrayList<>();
        
        long[] cntA = new long[n + 1];
        long[] cntB = new long[n + 1];
        
        for (int i = 0; i < n; i++) {
            String[] parts = br.readLine().split(" ");
            int a = Integer.parseInt(parts[0]);
            int b = Integer.parseInt(parts[1]);
            vp.add(new int[]{a, b});
            cntA[a]++;
            cntB[b]++;
        }
        
        // 总的三元组数量 C(n, 3)
        long total = (long) n * (n - 1) * (n - 2) / 6;
        
        long A = 0, B = 0;
        
        // 计算A类型的数量（三个元素a值相同）
        for (long x : cntA) {
            if (x >= 2) {
                A += x * (x - 1) / 2 * (n - x);  // 两个a相同，一个不同
            }
            if (x >= 3) {
                A += x * (x - 1) * (x - 2) / 6;  // 三个a都相同
            }
        }
        
        // 计算B类型的数量（三个元素b值相同）
        for (long x : cntB) {
            if (x >= 2) {
                B += x * (x - 1) / 2 * (n - x);  // 两个b相同，一个不同
            }
            if (x >= 3) {
                B += x * (x - 1) * (x - 2) / 6;  // 三个b都相同
            }
        }
        
        // 计算AB类型的数量（三个元素a值相同且b值相同）
        long AB = 0;
        for (int[] pair : vp) {
            int a = pair[0];
            int b = pair[1];
            AB += (cntA[a] - 1) * (cntB[b] - 1);
        }
        
        // 容斥原理计算结果
        long ans = total - A - B + AB;
        pw.println(ans);
        pw.flush();
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
)

func main() {
	in := bufio.NewReader(os.Stdin)
	out := bufio.NewWriter(os.Stdout)
	defer out.Flush()
	
	var n int
	fmt.Fscan(in, &n)
	
	// 使用切片存储点对
	type pair struct{ a, b int }
	vp := make([]pair, n)
	
	cntA := make(map[int]int)
	cntB := make(map[int]int)
	
	for i := 0; i < n; i++ {
		var a, b int
		fmt.Fscan(in, &a, &b)
		vp[i] = pair{a, b}
		cntA[a]++
		cntB[b]++
	}
	
	// 总的三元组数量 C(n, 3)
	total := int64(n) * int64(n-1) * int64(n-2) / 6
	
	var A, B int64
	
	// 计算A类型（a值相同）
	for _, x := range cntA {
		x64 := int64(x)
		if x >= 2 {
			A += x64 * (x64 - 1) / 2 * int64(n-x)
		}
		if x >= 3 {
			A += x64 * (x64 - 1) * (x64 - 2) / 6
		}
	}
	
	// 计算B类型（b值相同）
	for _, x := range cntB {
		x64 := int64(x)
		if x >= 2 {
			B += x64 * (x64 - 1) / 2 * int64(n-x)
		}
		if x >= 3 {
			B += x64 * (x64 - 1) * (x64 - 2) / 6
		}
	}
	
	// 计算AB类型
	var AB int64
	for _, p := range vp {
		AB += int64((cntA[p.a] - 1) * (cntB[p.b] - 1))
	}
	
	// 容斥原理
	ans := total - A - B + AB
	fmt.Fprintln(out, ans)
}
```
