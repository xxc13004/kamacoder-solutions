# 广度优先搜索 (BFS) 与图的遍历
[题目链接](https://kamacoder.com/problempage.php?pid=1470)
## 思路分析
本题可以转化为一个有向图的遍历问题。将每个能源站看作图中的一个节点。如果能源站 i 能够直接开启能源站 j，那么就从节点 i 向节点 j 连一条有向边。
判断能源站 i 能否开启能源站 j 的条件是它们之间的直线距离不超过能源站 i 的半径 ri。为了避免浮点数精度问题，我们可以将距离公式两边平方，即判断 (xi - xj)^2 + (yi - yj)^2 <= ri^2。需要注意的是，由于坐标和半径最大可达 10^9，其平方可能达到 10^18 级别，所以计算距离和半径平方时必须使用 64 位整数以防止整数溢出。

因为能源站的数量 n 较小（最大为 100），我们可以直接枚举每一个能源站作为初始开启的起点。对于每一个起点，使用广度优先搜索（BFS）或深度优先搜索（DFS）遍历能够到达的所有节点，并统计遍历到的节点个数。
最后，取所有起点中能够遍历到节点数的最大值即为答案。

## 复杂度分析
1. 时间复杂度：O(T * N^3)；
2. 空间复杂度：O(N^2)。

## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

struct Station {
    long long x, y, r;
};

void solve() {
    int n;
    cin >> n;
    vector<Station> stations(n);
    for (int i = 0; i < n; ++i) {
        cin >> stations[i].x >> stations[i].y >> stations[i].r;
    }

    // 预处理有向图的邻接表
    vector<vector<int>> adj(n);
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            if (i == j) continue;
            long long dx = stations[i].x - stations[j].x;
            long long dy = stations[i].y - stations[j].y;
            long long dist_sq = dx * dx + dy * dy;
            long long r_sq = stations[i].r * stations[i].r;
            if (dist_sq <= r_sq) {
                adj[i].push_back(j);
            }
        }
    }

    int max_activated = 0;
    // 枚举每个起点
    for (int i = 0; i < n; ++i) {
        vector<bool> visited(n, false);
        queue<int> q;
        q.push(i);
        visited[i] = true;
        int count = 0;

        while (!q.empty()) {
            int curr = q.front();
            q.pop();
            count++;

            for (int neighbor : adj[curr]) {
                if (!visited[neighbor]) {
                    visited[neighbor] = true;
                    q.push(neighbor);
                }
            }
        }
        if (count > max_activated) {
            max_activated = count;
        }
    }
    cout << max_activated << "\n";
}

int main() {
    // 优化输入输出
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t;
    if (cin >> t) {
        while (t--) {
            solve();
        }
    }
    return 0;
}
```

``` python
import sys
from collections import deque

def solve():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    T = int(input_data[0])
    idx = 1
    out = []
    
    for _ in range(T):
        n = int(input_data[idx])
        idx += 1
        stations = []
        for _ in range(n):
            x, y, r = int(input_data[idx]), int(input_data[idx+1]), int(input_data[idx+2])
            stations.append((x, y, r))
            idx += 3
            
        adj = [[] for _ in range(n)]
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                dx = stations[i][0] - stations[j][0]
                dy = stations[i][1] - stations[j][1]
                if dx * dx + dy * dy <= stations[i][2] * stations[i][2]:
                    adj[i].append(j)
                    
        max_activated = 0
        for i in range(n):
            visited = [False] * n
            q = deque([i])
            visited[i] = True
            count = 0
            
            while q:
                curr = q.popleft()
                count += 1
                for neighbor in adj[curr]:
                    if not visited[neighbor]:
                        visited[neighbor] = True
                        q.append(neighbor)
                        
            if count > max_activated:
                max_activated = count
                
        out.append(str(max_activated))
        
    sys.stdout.write('\n'.join(out) + '\n')

if __name__ == '__main__':
    solve()
```

### Java
``` java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.IOException;
import java.util.StringTokenizer;
import java.util.ArrayList;
import java.util.LinkedList;
import java.util.Queue;

public class Main {
    static class Station {
        long x, y, r;
        Station(long x, long y, long r) {
            this.x = x;
            this.y = y;
            this.r = r;
        }
    }

    public static void main(String[] args) throws IOException {
        // 优化输入
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        String line = br.readLine();
        if (line == null) return;
        
        int t = Integer.parseInt(line.trim());
        StringBuilder sb = new StringBuilder();

        for (int k = 0; k < t; k++) {
            line = br.readLine();
            while(line.trim().isEmpty()) {
                line = br.readLine();
            }
            int n = Integer.parseInt(line.trim());
            Station[] stations = new Station[n];
            for (int i = 0; i < n; i++) {
                StringTokenizer st = new StringTokenizer(br.readLine());
                long x = Long.parseLong(st.nextToken());
                long y = Long.parseLong(st.nextToken());
                long r = Long.parseLong(st.nextToken());
                stations[i] = new Station(x, y, r);
            }

            ArrayList<Integer>[] adj = new ArrayList[n];
            for (int i = 0; i < n; i++) {
                adj[i] = new ArrayList<>();
                for (int j = 0; j < n; j++) {
                    if (i == j) continue;
                    long dx = stations[i].x - stations[j].x;
                    long dy = stations[i].y - stations[j].y;
                    if (dx * dx + dy * dy <= stations[i].r * stations[i].r) {
                        adj[i].add(j);
                    }
                }
            }

            int maxActivated = 0;
            for (int i = 0; i < n; i++) {
                boolean[] visited = new boolean[n];
                Queue<Integer> queue = new LinkedList<>();
                queue.offer(i);
                visited[i] = true;
                int count = 0;

                while (!queue.isEmpty()) {
                    int curr = queue.poll();
                    count++;
                    for (int neighbor : adj[curr]) {
                        if (!visited[neighbor]) {
                            visited[neighbor] = true;
                            queue.offer(neighbor);
                        }
                    }
                }
                if (count > maxActivated) {
                    maxActivated = count;
                }
            }
            sb.append(maxActivated).append("\n");
        }
        System.out.print(sb.toString());
    }
}
```

``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
)

type Station struct {
	x, y, r int64
}

func main() {
    // 优化输入
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)

	if !scanner.Scan() {
		return
	}
	t, _ := strconv.Atoi(scanner.Text())

	for k := 0; k < t; k++ {
		scanner.Scan()
		n, _ := strconv.Atoi(scanner.Text())

		stations := make([]Station, n)
		for i := 0; i < n; i++ {
			scanner.Scan()
			x, _ := strconv.ParseInt(scanner.Text(), 10, 64)
			scanner.Scan()
			y, _ := strconv.ParseInt(scanner.Text(), 10, 64)
			scanner.Scan()
			r, _ := strconv.ParseInt(scanner.Text(), 10, 64)
			stations[i] = Station{x, y, r}
		}

		adj := make([][]int, n)
		for i := 0; i < n; i++ {
			for j := 0; j < n; j++ {
				if i == j {
					continue
				}
				dx := stations[i].x - stations[j].x
				dy := stations[i].y - stations[j].y
				if dx*dx+dy*dy <= stations[i].r*stations[i].r {
					adj[i] = append(adj[i], j)
				}
			}
		}

		maxActivated := 0
		for i := 0; i < n; i++ {
			visited := make([]bool, n)
			queue := []int{i}
			visited[i] = true
			count := 0

			for len(queue) > 0 {
				curr := queue[0]
				queue = queue[1:]
				count++

				for _, neighbor := range adj[curr] {
					if !visited[neighbor] {
						visited[neighbor] = true
						queue = append(queue, neighbor)
					}
				}
			}
			if count > maxActivated {
				maxActivated = count
			}
		}
		fmt.Println(maxActivated)
	}
}
```
