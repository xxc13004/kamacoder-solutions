# 动态规划
[题目链接](https://kamacoder.com/problempage.php?pid=1451)
## 思路分析
这道题要求将所有的图片分为两组（左右两列），使得这两组的高度和之差最小。
这在算法上等价于经典的 **0-1 背包问题（子集划分问题）**。

1. 设所有图片的高度总和为 `sum`。
2. 我们的目标是从这些图片中挑出一部分放到其中一列，使其高度和尽可能的接近总和的一半，即 `sum / 2`（向下取整）。
3. 假设我们能凑出的最接近 `sum / 2` 的高度和为 `target`，那么另一列的高度和必然是 `sum - target`。
4. 因为 `target <= sum / 2`，所以两列的高度差就是：`(sum - target) - target = sum - 2 * target`。
5. **状态定义**：`dp[j]` 表示是否能从图片中挑出若干张，使其高度和恰好等于 `j`。
6. **状态转移方程**：遍历每张图片的高度 `h`，对于每个容量 `j`（从 `sum / 2` 倒序遍历到 `h`），有 `dp[j] = dp[j] || dp[j - h]`。倒序遍历是为了保证每张图片在一次判断中只被使用一次（0-1背包的精髓）。
7. 最后在 `dp` 数组中找到最大的为 `true` 的 `j`，代入公式即可。

## 复杂度分析
1. 时间复杂度：O(n * sum / 2)；
2. 空间复杂度：O(sum / 2)。
## 代码实现
### Cpp
```cpp
#include <iostream>
#include <vector>
#include <string>
#include <sstream>
#include <numeric>

using namespace std;

int main() {
    string s;
    // 1. 解析以逗号分隔的输入
    vector<int> heights;
    stringstream ss(s);
    string item;
    while (getline(ss, item, ',')) {
        if (!item.empty()) {
            heights.push_back(stoi(item));
        }
    }

    if (heights.empty()) {
        cout << 0 << endl;
        return 0;
    }

    // 2. 计算总和及目标容量
    int sum = 0;
    for (int h : heights) sum += h;
    int target = sum / 2;

    // 3. 0-1背包动态规划
    vector<bool> dp(target + 1, false);
    dp[0] = true;

    for (int h : heights) {
        for (int j = target; j >= h; j--) {
            dp[j] = dp[j] || dp[j - h];
        }
    }

    // 4. 找到最大的且能凑出的 target
    int max_achievable = 0;
    for (int j = target; j >= 0; j--) {
        if (dp[j]) {
            max_achievable = j;
            break;
        }
    }

    // 5. 输出差值
    cout << sum - 2 * max_achievable << endl;

    return 0;
}
```

### Python
```python
import sys

def main():
    # 1. 读取输入并解析
    line = sys.stdin.read().strip()        
    heights = list(map(int, line.split(',')))
        
    # 2. 计算总和及目标容量
    total_sum = sum(heights)
    target = total_sum // 2
    
    # 3. 0-1背包动态规划
    # dp[j] 表示能否凑出总高度为 j 的组合
    dp = [False] * (target + 1)
    dp[0] = True
    
    for h in heights:
        # 必须倒序遍历，保证每张图片只用一次
        for j in range(target, h - 1, -1):
            dp[j] = dp[j] or dp[j - h]
            
    # 4. 找到最接近 sum/2 的高度和
    max_achievable = 0
    for j in range(target, -1, -1):
        if dp[j]:
            max_achievable = j
            break
            
    # 5. 输出两列高度差
    print(total_sum - 2 * max_achievable)

if __name__ == "__main__":
    main()
```

### Java
```java
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String line = scanner.nextLine().trim();

        // 1. 解析输入
        String[] parts = line.split(",");
        int[] heights = new int[parts.length];
        int sum = 0;
        
        for (int i = 0; i < parts.length; i++) {
            heights[i] = Integer.parseInt(parts[i]);
            sum += heights[i];
        }

        // 2. 计算目标容量
        int target = sum / 2;

        // 3. 0-1背包动态规划
        boolean[] dp = new boolean[target + 1];
        dp[0] = true;

        for (int h : heights) {
            for (int j = target; j >= h; j--) {
                dp[j] = dp[j] || dp[j - h];
            }
        }

        // 4. 寻找最大的合法高度和
        int maxAchievable = 0;
        for (int j = target; j >= 0; j--) {
            if (dp[j]) {
                maxAchievable = j;
                break;
            }
        }

        // 5. 输出差值
        System.out.println(sum - 2 * maxAchievable);
    }
}
```

### Go
*(使用了兼容所有 Go 版本的切片及转换方式)*
```go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	line, _ := reader.ReadString('\n')
	line = strings.TrimSpace(line)

	// 1. 解析输入
	parts := strings.Split(line, ",")
	var heights []int
	sum := 0

	for _, part := range parts {
		if len(part) > 0 {
			h, _ := strconv.Atoi(part)
			heights = append(heights, h)
			sum += h
		}
	}

	if len(heights) == 0 {
		fmt.Println(0)
		return
	}

	// 2. 计算目标容量
	target := sum / 2

	// 3. 0-1背包动态规划
	dp := make([]bool, target+1)
	dp[0] = true

	for _, h := range heights {
		// 倒序遍历避免当前图片被重复使用
		for j := target; j >= h; j-- {
			dp[j] = dp[j] || dp[j-h]
		}
	}

	// 4. 找到最大的且能凑出的 target
	maxAchievable := 0
	for j := target; j >= 0; j-- {
		if dp[j] {
			maxAchievable = j
			break
		}
	}

	// 5. 输出差值
	fmt.Println(sum - 2*maxAchievable)
}
```
