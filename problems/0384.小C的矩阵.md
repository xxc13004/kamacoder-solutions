# 贪心与排序不等式
[题目链接](https://kamacoder.com/problempage.php?pid=1464)
## 思路分析
这道题目要求我们通过任意次行变换和列变换，使得矩阵的权值和最大化。权值和的计算公式为矩阵所有元素的 wij = vij * (ai + bj) 的总和。

首先我们从数学的角度把总权值 W 展开：
W = sum(vij * (ai + bj)) = sum(vij * ai) + sum(vij * bj)

观察这两项，我们可以将其分别按行和按列进行提取合并：
第一部分：对于第 i 行，ai 是固定的，因此这一行的和可以写为 ai * sum(vij) (j 从 1 到 m)。即 ai * 第 i 行的元素和。
第二部分：对于第 j 列，bj 是固定的，因此这一列的和可以写为 bj * sum(vij) (i 从 1 到 n)。即 bj * 第 j 列的元素和。

所以，总权值可以转化为：
W = sum(ai * 第 i 行的元素和) + sum(bj * 第 j 列的元素和)

题目中说可以进行任意次“行变换”和“列变换”。行变换的本质是可以将矩阵的行任意打乱重排，列变换的本质是可以将矩阵的列任意打乱重排。
关键点在于：**交换任意两行，只会改变每一行作为一个整体所对应的 ai，而不会改变任何一列的元素总和；同理，交换任意两列，只会改变每一列整体对应的 bj，不会改变任何一行的元素总和。**

因此，行变换和列变换是完全相互独立的。
根据经典的“排序不等式”（大配大，小配小，和最大），要使得乘积之和最大，我们只需要：
1. 计算出初始矩阵的每一行的元素和，并存入数组 row_sums。
2. 计算出初始矩阵的每一列的元素和，并存入数组 col_sums。
3. 将 a 数组和 row_sums 数组都按从小到大排序，求点积。
4. 将 b 数组和 col_sums 数组都按从小到大排序，求点积。
5. 将两部分点积相加，即为最大权值和。

注意：由于数据规模较大，乘积之和会超出 32 位整数的范围，需要使用 64 位整数。

## 复杂度分析
1. 时间复杂度：O(n * m + nlogn + mlogm)；
2. 空间复杂度：O(n + m)。

## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    // 优化输入输出
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<long long> a(n);
    for (int i = 0; i < n; ++i) cin >> a[i];

    vector<long long> b(m);
    for (int j = 0; j < m; ++j) cin >> b[j];

    vector<long long> row_sums(n, 0);
    vector<long long> col_sums(m, 0);

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            long long val;
            cin >> val;
            row_sums[i] += val;
            col_sums[j] += val;
        }
    }

    sort(a.begin(), a.end());
    sort(row_sums.begin(), row_sums.end());
    sort(b.begin(), b.end());
    sort(col_sums.begin(), col_sums.end());

    long long max_weight = 0;
    for (int i = 0; i < n; ++i) {
        max_weight += a[i] * row_sums[i];
    }
    for (int j = 0; j < m; ++j) {
        max_weight += b[j] * col_sums[j];
    }

    cout << max_weight << "\n";

    return 0;
}
```

### Python
``` python
import sys

def main():
    # 一次性读取所有输入并切分为数组
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    n = int(input_data[0])
    m = int(input_data[1])
    
    ptr = 2
    a = [int(x) for x in input_data[ptr : ptr+n]]
    ptr += n
    
    b = [int(x) for x in input_data[ptr : ptr+m]]
    ptr += m
    
    row_sums = [0] * n
    col_sums = [0] * m
    
    for i in range(n):
        for j in range(m):
            val = int(input_data[ptr])
            ptr += 1
            row_sums[i] += val
            col_sums[j] += val
            
    a.sort()
    row_sums.sort()
    b.sort()
    col_sums.sort()
    
    ans = 0
    for i in range(n):
        ans += a[i] * row_sums[i]
    for j in range(m):
        ans += b[j] * col_sums[j]
        
    print(ans)

if __name__ == '__main__':
    main()
```

### Java
``` java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.IOException;
import java.util.StringTokenizer;
import java.util.Arrays;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = null;
        
        String line = br.readLine();
        while (line != null && line.trim().isEmpty()) {
            line = br.readLine();
        }
        if (line == null) return;
        
        st = new StringTokenizer(line);
        int n = Integer.parseInt(st.nextToken());
        int m = Integer.parseInt(st.nextToken());
        
        long[] a = new long[n];
        long[] b = new long[m];
        
        for (int i = 0; i < n; i++) {
            if (!st.hasMoreTokens()) st = new StringTokenizer(br.readLine());
            a[i] = Long.parseLong(st.nextToken());
        }
        
        for (int j = 0; j < m; j++) {
            if (!st.hasMoreTokens()) st = new StringTokenizer(br.readLine());
            b[j] = Long.parseLong(st.nextToken());
        }
        
        long[] row_sums = new long[n];
        long[] col_sums = new long[m];
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (!st.hasMoreTokens()) st = new StringTokenizer(br.readLine());
                long val = Long.parseLong(st.nextToken());
                row_sums[i] += val;
                col_sums[j] += val;
            }
        }
        
        Arrays.sort(a);
        Arrays.sort(row_sums);
        Arrays.sort(b);
        Arrays.sort(col_sums);
        
        long max_weight = 0;
        for (int i = 0; i < n; i++) {
            max_weight += a[i] * row_sums[i];
        }
        for (int j = 0; j < m; j++) {
            max_weight += b[j] * col_sums[j];
        }
        
        System.out.println(max_weight);
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"sort"
	"strconv"
)

func main() {
    // 优化输入
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)
    // 预留足够大的buffer，应对极大量数据输入
	buf := make([]byte, 1024*1024*10)
	scanner.Buffer(buf, 1024*1024*10)

	if !scanner.Scan() {
		return
	}
	n, _ := strconv.Atoi(scanner.Text())
	scanner.Scan()
	m, _ := strconv.Atoi(scanner.Text())

	a := make([]int, n)
	for i := 0; i < n; i++ {
		scanner.Scan()
		a[i], _ = strconv.Atoi(scanner.Text())
	}

	b := make([]int, m)
	for j := 0; j < m; j++ {
		scanner.Scan()
		b[j], _ = strconv.Atoi(scanner.Text())
	}

	row_sums := make([]int, n)
	col_sums := make([]int, m)

	for i := 0; i < n; i++ {
		for j := 0; j < m; j++ {
			scanner.Scan()
			val, _ := strconv.Atoi(scanner.Text())
			row_sums[i] += val
			col_sums[j] += val
		}
	}

	sort.Ints(a)
	sort.Ints(row_sums)
	sort.Ints(b)
	sort.Ints(col_sums)

	var max_weight int64 = 0
	for i := 0; i < n; i++ {
		max_weight += int64(a[i]) * int64(row_sums[i])
	}
	for j := 0; j < m; j++ {
		max_weight += int64(b[j]) * int64(col_sums[j])
	}

	fmt.Println(max_weight)
}
```
