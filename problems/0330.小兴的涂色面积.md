[题目链接](https://kamacoder.com/problempage.php?pid=1410)
## 思路分析
每次操作完面积的变化量都是3/4个圆，并且每次操作完的的半径都会减半。奇数次是涂黑也就是加上这个面积，偶数次是涂白，就是减去这个面积。
## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(1)。

``` cpp
}
```

### Python
``` python
import sys
import math

def main() -> None:
    input_data = sys.stdin.read().strip().split()
    if not input_data:
        return
    
    r = float(input_data[0])
    n = int(input_data[1])
    
    pi = 3.141592654
    ans = pi * r * r * 3 / 4  # 第一次的面积
    
    op = -1  # 符号位，表示这一次是减少还是增加
    current_r = r / 2.0
    
    for i in range(2, n + 1):
        ans += op * pi * current_r * current_r * 3 / 4
        current_r /= 2.0
        op = -op  # 切换符号
    
    # 保留6位小数输出
    print(f"{ans:.7f}")

if __name__ == "__main__":
    main()
```

### Go
``` go
package main

import (
    "bufio"
    "fmt"
    "os"
    "strconv"
    "strings"
)

func main() {
    reader := bufio.NewReader(os.Stdin)
    writer := bufio.NewWriter(os.Stdout)
    defer writer.Flush()
    
    line, _ := reader.ReadString('\n')
    line = strings.TrimSpace(line)
    parts := strings.Fields(line)
    
    if len(parts) < 2 {
        return
    }
    
    r, _ := strconv.ParseFloat(parts[0], 64)
    n, _ := strconv.Atoi(parts[1])
    
    const pi = 3.141592654
    ans := pi * r * r * 3.0 / 4.0 // 第一次的面积
    
    op := -1 // 符号位，表示这一次是减少还是增加
    currentR := r / 2.0
    
    for i := 2; i <= n; i++ {
        ans += float64(op) * pi * currentR * currentR * 3.0 / 4.0
        currentR /= 2.0
        op = -op // 切换符号
    }
    
    fmt.Fprintf(writer, "%.7f\n", ans)
}
```
