# 数学 + 枚举
[题目链接](https://kamacoder.com/problempage.php?pid=1440)
## 思路分析

整个平面的范围是[0, N], [0, M]，对于每一个像素点进行表示(i, j)为左上角的坐标，像素为[x ∈ (i, i+1), y ∈ (j, j+1)]。

像素点中每个子区域的中心坐标为:

1. x = i + (a + 0.5) / S
2. y = j + (b + 0.5) / S

其中a, b为0...S-1。

有了上述表示过后，我们可以使用**叉积法**判断点是否在三角形内。

对于点P，计算叉积：

1. cross(AB, AP)
2. cross(BC, BP)
3. cross(CA, CP)

- 如果结果三个符号都相同，则在内部；
- 如果为0，则表示在边上。

依次枚举每个像素，然后枚举该像素的每个部分即可。

## 复杂度分析
1. 时间复杂度：O(M * N * S * S)；
2. 空间复杂度：O(1)。

### Cpp
``` cpp
#include <bits/stdc++.h>
#define endl '\n'
const int N = 500000, MOD = 1000000007;
using namespace std;
using LL = long long;
using PIL = pair<int, LL>;
using PII = pair<int, int>;
using T3I = tuple<int, int, int>;
int T = 0, dir[]{1, 0, -1, 0, 1};
const double EPS = 1e-9;
// 计算叉积 (B - A) x (P - A)
double cross(double ax, double ay,
             double bx, double by,
             double px, double py) {
    return (bx - ax) * (py - ay) - (by - ay) * (px - ax);
}
// 判断点是否在三角形内（含边）
bool inTriangle(double ax, double ay,
                double bx, double by,
                double cx, double cy,
                double px, double py) {

    double c1 = cross(ax, ay, bx, by, px, py);
    double c2 = cross(bx, by, cx, cy, px, py);
    double c3 = cross(cx, cy, ax, ay, px, py);

    if ((c1 >= -EPS && c2 >= -EPS && c3 >= -EPS) ||
        (c1 <= EPS && c2 <= EPS && c3 <= EPS))
        return true;

    return false;
}
int main() {
    cin.tie(nullptr) -> sync_with_stdio(false);
    int N, M, S;
    cin >> N >> M >> S;
    double ax, ay, bx, by, cx, cy;
    cin >> ax >> ay;
    cin >> bx >> by;
    cin >> cx >> cy;
    vector<long long> result(S * S + 1, 0);
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < M; j++) {
            int cnt = 0;  // 统计每个像素点满足条件的区域数量
            for (int a = 0; a < S; a++) {
                for (int b = 0; b < S; b++) {
                    double px = i + (a + 0.5) / S;
                    double py = j + (b + 0.5) / S;
                    if (inTriangle(ax, ay, bx, by, cx, cy, px, py))
                        cnt++;
                }
            }
            result[cnt]++;
        }
    }
    // 输出结果
    for (int i = 0; i <= S * S; i++) {
        cout << result[i] << " ";
    }
    return 0;
}
```

### Go
``` go
package main

import (
	"bufio"
	"os"
	"strconv"
)

func cross(ax, ay, bx, by, px, py float64) float64 {
	return (bx-ax)*(py-ay) - (by-ay)*(px-ax)
}

func inTriangle(ax, ay, bx, by, cx, cy, px, py, eps float64) bool {
	c1 := cross(ax, ay, bx, by, px, py)
	c2 := cross(bx, by, cx, cy, px, py)
	c3 := cross(cx, cy, ax, ay, px, py)
	
	return (c1 >= -eps && c2 >= -eps && c3 >= -eps) ||
		(c1 <= eps && c2 <= eps && c3 <= eps)
}

func main() {
	// 高性能输入：使用 bufio.Scanner 和 ScanWords 模式
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)
	
	// 设置大缓冲区（16MB）
	buf := make([]byte, 0, 16*1024*1024)
	scanner.Buffer(buf, 32*1024*1024)
	
	// 辅助函数：读取下一个单词并转换为 int
	nextInt := func() int {
		scanner.Scan()
		val, _ := strconv.Atoi(scanner.Text())
		return val
	}
	
	// 辅助函数：读取下一个单词并转换为 float64
	nextFloat := func() float64 {
		scanner.Scan()
		val, _ := strconv.ParseFloat(scanner.Text(), 64)
		return val
	}
	
	N := nextInt()
	M := nextInt()
	S := nextInt()
	
	ax := nextFloat()
	ay := nextFloat()
	bx := nextFloat()
	by := nextFloat()
	cx := nextFloat()
	cy := nextFloat()
	
	result := make([]int, S*S+1)
	eps := 1e-9
	
	for i := 0; i < N; i++ {
		for j := 0; j < M; j++ {
			cnt := 0
			for a := 0; a < S; a++ {
				for b := 0; b < S; b++ {
					px := float64(i) + (float64(a)+0.5)/float64(S)
					py := float64(j) + (float64(b)+0.5)/float64(S)
					if inTriangle(ax, ay, bx, by, cx, cy, px, py, eps) {
						cnt++
					}
				}
			}
			result[cnt]++
		}
	}
	
	// 高性能输出：使用 bufio.Writer
	writer := bufio.NewWriterSize(os.Stdout, 16*1024*1024)
	defer writer.Flush()
	
	for i := 0; i <= S*S; i++ {
		if i > 0 {
			writer.WriteByte(' ')
		}
		writer.WriteString(strconv.Itoa(result[i]))
	}
	writer.WriteByte('\n')
}
```
