# 模拟
[题目链接](https://kamacoder.com/problempage.php?pid=1368)
## 思路分析
分情况讨论：

1. 有下划线_:
    - 有大写字母：因为下划线不能和大写字母同时出现，所以这里直接输出"indistinct";
    - 无大写字母：判断是否第一个或者最后一个字符是下划线_，或者是否字符串中存在两个以上的下划线_连在一起的情况，也是"indistinct"，否则就是下划线命名法，直接输出即可。
2. 无下划线_：
    - 第一个字符是大写字母：不符合定义，输出"indistinct"；
    - 第一个字符不是大写字母：符合小驼峰命名法，直接将其转化为下划线命名法输出即可，每次碰到大写字母就在其前面添加一个_，然后将其变成小写字母添加到字符串。
## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(n)。

``` cpp
    int n;
    cin >> n;
    vector<string> strs(n);
    for (auto &s : strs) cin >> s;
    for (auto &s : strs) {
        if (s.find('_') != -1) { // 包含下划线
            // 判断是否包含大写字母
            if (any_of(s.begin(), s.end(), [](char c) { return isupper(c); })) {
                cout << "indistinct" << endl;
            } else {
                if (s[0] == '_' || s.back() == '_' || s.find("__") != -1) {
                    cout << "indistinct" << endl;
                } else {
                    cout << s << endl;
                }
            }
        } else { // 不包含下划线
            if (s[0] >= 'A' && s[0] <= 'Z') {
                cout << "indistinct" << endl;
            } else {
                // 转换为下划线命名法
                string res;
                for (char c : s) {
                    if (c >= 'A' && c <= 'Z') {
                        res += '_';
                        res += tolower(c);
                    } else {
                        res += c;
                    }
                }
                cout << res << endl;
            }
        }
    }
}
```

``` python
n = int(input())
strs = [input() for _ in range(n)]

for s in strs:
    if s.find('_') != -1:  # 包含下划线
        # 判断是否包含大写字母
        if any(c.isupper() for c in s):
            print('indistinct')
        else:
            if s[0] == '_' or s[-1] == '_' or '__' in s:
                print('indistinct')
            else:
                print(s)
    else:
        if s[0].isupper():
            print('indistinct')
        else:
            # 将字符串转化为下划线命名法
            result = []
            for c in s:
                if c.isupper():
                    result.append('_')
                    result.append(c.lower())
                else:
                    result.append(c)
            print(''.join(result))
```

### Java
``` java
import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = scanner.nextInt();
        scanner.nextLine(); // 读取换行符
        String[] strs = new String[n];
        for (int i = 0; i < n; i++) {
            strs[i] = scanner.nextLine();
        }
        for (String s : strs) {
            if (s.contains("_")) { // 包含下划线
                // 判断是否包含大写字母
                boolean containsUpper = false;
                for (char c : s.toCharArray()) {
                    if (Character.isUpperCase(c)) {
                        containsUpper = true;
                        break;
                    }
                }
                if (containsUpper) {
                    System.out.println("indistinct");
                } else {
                    if (s.charAt(0) == '_' || s.charAt(s.length() - 1) == '_' || s.contains("__")) {
                        System.out.println("indistinct");
                    } else {
                        System.out.println(s);
                    }
                }
            } else { // 不包含下划线
                if (Character.isUpperCase(s.charAt(0))) {
                    System.out.println("indistinct");
                } else {
                    // 转换为下划线命名法
                    StringBuilder res = new StringBuilder();
                    for (char c : s.toCharArray()) {
                        if (Character.isUpperCase(c)) {
                            res.append("_");
                            res.append(Character.toLowerCase(c));
                        } else {
                            res.append(c);
                        }
                    }
                    System.out.println(res.toString());
                }
            }
        }
        scanner.close();
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strings"
	"unicode"
)

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Scan()
	var n int
	fmt.Sscanf(scanner.Text(), "%d", &n)
	strs := make([]string, n)
	for i := 0; i < n; i++ {
		scanner.Scan()
		strs[i] = scanner.Text()
	}
	for _, s := range strs {
		if strings.Contains(s, "_") { // 包含下划线
			// 判断是否包含大写字母
			containsUpper := false
			for _, c := range s {
				if unicode.IsUpper(c) {
					containsUpper = true
					break
				}
			}
			if containsUpper {
				fmt.Println("indistinct")
			} else {
				if s[0] == '_' || s[len(s)-1] == '_' || strings.Contains(s, "__") {
					fmt.Println("indistinct")
				} else {
					fmt.Println(s)
				}
			}
		} else { // 不包含下划线
			if unicode.IsUpper(rune(s[0])) {
				fmt.Println("indistinct")
			} else {
				// 转换为下划线命名法
				var res string
				for _, c := range s {
					if unicode.IsUpper(c) {
						res += "_"
						res += string(unicode.ToLower(c))
					} else {
						res += string(c)
					}
				}
				fmt.Println(res)
			}
		}
	}
}
```
