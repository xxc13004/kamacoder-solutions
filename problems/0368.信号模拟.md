# 数学与组合概率
[题目链接](https://kamacoder.com/problempage.php?pid=1448)
## 思路分析

<img width="973" height="1108" alt="image" src="https://github.com/user-attachments/assets/9f316fd6-dcc9-4e19-b349-ab8cffd0f6ae" />


1. **问题转化**：
   题意描述了有 $2n$ 个仪器，左端 $2n$ 个接线点被随机两两配对，右端 $2n$ 个接线点也被随机两两配对。这种配对方式实际上是构造了一个以这 $2n$ 个仪器为主体（每个仪器视作一个拥有 2 个度数，左端出右端进的节点）的**2-正则图**。
   因为每个点度数为 2，所以最后必然会形成若干个独立的循环（即连通块/环）。
   
2. **所需信号源的最少数量 $x$**：
   对于每一个独立循环（环），只需要放置 **1** 个信号源，信号就能顺着该环路传遍整个循环，让该环内的所有接收器都获得信号。如果环内全是信号源（即 $0$ 个接收器），也是 $1$ 个信号源起步（或环中所有均可满足）。总体来看，让图中所有接收器均能获取到信号，最少需要的信号源总数 $x$ 就等于这个 2-正则图中**环的数量（循环个数）**。

3. **图的循环个数分布**：
   我们要求的即是由两个随机完美匹配组成的图的环数 $C_n$ 的方差。
   通过多项式推导和生成函数分析，已知在 $2n$ 个顶点的完全图下，由两个随机完美匹配构成环数的生成函数为：
   $$ G_n(y) = \prod_{i=0}^{n-1} \frac{y + 2i}{2i + 1} $$
   这表明，环数 $C_n$ 等价于 $n$ 个相互独立的**伯努利随机变量**的和：
   $$ C_n = \sum_{i=0}^{n-1} X_i $$
   其中 $X_i \sim \text{Bernoulli}\left( \frac{1}{2i+1} \right)$。

4. **方差计算公式**：
   由于相互独立，和的方差等于方差的和。单个伯努利变量的方差为 $p_i(1 - p_i)$：
   $$ Var(X_i) = \frac{1}{2i+1} \left( 1 - \frac{1}{2i+1} \right) = \frac{2i}{(2i+1)^2} $$
   因此，环数的方差 $Var(C_n)$ 就是：
   $$ Var(C_n) = \sum_{i=0}^{n-1} \frac{2i}{(2i+1)^2} \pmod{998244353} $$

5. **算法设计**：
   由于最大询问次数 $T \le 10^4$ 且 $N \le 10^6$，若每次询问都循环求和必将超时。
   所以我们需要预处理出 $N \in [1, 10^6]$ 的所有答案前缀和。其中计算分数取模需要利用乘法逆元，为使整体预处理达到 $O(N)$ 级别，可利用线性求逆元算法预处理所有到 $2 \times 10^6$ 的逆元。

## 复杂度分析
1. 时间复杂度：$O(N + T)$。预处理递推数组和所有的乘法逆元耗时 $O(N)$，每次查询可以直接 $O(1)$ 给出结果。
2. 空间复杂度：$O(N)$。
## 代码实现

### Cpp
``` cpp
#include <iostream>
#include <vector>

using namespace std;

const int M = 998244353;
const int MAXN = 1000000;
long long ans[MAXN + 5];
long long inv[2 * MAXN + 5];

void precompute() {
    // 线性预处理逆元
    inv[1] = 1;
    for (int i = 2; i <= 2 * MAXN + 1; ++i) {
        inv[i] = (M - M / i) * inv[M % i] % M;
    }
    // 预处理前缀方差和
    ans[1] = 0;
    for (long long n = 2; n <= MAXN; ++n) {
        long long num = 2 * (n - 1);
        long long den_inv = inv[2 * n - 1];
        long long term = num * den_inv % M * den_inv % M;
        ans[n] = (ans[n - 1] + term) % M;
    }
}

int main() {
    // 优化输入输出
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    precompute();
    
    int T;
    if (cin >> T) {
        while (T--) {
            int n;
            cin >> n;
            cout << ans[n] << "\n";
        }
    }
    return 0;
}
```

``` java
import java.io.InputStream;
import java.io.IOException;
import java.io.PrintWriter;

public class Main {
    static final int M = 998244353;
    static final int MAXN = 1000000;
    static long[] ans = new long[MAXN + 1];
    static long[] inv = new long[2 * MAXN + 5];

    public static void main(String[] args) throws IOException {
        // 线性预处理逆元
        inv[1] = 1;
        for (int i = 2; i <= 2 * MAXN + 1; i++) {
            inv[i] = (M - M / i) * inv[M % i] % M;
        }
        
        // 预处理前缀方差和
        ans[1] = 0;
        for (int n = 2; n <= MAXN; n++) {
            long num = 2L * (n - 1);
            long den_inv = inv[2 * n - 1];
            long term = num * den_inv % M * den_inv % M;
            ans[n] = (ans[n - 1] + term) % M;
        }

        FastScanner scanner = new FastScanner(System.in);
        PrintWriter out = new PrintWriter(System.out);
        
        int T = scanner.nextInt();
        while (T-- > 0) {
            int n = scanner.nextInt();
            out.println(ans[n]);
        }
        out.flush();
    }

    // Fast IO Scanner
    static class FastScanner {
        InputStream is;
        byte[] buffer = new byte[1024 * 8];
        int head = 0, tail = 0;

        public FastScanner(InputStream is) {
            this.is = is;
        }

        private int read() throws IOException {
            if (head >= tail) {
                head = 0;
                tail = is.read(buffer, 0, buffer.length);
                if (tail <= 0) return -1;
            }
            return buffer[head++];
        }

        public int nextInt() throws IOException {
            int c = read();
            while (c <= 32) {
                if (c == -1) return -1;
                c = read();
            }
            int res = 0;
            while (c > 32) {
                res = res * 10 + c - '0';
                c = read();
            }
            return res;
        }
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
)

const M = 998244353
const MAXN = 1000000

func main() {
	// 线性预处理逆元
	inv := make([]int64, 2*MAXN+5)
	inv[1] = 1
	for i := 2; i <= 2*MAXN+1; i++ {
		inv[i] = int64(M-M/i) * inv[M%i] % M
	}

	// 预处理前缀方差和
	ans := make([]int64, MAXN+1)
	ans[1] = 0
	for n := 2; n <= MAXN; n++ {
		num := int64(2 * (n - 1))
		den_inv := inv[2*n-1]
		term := num * den_inv % M * den_inv % M
		ans[n] = (ans[n-1] + term) % M
	}

	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)

	if !scanner.Scan() {
		return
	}
	T, _ := strconv.Atoi(scanner.Text())

	writer := bufio.NewWriter(os.Stdout)
	defer writer.Flush()

	for i := 0; i < T; i++ {
		scanner.Scan()
		n, _ := strconv.Atoi(scanner.Text())
		fmt.Fprintln(writer, ans[n])
	}
}
```
