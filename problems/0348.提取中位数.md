# 排序 + 双指针
[题目链接](https://kamacoder.com/problempage.php?pid=1428)
## 思路分析

按照数组长度进行奇偶分类即可。

1. 如果是奇数。那么就先输出中间的数字，然后左指针指向前一个数，右指针指向后一个数。最后按照左右交替的方式依次输出剩余的数。
2. 如果是偶数。那就左指针指向n/2-1，右指针指向n/2。最后按照左右交替的方式输出每一个数。

## 复杂度分析
1. 时间复杂度：O(nlogn)；
2. 空间复杂度：O(1)。

### Python
``` python
import sys

def main():
    input = sys.stdin.readline
    n = int(input())
    nums = list(map(int, input().split()))
    nums.sort()
    
    l = n // 2 - 1
    r = n // 2
    result = []
    
    if n & 1:
        result.append(str(nums[n >> 1]))
        n -= 1
        r += 1
    
    for i in range(n):
        if i & 1:
            result.append(str(nums[r]))
            r += 1
        else:
            result.append(str(nums[l]))
            l -= 1
    
    print(' '.join(result))

if __name__ == "__main__":
    main()
```

``` java
import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        PrintWriter pw = new PrintWriter(System.out);
        
        // 读取n
        int n = Integer.parseInt(br.readLine());
        
        // 读取数组
        String[] tokens = br.readLine().split(" ");
        int[] nums = new int[n];
        for (int i = 0; i < n; i++) {
            nums[i] = Integer.parseInt(tokens[i]);
        }
        
        // 排序
        Arrays.sort(nums);
        
        int l = n / 2 - 1;
        int r = n / 2;
        StringBuilder sb = new StringBuilder();
        
        if ((n & 1) == 1) {
            sb.append(nums[n >> 1]).append(' ');
            n--;
            r++;
        }
        
        for (int i = 0; i < n; i++) {
            if ((i & 1) == 1) {
                sb.append(nums[r]).append(' ');
                r++;
            } else {
                sb.append(nums[l]).append(' ');
                l--;
            }
        }
        
        pw.println(sb.toString().trim());
        pw.flush();
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"sort"
	"strconv"
	"strings"
)

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)
	
	// 读取n
	scanner.Scan()
	n, _ := strconv.Atoi(scanner.Text())
	
	// 读取数组
	nums := make([]int, n)
	for i := 0; i < n; i++ {
		scanner.Scan()
		nums[i], _ = strconv.Atoi(scanner.Text())
	}
	
	// 排序
	sort.Ints(nums)
	
	l := n/2 - 1
	r := n / 2
	result := make([]string, 0, n)
	
	if n&1 == 1 {
		result = append(result, strconv.Itoa(nums[n>>1]))
		n--
		r++
	}
	
	for i := 0; i < n; i++ {
		if i&1 == 1 {
			result = append(result, strconv.Itoa(nums[r]))
			r++
		} else {
			result = append(result, strconv.Itoa(nums[l]))
			l--
		}
	}
	
	fmt.Println(strings.Join(result, " "))
}
```
