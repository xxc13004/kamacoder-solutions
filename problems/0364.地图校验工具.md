# DFS/BFS
[题目链接](https://kamacoder.com/problempage.php?pid=1444)
## 思路分析

从任意一个节点出发（假设是0），将经过的节点都标记为true，最后遍历vis数组。

1. 如果有为false的节点，那么说明不是联通的；

java的为BFS的版本，其他语言都是DFS。这里java用DFS递归会爆栈，不知道为啥。

## 复杂度分析
1. 时间复杂度：O(n + m)；
2. 空间复杂度：O(n + m)。
## 代码实现
### Cpp
``` cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<vector<int>> g(n);
    
    for (int i = 0; i < m; i++) {
        int a, b;
        cin >> a >> b;
        g[a].push_back(b);
        g[b].push_back(a);
    }

    vector<bool> vis(n, false);

    function<void(int)> dfs = [&](int u) {
        vis[u] = true;
        for (int v : g[u]) {
            if (!vis[v]) dfs(v);
        }
    };

    dfs(0);  // 从0号节点开始

    for (int i = 0; i < n; i++) {
        if (!vis[i]) {
            cout << "No";
            return 0;
        }
    }

    cout << "Yes";

    return 0;
}
```
### Python
``` python
import sys
sys.setrecursionlimit(10**7)

def main():
    input = sys.stdin.readline

    n, m = map(int, input().split())

    # 建立邻接表
    g = [[] for _ in range(n)]

    for _ in range(m):
        a, b = map(int, input().split())
        g[a].append(b)
        g[b].append(a)

    visited = [False] * n

    # 深度优先搜索
    def dfs(u):
        visited[u] = True
        for v in g[u]:
            if not visited[v]:
                dfs(v)

    # 从0号节点开始
    dfs(0)

    # 判断是否所有节点都访问到
    for v in visited:
        if not v:
            print("No")
            return

    print("Yes")


if __name__ == "__main__":
    main()
```

### Java => BFS
``` java
import java.io.*;
import java.util.*;

public class Main {

    public static void main(String[] args) throws Exception {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());

        int n = Integer.parseInt(st.nextToken());
        int m = Integer.parseInt(st.nextToken());

        List<Integer>[] g = new ArrayList[n];
        for (int i = 0; i < n; i++) g[i] = new ArrayList<>();

        for (int i = 0; i < m; i++) {
            st = new StringTokenizer(br.readLine());
            int a = Integer.parseInt(st.nextToken());
            int b = Integer.parseInt(st.nextToken());

            g[a].add(b);
            g[b].add(a);
        }

        boolean[] vis = new boolean[n];

        Queue<Integer> q = new ArrayDeque<>();
        q.add(0);
        vis[0] = true;

        while (!q.isEmpty()) {
            int u = q.poll();
            for (int v : g[u]) {
                if (!vis[v]) {
                    vis[v] = true;
                    q.add(v);
                }
            }
        }

        for (boolean v : vis) {
            if (!v) {
                System.out.println("No");
                return;
            }
        }

        System.out.println("Yes");
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
)

var g [][]int
var vis []bool

func dfs(u int) {
	vis[u] = true
	for _, v := range g[u] {
		if !vis[v] {
			dfs(v)
		}
	}
}

func main() {
	in := bufio.NewReader(os.Stdin)

	var n, m int
	fmt.Fscan(in, &n, &m)

	g = make([][]int, n)

	for i := 0; i < m; i++ {
		var a, b int
		fmt.Fscan(in, &a, &b)

		g[a] = append(g[a], b)
		g[b] = append(g[b], a)
	}

	vis = make([]bool, n)

	dfs(0)

	for i := 0; i < n; i++ {
		if !vis[i] {
			fmt.Println("No")
			return
		}
	}

	fmt.Println("Yes")
}
```
