## 思路分析
我们可以分为以下几种情况进行讨论：

1. 加法、乘法：因为满足交换律，所以是一样的，永远都是等于；
2. 减法、除法：输出的符号与a与b的大小同号；
3. 乘方：由于数据比较大，所以可以先计算对数，a^b 与 b^a 的大小等价于两边都取对数的大小，即blog(a) 与 alog(b)，所以最终的结果就是比较这两个数的大小。

## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(1)。

### Java
``` java
import java.io.*;
import java.util.*;

public class Main {
    static char getop(double a, double b) {
        if (a > b) return '>';
        if (a < b) return '<';
        return '=';
    }
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        
        int n = Integer.parseInt(br.readLine());
        
        
        
        while (n-- > 0) {
            String[] parts = br.readLine().split(" ");
            int a = Integer.parseInt(parts[0]);
            int b = Integer.parseInt(parts[1]);
            char op = parts[2].charAt(0);
            
            switch (op) {
                case '+':
                    bw.write("=\n");
                    break;
                case '-':
                    bw.write(getop(a, b) + "\n");
                    break;
                case '*':
                    bw.write("=\n");
                    break;
                case '/':
                    bw.write(getop(a, b) + "\n");
                    break;
                default:
                    double x = b * Math.log10(a);
                    double y = a * Math.log10(b);
                    bw.write(getop(x, y) + "\n");
            }
        }
        bw.flush();
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"math"
	"os"
	"strconv"
	"strings"
)

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	writer := bufio.NewWriter(os.Stdout)
	defer writer.Flush()

	getop := func(a, b float64) byte {
		if a > b {
			return '>'
		}
		if a < b {
			return '<'
		}
		return '='
	}

	scanner.Scan()
	n, _ := strconv.Atoi(scanner.Text())

	for i := 0; i < n; i++ {
		scanner.Scan()
		parts := strings.Fields(scanner.Text())
		a, _ := strconv.Atoi(parts[0])
		b, _ := strconv.Atoi(parts[1])
		op := parts[2][0]

		switch op {
		case '+':
			fmt.Fprintln(writer, "=")
		case '-':
			fmt.Fprintf(writer, "%c\n", getop(float64(a), float64(b)))
		case '*':
			fmt.Fprintln(writer, "=")
		case '/':
			fmt.Fprintf(writer, "%c\n", getop(float64(a), float64(b)))
		default:
			x := float64(b) * math.Log10(float64(a))
			y := float64(a) * math.Log10(float64(b))
			fmt.Fprintf(writer, "%c\n", getop(x, y))
		}
	}
}
```
