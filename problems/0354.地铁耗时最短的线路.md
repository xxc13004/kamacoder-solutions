# Dijkstra
[题目链接](https://kamacoder.com/problempage.php?pid=1434)
## 思路分析

按照输入建立无向图，然后就是求源节点到目标节点的最短路径。

可以使用优先队列实现Dijkstra算法。但是这题要求完整的路径。所以我们需要在每次更新最短路径的时候，记录当前节点的前驱节点，最后从目标节点到起始节点，反向求出路径，再反向输出就是完整的路径。

``` cpp
#include<bits/stdc++.h>
#define endl '\n'
const int N = 100005, MOD = 1000000007;
using namespace std;
using LL = long long;
using PIL = pair<int, LL>;
using PII = pair<int, int>;
using T3I = tuple<int, int, int>;
int t = 0, dir[]{1, 0, -1, 0, 1};
int main() {
    cin.tie(nullptr) -> sync_with_stdio(false);
    int n, t;
    cin >> n;
    vector<vector<pair<char, int>>> vp(128);
    string source, target, x, y;
    cin >> source >> target;
    cin >> x;
    while (x != "0000") {
        cin >> y >> t;  // 读取目标节点和路径的权重
        char cx = x[0], cy = y[0];  // 字符串转化为单个字符
        vp[cx].emplace_back(cy, t);
        vp[cy].emplace_back(cx, t);
        cin >> x;  // 读取下一行的第一个字符串
    }
    priority_queue<pair<int, char>, vector<pair<int, char>>, greater<pair<int, char>>> pq;
    vector<char> parent(128);  // 存储当前节点的父亲的信息
    vector<int> dist(128, MOD);  // 存储源节点到其他节点的距离
    pq.emplace(0, source[0]);
    dist[source[0]] = 0;
    while (!pq.empty()) {
        auto [d, node] = pq.top(); pq.pop();
        if (dist[node] < d) continue;
        for (auto [nxt, t] : vp[node]) {
            int nd = d + t;
            if (nd < dist[nxt]) {
                dist[nxt] = nd;
                parent[nxt] = node;
                pq.emplace(nd, nxt);
            }
        }
    }
    vector<char> ans;
    for (char i = target[0]; i != source[0]; i = parent[i]) {
        ans.push_back(i);
    }
    ans.push_back(source[0]);
    reverse(ans.begin(), ans.end());
    for (auto e : ans) cout << e << ' ';
    return 0;
}
```

### Python
``` python
import sys
import heapq

INF = 10**18

def main():
    input = sys.stdin.readline

    n = int(input().strip())  # 不直接使用
    source, target = input().split()

    vp = [[] for _ in range(128)]

    while True:
        line = input().strip()
        if line == "0000":
            break
        x, y, t = line.split()
        t = int(t)

        cx = ord(x[0])
        cy = ord(y[0])

        vp[cx].append((cy, t))
        vp[cy].append((cx, t))

    dist = [INF] * 128
    parent = [-1] * 128

    s = ord(source[0])
    e = ord(target[0])

    pq = [(0, s)]
    dist[s] = 0

    while pq:
        d, node = heapq.heappop(pq)
        if d > dist[node]:
            continue

        for nxt, w in vp[node]:
            nd = d + w
            if nd < dist[nxt]:
                dist[nxt] = nd
                parent[nxt] = node
                heapq.heappush(pq, (nd, nxt))

    path = []
    cur = e
    while cur != -1:
        path.append(chr(cur))
        if cur == s:
            break
        cur = parent[cur]

    path.reverse()
    print(" ".join(path))


if __name__ == "__main__":
    main()
```

``` java
import java.io.*;
import java.util.*;

public class Main {

    static class Edge {
        int to, w;
        Edge(int t, int w){
            this.to = t;
            this.w = w;
        }
    }

    public static void main(String[] args) throws Exception {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;

        int n = Integer.parseInt(br.readLine().trim()); // 不直接使用

        st = new StringTokenizer(br.readLine());
        char source = st.nextToken().charAt(0);
        char target = st.nextToken().charAt(0);

        List<Edge>[] g = new ArrayList[128];
        for(int i=0;i<128;i++) g[i] = new ArrayList<>();

        while(true){
            String line = br.readLine().trim();
            if(line.equals("0000")) break;

            st = new StringTokenizer(line);
            char x = st.nextToken().charAt(0);
            char y = st.nextToken().charAt(0);
            int w = Integer.parseInt(st.nextToken());

            g[x].add(new Edge(y,w));
            g[y].add(new Edge(x,w));
        }

        int INF = Integer.MAX_VALUE;
        int[] dist = new int[128];
        int[] parent = new int[128];
        Arrays.fill(dist, INF);

        PriorityQueue<int[]> pq = new PriorityQueue<>(Comparator.comparingInt(a->a[0]));

        dist[source] = 0;
        pq.add(new int[]{0, source});

        while(!pq.isEmpty()){
            int[] cur = pq.poll();
            int d = cur[0];
            int u = cur[1];

            if(d > dist[u]) continue;

            for(Edge e : g[u]){
                int nd = d + e.w;
                if(nd < dist[e.to]){
                    dist[e.to] = nd;
                    parent[e.to] = u;
                    pq.add(new int[]{nd, e.to});
                }
            }
        }

        List<Character> path = new ArrayList<>();
        int cur = target;

        while(true){
            path.add((char)cur);
            if(cur == source) break;
            cur = parent[cur];
        }

        Collections.reverse(path);

        StringBuilder sb = new StringBuilder();
        for(char c : path) sb.append(c).append(' ');
        System.out.println(sb.toString());
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"container/heap"
	"fmt"
	"os"
)

const INF = int(1e18)

type Edge struct {
	to int
	w  int
}

type Node struct {
	d int
	u int
}

type PQ []Node

func (pq PQ) Len() int { return len(pq) }
func (pq PQ) Less(i, j int) bool { return pq[i].d < pq[j].d }
func (pq PQ) Swap(i, j int) { pq[i], pq[j] = pq[j], pq[i] }

func (pq *PQ) Push(x interface{}) {
	*pq = append(*pq, x.(Node))
}
func (pq *PQ) Pop() interface{} {
	old := *pq
	x := old[len(old)-1]
	*pq = old[:len(old)-1]
	return x
}

func main() {

	in := bufio.NewReader(os.Stdin)

	var n int
	fmt.Fscan(in, &n)

	var sStr, tStr string
	fmt.Fscan(in, &sStr, &tStr)

	s := int(sStr[0])
	t := int(tStr[0])

	g := make([][]Edge, 128)

	for {
		var x string
		fmt.Fscan(in, &x)
		if x == "0000" {
			break
		}
		var y string
		var w int
		fmt.Fscan(in, &y, &w)

		cx := int(x[0])
		cy := int(y[0])

		g[cx] = append(g[cx], Edge{cy, w})
		g[cy] = append(g[cy], Edge{cx, w})
	}

	dist := make([]int, 128)
	parent := make([]int, 128)
	for i := range dist {
		dist[i] = INF
	}

	pq := &PQ{}
	heap.Init(pq)

	dist[s] = 0
	heap.Push(pq, Node{0, s})

	for pq.Len() > 0 {
		cur := heap.Pop(pq).(Node)
		if cur.d > dist[cur.u] {
			continue
		}
		for _, e := range g[cur.u] {
			nd := cur.d + e.w
			if nd < dist[e.to] {
				dist[e.to] = nd
				parent[e.to] = cur.u
				heap.Push(pq, Node{nd, e.to})
			}
		}
	}

	path := []byte{}
	cur := t
	for {
		path = append(path, byte(cur))
		if cur == s {
			break
		}
		cur = parent[cur]
	}

	// reverse
	for i, j := 0, len(path)-1; i < j; i, j = i+1, j-1 {
		path[i], path[j] = path[j], path[i]
	}

	out := bufio.NewWriter(os.Stdout)
	for _, c := range path {
		fmt.Fprintf(out, "%c ", c)
	}
	out.Flush()
}
```
