# 贪心与最长递增子序列（LIS）
[题目链接](https://kamacoder.com/problempage.php?pid=1463)

## 思路分析
本题要求我们将一个包含互不相同元素的数组分配到多个队列中，并且最终能够通过每次从任意队列头部取出一个元素的方式，合并出一个严格递减的序列。由于队列具有先进先出（FIFO）的特性，为了保证最终输出严格递减，必须满足两个核心条件：
1. **队列内的单调性**：放入每一个单独队列中的元素序列，必须是严格递减的。如果某个队列内部出现了递增，那么先进入队列的较小元素将先被取出，使得最终合并出的序列不可能严格递减。
2. **奇偶性隔离**：题目要求，如果队列非空，新加入的元素必须和队列中已有元素具有相同的奇偶性。这就意味着奇数和偶数完全被隔离，不可能出现在同一个队列中。

结合以上条件，问题就可以转化为：对于原数组中的奇数子序列和偶数子序列，分别求它们最少可以被划分成多少个严格递减的子序列。

根据离散数学中的**Dilworth定理**（狄尔沃斯定理），把一个序列划分成最少的严格递减子序列的数量，恰好等于该序列的**最长递增子序列（LIS）的长度**。

因此，算法步骤如下：
1. 将原数组拆分成两个数组：一个只包含奇数，另一个只包含偶数（注意保持它们在原数组中的相对顺序）。
2. 对这两个数组分别求解最长递增子序列（LIS）的长度。
3. 两个 LIS 长度相加，即为至少需要的最少队列数。
4. LIS 的求解可以使用贪心加二分查找的方法，将时间复杂度优化至 O(nlogn)。

## 复杂度分析
1. 时间复杂度：O(nlogn)；
2. 空间复杂度：O(n)。

## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// 求最长递增子序列长度的函数 (O(NlogN))
int getLIS(const vector<int>& arr) {
    vector<int> tails;
    for (int x : arr) {
        auto it = lower_bound(tails.begin(), tails.end(), x);
        if (it == tails.end()) {
            tails.push_back(x);
        } else {
            *it = x;
        }
    }
    return tails.size();
}

int main() {
    // 优化输入输出
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    if (!(cin >> n)) return 0;
    
    vector<int> odds, evens;
    odds.reserve(n);
    evens.reserve(n);
    
    for (int i = 0; i < n; ++i) {
        int val;
        cin >> val;
        if (val % 2 == 0) {
            evens.push_back(val);
        } else {
            odds.push_back(val);
        }
    }
    
    // 总队列数等于奇数LIS长度加上偶数LIS长度
    cout << getLIS(odds) + getLIS(evens) << "\n";
    
    return 0;
}
```

``` python
import sys
from bisect import bisect_left

def get_lis(arr):
    tails = []
    for x in arr:
        idx = bisect_left(tails, x)
        if idx == len(tails):
            tails.append(x)
        else:
            tails[idx] = x
    return len(tails)

def main():
    # 优化输入，一次性读取所有数据
    input_data = sys.stdin.read().split()
    if not input_data:
        return
        
    n = int(input_data[0])
    odds = []
    evens = []
    
    for i in range(1, n + 1):
        val = int(input_data[i])
        if val % 2 == 0:
            evens.append(val)
        else:
            odds.append(val)
            
    # 总队列数等于奇数LIS长度加上偶数LIS长度
    ans = get_lis(odds) + get_lis(evens)
    print(ans)

if __name__ == '__main__':
    main()
```

### Java
``` java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.IOException;
import java.util.StringTokenizer;
import java.util.ArrayList;

public class Main {
    // 求最长递增子序列长度的函数
    public static int getLIS(ArrayList<Integer> arr) {
        ArrayList<Integer> tails = new ArrayList<>();
        for (int x : arr) {
            int left = 0, right = tails.size() - 1;
            while (left <= right) {
                int mid = left + (right - left) / 2;
                if (tails.get(mid) < x) {
                    left = mid + 1;
                } else {
                    right = mid - 1;
                }
            }
            if (left == tails.size()) {
                tails.add(x);
            } else {
                tails.set(left, x);
            }
        }
        return tails.size();
    }

    public static void main(String[] args) throws IOException {
        // 优化输入
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = null;
        
        String line = br.readLine();
        while (line != null && line.trim().isEmpty()) {
            line = br.readLine();
        }
        if (line == null) return;
        
        st = new StringTokenizer(line);
        int n = Integer.parseInt(st.nextToken());
        
        ArrayList<Integer> odds = new ArrayList<>();
        ArrayList<Integer> evens = new ArrayList<>();
        
        int count = 0;
        while (count < n) {
            while (st == null || !st.hasMoreTokens()) {
                line = br.readLine();
                if (line == null) break;
                st = new StringTokenizer(line);
            }
            if (st != null && st.hasMoreTokens()) {
                int val = Integer.parseInt(st.nextToken());
                if (val % 2 == 0) {
                    evens.add(val);
                } else {
                    odds.add(val);
                }
                count++;
            }
        }
        
        System.out.println(getLIS(odds) + getLIS(evens));
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
)

// 求最长递增子序列长度的函数
func getLIS(arr []int) int {
	tails := make([]int, 0)
	for _, x := range arr {
		left, right := 0, len(tails)-1
		for left <= right {
			mid := left + (right-left)/2
			if tails[mid] < x {
				left = mid + 1
			} else {
				right = mid - 1
			}
		}
		if left == len(tails) {
			tails = append(tails, x)
		} else {
			tails[left] = x
		}
	}
	return len(tails)
}

func main() {
    // 优化大容量输入
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)
	buf := make([]byte, 1024*1024)
	scanner.Buffer(buf, 1024*1024*10)

	if !scanner.Scan() {
		return
	}
	n, _ := strconv.Atoi(scanner.Text())

	odds := make([]int, 0, n)
	evens := make([]int, 0, n)

	for i := 0; i < n; i++ {
		scanner.Scan()
		val, _ := strconv.Atoi(scanner.Text())
		if val%2 == 0 {
			evens = append(evens, val)
		} else {
			odds = append(odds, val)
		}
	}

	fmt.Println(getLIS(odds) + getLIS(evens))
}
```
