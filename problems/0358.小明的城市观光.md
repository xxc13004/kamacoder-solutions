# BFS
[题目链接](https://kamacoder.com/problempage.php?pid=1438)
## 思路分析

要求从左上角到右下角的最短的路径，只需要用到广度优先搜索即可。将1看作是路径，0看作是障碍。

如果最后队列为空，跳出循环，则表示没有路径可达，输出-1。

## 复杂度分析
1. 时间复杂度：O(n^2)；
2. 空间复杂度：O(n^2)。
## 代码实现

``` cpp
#include <bits/stdc++.h>
#define endl '\n'
const int N = 500000, MOD = 1000000007;
using namespace std;
using LL = long long;
using PIL = pair<int, LL>;
using PII = pair<int, int>;
using T3I = tuple<int, int, int>;
int t = 0, dir[]{1, 0, -1, 0, 1};
int main() {
    cin.tie(nullptr) -> sync_with_stdio(false);
    int n;
    cin >> n;
    vector<vector<char>> grids(n, vector<char> (n));
    vector<vector<int>> vis(n, vector<int> (n));
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            cin >> grids[i][j];
    queue<PII> q;
    q.emplace(0, 0);
    vis[0][0] = 1;
    int ans = 0;
    while (!q.empty()) {
        int m = q.size();
        ++ans;
        for (int i = 0; i < m; ++i) {
            auto [x, y] = q.front(); q.pop();
            for (int j = 0; j < 4; ++j) {
                int dx = x + dir[j], dy = y + dir[j + 1];
                if (dx < 0 || dx >= n || dy < 0 || dy >= n || vis[dx][dy] || grids[dx][dy] == '0') continue;
                if (dx == n - 1 && dy == n - 1) {  // 找到最短路径，直接结束
                    cout << ans << endl;
                    return 0;
                }
                vis[dx][dy] = 1;
                q.emplace(dx, dy);
            }
        }
    }
    cout << -1 << endl;  // 没找到，输出-1
    return 0;
}
```

### Python
``` python
from collections import deque
import sys

def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    
    n = int(data[0])
    grids = []
    idx = 1
    for i in range(n):
        row = []
        for j in range(n):
            row.append(data[idx])
            idx += 1
        grids.append(row)
    
    vis = [[0] * n for _ in range(n)]
    q = deque()
    q.append((0, 0))
    vis[0][0] = 1
    
    dirs = [(1, 0), (0, -1), (-1, 0), (0, 1)]  # 下、左、上、右
    ans = 0
    
    # 检查起点是否可走
    if grids[0][0] == '0':
        print(-1)
        return
    
    while q:
        ans += 1
        for _ in range(len(q)):
            x, y = q.popleft()
            for dx, dy in dirs:
                nx, ny = x + dx, y + dy
                if nx < 0 or nx >= n or ny < 0 or ny >= n or vis[nx][ny] or grids[nx][ny] == '0':
                    continue
                if nx == n - 1 and ny == n - 1:
                    print(ans)
                    return
                vis[nx][ny] = 1
                q.append((nx, ny))
    
    print(-1)

if __name__ == "__main__":
    main()
```

``` java
import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        
        // 读取第一行：n
        int n = Integer.parseInt(br.readLine().trim());
        
        char[][] grids = new char[n][n];
        
        // 读取接下来的n行，每行包含n个空格分隔的数字
        for (int i = 0; i < n; i++) {
            String[] parts = br.readLine().trim().split("\\s+");
            for (int j = 0; j < n; j++) {
                grids[i][j] = parts[j].charAt(0);
            }
        }
        
        // 检查起点是否可走
        if (grids[0][0] == '0') {
            System.out.println(-1);
            return;
        }
        
        int[][] vis = new int[n][n];
        Queue<int[]> q = new LinkedList<>();
        q.offer(new int[]{0, 0});
        vis[0][0] = 1;
        
        int[] dx = {1, 0, -1, 0}; // 下、左、上、右
        int[] dy = {0, -1, 0, 1};
        int ans = 0;
        
        while (!q.isEmpty()) {
            ans++;
            int size = q.size();
            for (int i = 0; i < size; i++) {
                int[] pos = q.poll();
                int x = pos[0], y = pos[1];
                for (int j = 0; j < 4; j++) {
                    int nx = x + dx[j];
                    int ny = y + dy[j];
                    if (nx < 0 || nx >= n || ny < 0 || ny >= n || vis[nx][ny] == 1 || grids[nx][ny] == '0') {
                        continue;
                    }
                    if (nx == n - 1 && ny == n - 1) {
                        System.out.println(ans);
                        return;
                    }
                    vis[nx][ny] = 1;
                    q.offer(new int[]{nx, ny});
                }
            }
        }
        
        System.out.println(-1);
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
	"strings"
)

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	
	// 读取第一行：n
	scanner.Scan()
	n, _ := strconv.Atoi(scanner.Text())
	
	grids := make([][]byte, n)
	for i := 0; i < n; i++ {
		scanner.Scan()
		parts := strings.Fields(scanner.Text())
		row := make([]byte, n)
		for j := 0; j < n; j++ {
			row[j] = parts[j][0]
		}
		grids[i] = row
	}
	
	// 检查起点是否可走
	if grids[0][0] == '0' {
		fmt.Println(-1)
		return
	}
	
	vis := make([][]bool, n)
	for i := range vis {
		vis[i] = make([]bool, n)
	}
	
	type point struct{ x, y int }
	q := []point{{0, 0}}
	vis[0][0] = true
	
	dx := []int{1, 0, -1, 0} // 下、左、上、右
	dy := []int{0, -1, 0, 1}
	ans := 0
	
	for len(q) > 0 {
		ans++
		size := len(q)
		for i := 0; i < size; i++ {
			p := q[0]
			q = q[1:]
			for j := 0; j < 4; j++ {
				nx, ny := p.x+dx[j], p.y+dy[j]
				if nx < 0 || nx >= n || ny < 0 || ny >= n || vis[nx][ny] || grids[nx][ny] == '0' {
					continue
				}
				if nx == n-1 && ny == n-1 {
					fmt.Println(ans)
					return
				}
				vis[nx][ny] = true
				q = append(q, point{nx, ny})
			}
		}
	}
	
	fmt.Println(-1)
}
```
