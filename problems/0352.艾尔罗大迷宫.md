# BFS
[题目链接](https://kamacoder.com/problempage.php?pid=1432)
## 思路分析

直接进行深搜或者广搜即可。

从0,0开始搜索，将所有能通过的节点标记为1，最后统计vis矩阵中0的数量。

## 复杂度分析
1. 时间复杂度：O(n^2)；
2. 空间复杂度：O(n^2)。

``` cpp
#include<bits/stdc++.h>
#define endl '\n'
const int N = 100005, MOD = 1000000007;
using namespace std;
using LL = long long;
using PIL = pair<int, LL>;
using PII = pair<int, int>;
using T3I = tuple<int, int, int>;
int t = 0, dir[]{1, 0, -1, 0, 1};
int main() {
    cin.tie(nullptr) -> sync_with_stdio(false);
    int n;
    cin >> n;
    vector<vector<char>> grids(n, vector<char> (n));
    vector<vector<int>> vis(n, vector<int> (n));
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j) 
            cin >> grids[i][j];
    queue<PII> q;
    q.emplace(0, 0);
    vis[0][0] = 1;
    while (!q.empty()) {
        auto [x, y] = q.front(); q.pop();
        for (int i = 0; i < 4; ++i) {
            int dx = x + dir[i], dy = y + dir[i + 1];
            // 确保该点合法，未访问过，且可达
            if (dx < 0 || dx >= n || dy < 0 || dy >= n || vis[dx][dy] || grids[dx][dy] == '1') continue;
            vis[dx][dy] = 1;
            q.emplace(dx, dy);
        }
    }
    int ans = 0;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            if (!vis[i][j]) ++ans;
        }
    }
    cout << ans << endl;
    return 0;
}
```

### Python
``` python
import sys
from collections import deque
 
input_data = sys.stdin.read().split()
n = int(input_data[0])
grids = []
idx = 1
 
# 读取网格
for i in range(n):
    rows = []
    for j in range(n):
        row = input_data[idx]
        idx += 1
        rows.append(row)
    grids.append(rows)

# 方向数组
dirs = [(1, 0), (-1, 0), (0, 1), (0, -1)]
 
# 访问标记
vis = [[False] * n for _ in range(n)]
 
# BFS队列
queue = deque()
queue.append((0, 0))
vis[0][0] = True
 
while queue:
    x, y = queue.popleft()
     
    for dx, dy in dirs:
        nx, ny = x + dx, y + dy
         
        # 检查边界、是否访问过、是否可达
        if 0 <= nx < n and 0 <= ny < n and not vis[nx][ny] and grids[nx][ny] == '0':
            vis[nx][ny] = True
            queue.append((nx, ny))
 
# 统计不可达的点
ans = 0
for i in range(n):
    for j in range(n):
        if not vis[i][j]:
            ans += 1
 
print(ans)
```
### Java
``` java
import java.io.*;
import java.util.*;
 
public class Main {
    static final int[] dir = {1, 0, -1, 0, 1};
 
    // 高速输入
    static class FastScanner {
        private final byte[] buffer = new byte[1 << 16];
        private int ptr = 0, len = 0;
        private final InputStream in = System.in;
 
        private int readByte() throws IOException {
            if (ptr >= len) {
                len = in.read(buffer);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }
 
        int nextInt() throws IOException {
            int c, sign = 1, val = 0;
            do {
                c = readByte();
            } while (c <= ' ');
 
            if (c == '-') {
                sign = -1;
                c = readByte();
            }
            while (c > ' ') {
                val = val * 10 + (c - '0');
                c = readByte();
            }
            return val * sign;
        }
 
        char nextChar() throws IOException {
            int c;
            do {
                c = readByte();
            } while (c <= ' ');
            return (char) c;
        }
    }
 
    public static void main(String[] args) throws Exception {
        FastScanner fs = new FastScanner();
        int n = fs.nextInt();
 
        char[][] grid = new char[n][n];
        boolean[][] vis = new boolean[n][n];
 
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                grid[i][j] = fs.nextChar(); // 读 '0' 或 '1'
            }
        }
 
        ArrayDeque<int[]> q = new ArrayDeque<>();
        q.offer(new int[]{0, 0});
        vis[0][0] = true;
 
        while (!q.isEmpty()) {
            int[] cur = q.poll();
            int x = cur[0], y = cur[1];
 
            for (int i = 0; i < 4; i++) {
                int dx = x + dir[i];
                int dy = y + dir[i + 1];
 
                if (dx < 0 || dx >= n || dy < 0 || dy >= n) continue;
                if (vis[dx][dy] || grid[dx][dy] == '1') continue;
 
                vis[dx][dy] = true;
                q.offer(new int[]{dx, dy});
            }
        }
 
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (!vis[i][j]) ans++;
            }
        }
 
        System.out.println(ans);
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
)

var dir = []int{1, 0, -1, 0, 1}

type FastScanner struct {
	r *bufio.Reader
}

func NewFastScanner() *FastScanner {
	return &FastScanner{
		r: bufio.NewReaderSize(os.Stdin, 1<<16),
	}
}

func (fs *FastScanner) nextInt() int {
	sign, val := 1, 0
	c, _ := fs.r.ReadByte()
	for c <= ' ' {
		c, _ = fs.r.ReadByte()
	}
	if c == '-' {
		sign = -1
		c, _ = fs.r.ReadByte()
	}
	for c > ' ' {
		val = val*10 + int(c-'0')
		c, _ = fs.r.ReadByte()
	}
	return val * sign
}

func (fs *FastScanner) nextChar() byte {
	c, _ := fs.r.ReadByte()
	for c <= ' ' {
		c, _ = fs.r.ReadByte()
	}
	return c
}

func main() {
	fs := NewFastScanner()
	n := fs.nextInt()

	grid := make([][]byte, n)
	vis := make([][]bool, n)
	for i := 0; i < n; i++ {
		grid[i] = make([]byte, n)
		vis[i] = make([]bool, n)
		for j := 0; j < n; j++ {
			grid[i][j] = fs.nextChar()
		}
	}

	// 队列（用数组模拟）
	qx := make([]int, n*n)
	qy := make([]int, n*n)
	head, tail := 0, 0

	qx[tail], qy[tail] = 0, 0
	tail++
	vis[0][0] = true

	for head < tail {
		x, y := qx[head], qy[head]
		head++

		for i := 0; i < 4; i++ {
			dx := x + dir[i]
			dy := y + dir[i+1]

			if dx < 0 || dx >= n || dy < 0 || dy >= n {
				continue
			}
			if vis[dx][dy] || grid[dx][dy] == '1' {
				continue
			}

			vis[dx][dy] = true
			qx[tail], qy[tail] = dx, dy
			tail++
		}
	}

	ans := 0
	for i := 0; i < n; i++ {
		for j := 0; j < n; j++ {
			if !vis[i][j] {
				ans++
			}
		}
	}

	fmt.Println(ans)
}
```
