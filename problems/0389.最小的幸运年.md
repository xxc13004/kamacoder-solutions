# 暴力枚举
[题目链接](https://kamacoder.com/problempage.php?pid=1469)
## 思路分析
题目要求找出一个严格大于给定年份且所有位数各不相同的最小年份（幸运年份）。
观察数据范围发现，给定的年份 n 最大为 1000000。对于 1000000，大于它的最小幸运年份是 1023456。两者之间的差值为 23456。
由于差值并不大（最多两万多次校验），我们完全可以采用“暴力枚举”的方法。具体思路如下：
1. 将给出的年份 n 加 1。
2. 检查当前 n 是否为“幸运年份”（即提取它的每一位数字，判断是否有重复数字）。
3. 判断过程可以使用一个整型变量作为二进制状态压缩的掩码。
4. 如果当前 n 是幸运年份，输出并处理下一组数据；如果不是，n 继续加 1，重复步骤 2。

## 复杂度分析
1. 时间复杂度：O(T * D * log10(n))，D 为找到下一个幸运年份所需的最大步数（最多约 2.3*10^4 步）;
2. 空间复杂度：O(1)。

## 代码实现
### Cpp
``` cpp
#include <iostream>

using namespace std;

// 检查数字的每一位是否都不相同
bool isLucky(int n) {
    int mask = 0;
    while (n > 0) {
        int digit = n % 10;
        // 如果该位对应的二进制位已经是1，说明数字重复
        if ((mask & (1 << digit)) != 0) {
            return false;
        }
        // 将该位对应的二进制位标记为1
        mask |= (1 << digit);
        n /= 10;
    }
    return true;
}

int main() {
    // 优化输入输出
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int t;
    if (cin >> t) {
        while (t--) {
            int n;
            cin >> n;
            n++; // 必须严格大于给定的年份
            while (!isLucky(n)) {
                n++;
            }
            cout << n << "\n";
        }
    }
    return 0;
}
```
### Python
``` python
import sys

def is_lucky(n):
    mask = 0
    while n > 0:
        digit = n % 10
        if mask & (1 << digit):
            return False
        mask |= (1 << digit)
        n //= 10
    return True

def main():
    # 优化输入，一次性读取所有内容
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    t = int(input_data[0])
    out = []
    
    for i in range(1, t + 1):
        n = int(input_data[i])
        n += 1
        while not is_lucky(n):
            n += 1
        out.append(str(n))
        
    # 优化输出
    sys.stdout.write('\n'.join(out) + '\n')

if __name__ == '__main__':
    main()
```
### Java
``` java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.IOException;

public class Main {
    
    static boolean isLucky(int n) {
        int mask = 0;
        while (n > 0) {
            int digit = n % 10;
            if ((mask & (1 << digit)) != 0) {
                return false;
            }
            mask |= (1 << digit);
            n /= 10;
        }
        return true;
    }

    public static void main(String[] args) throws IOException {
        // 优化输入
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        String line = br.readLine();
        if (line == null) return;
        
        int t = Integer.parseInt(line.trim());
        StringBuilder sb = new StringBuilder(); // 优化输出
        
        for (int i = 0; i < t; i++) {
            int n = Integer.parseInt(br.readLine().trim());
            n++;
            while (!isLucky(n)) {
                n++;
            }
            sb.append(n).append("\n");
        }
        System.out.print(sb.toString());
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
)

func isLucky(n int) bool {
	mask := 0
	for n > 0 {
		digit := n % 10
		if (mask & (1 << digit)) != 0 {
			return false
		}
		mask |= (1 << digit)
		n /= 10
	}
	return true
}

func main() {
    // 优化输入
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)
	
	if !scanner.Scan() {
		return
	}
	
	t, _ := strconv.Atoi(scanner.Text())
	
	for i := 0; i < t; i++ {
		scanner.Scan()
		n, _ := strconv.Atoi(scanner.Text())
		n++
		for !isLucky(n) {
			n++
		}
		fmt.Println(n)
	}
}
```
