[题目链接](https://kamacoder.com/problempage.php?pid=1370)
## 思路分析
题目意思是要制作两种面包各一个，问最短的时间是多少。也就是说我们要合理的安排机器。根据题目意思我们知道一共就两种情况，要么选同一种面包机，要么不同的两个分别制作A，B两种面包。

我们可以先遍历数组，记录制作面包B最快的的机器所需的时间，以及其索引，然后记录第二快的机器所需的时间。时间分别记为tb_1,tb_2。

然后遍历数组，枚举以每一个机器来制作A面包，在这种情况下，最优的组合一定是选制作B面包最快的机器，但是这有一个特殊情况，那就是最快的A面包和B面包重合了那么时间的计算方式也会变，所以在这种情况下最优解可能是选择时间第二快的B面包机。如果时间最快的B面包机的索引与当前遍历的索引一样，那表示是同一台面包机，也就是说总时间的计算方式为t1 = ta + tb_1；另外，我们还可以选取另外一种组合，那就是将这台机器与时间第二快的B面包机配合制作，那么时间就是t2 = max(ta, tb_2)；

然后比较这两个时间的大小，即ti = min(t1, t2)，最后与最终的ans比较大小ans = min(ans, ti)。

## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(n)。

### Cpp
``` cpp
#include<bits/stdc++.h>
using namespace std;
const int N = 10005, MOD = 1000000007;
using LL = long long;
using PIL = pair<int, LL>;
using PII = pair<int, int>;
int t, ans = MOD;
int main() {
    int n;
    cin >> n;
    int *ba = new int[n];  // 每台机器制作面包a的时间
    int *bb = new int[n];  // 每台机器制作面包b的时间
    for (int i = 0; i < n; ++i) scanf("%d", ba + i);
    for (int i = 0; i < n; ++i) scanf("%d", bb + i);
    // 只有一台机器
    if (n == 1) {
        cout << ba[0] + bb[0] << endl;
        return 0;
    }
    // 分别记录制作b面包时间最快的机器及其索引，以及第二快时间的机器
    int fv = MOD, fi = -1, sv = MOD;
    for (int i = 0; i < n; ++i) {
        if (bb[i] < fv) {
            sv = fv;
            fv = bb[i], fi = i;
        } else if (bb[i] < sv) {
            sv = bb[i];
        }
    }
    for (int i = 0; i < n; ++i) {
        int b = fv;
        if (fi == i) {
            // 说明只能选第二快的机器或者时间相加
            b = min(b + ba[i], sv);
        }
        ans = min(ans, max(ba[i], b));
    }
    cout << ans << endl;
    return 0;
}
```

``` java
import java.io.*;
import java.util.*;

public class Main {

    static final int MOD = 1000000007;
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int n = Integer.parseInt(br.readLine());
        int[] ba = new int[n];  // 每台机器制作面包a的时间
        int[] bb = new int[n];  // 每台机器制作面包b的时间
        
        StringTokenizer st = new StringTokenizer(br.readLine());
        for (int i = 0; i < n; ++i) ba[i] = Integer.parseInt(st.nextToken());
        
        st = new StringTokenizer(br.readLine());
        for (int i = 0; i < n; ++i) bb[i] = Integer.parseInt(st.nextToken());
        
        // 只有一台机器
        if (n == 1) {
            System.out.println(ba[0] + bb[0]);
            return;
        }
        
        // 分别记录制作b面包时间最快的机器和索引，以及第二快时间的机器
        int fv = MOD, fi = -1, sv = MOD;
        for (int i = 0; i < n; ++i) {
            if (bb[i] < fv) {
                sv = fv;
                fv = bb[i];
                fi = i;
            } else if (bb[i] < sv) {
                sv = bb[i];
            }
        }
        
        int ans = MOD;
        for (int i = 0; i < n; ++i) {
            int b = fv;
            if (fi == i) {
                // 时间相加或者选另一台机器
                b = Math.min(b + ba[i], sv);
            }
            ans = Math.min(ans, Math.max(ba[i], b));
        }
        
        System.out.println(ans);
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"math"
	"os"
	"strconv"
	"strings"
)

func main() {
	// 使用更快的输入方式
	reader := bufio.NewReader(os.Stdin)
	
	// 读取n
	line, _ := reader.ReadString('\n')
	n, _ := strconv.Atoi(strings.TrimSpace(line))
	
	ba := make([]int, n)  // 每台机器制作面包a的时间
	bb := make([]int, n)  // 每台机器制作面包b的时间
	
	// 读取ba数组
	line, _ = reader.ReadString('\n')
	parts := strings.Fields(line)
	for i := 0; i < n; i++ {
		ba[i], _ = strconv.Atoi(parts[i])
	}
	
	// 读取bb数组
	line, _ = reader.ReadString('\n')
	parts = strings.Fields(line)
	for i := 0; i < n; i++ {
		bb[i], _ = strconv.Atoi(parts[i])
	}
	
	// 只有一台机器
	if n == 1 {
		fmt.Println(ba[0] + bb[0])
		return
	}
	
	// 分别记录制作b面包时间最快的机器和索引，以及第二快时间的机器
	fv, fi, sv := math.MaxInt32, -1, math.MaxInt32
	for i := 0; i < n; i++ {
		if bb[i] < fv {
			sv = fv
			fv, fi = bb[i], i
		} else if bb[i] < sv {
			sv = bb[i]
		}
	}
	
	ans := math.MaxInt32
	for i := 0; i < n; i++ {
		b := fv
		if fi == i {
            // 时间相加或者选另外一台机器
			b = min(b + ba[i], sv)
		}
		ans = min(ans, max(ba[i], b))
	}
	
	fmt.Println(ans)
}

func min(a, b int) int {
	if a < b {
		return a
	}
	return b
}

func max(a, b int) int {
	if a > b {
		return a
	}
	return b
}
```
