# Dijkstra单源最短路径
[题目链接](https://kamacoder.com/problempage.php?pid=1445)
## 思路分析
本题主要考察图的最短路径算法，可以用典型的 Dijkstra 算法来解决。具体步骤如下：
1. **构建有向图**：提取所有的点和边，建立邻接表存储节点之间的单向边和距离（权重）。
2. **求解最短距离**：从用户所在的位置（源点 `NodeID`）作为起点，运行 Dijkstra 算法求解该点到图中其他所有节点的最短距离 `dist` 数组。
3. **筛选与排序**：
   - 过滤不可达的备选充电桩（即 `dist >= 3000` ）。
   - 将有效充电桩按距离升序排列。如果距离相同，按充电桩的 `NodeID` 升序排列。
4. **格式化输出**：提取前 3 个充电桩，按 `NodeID,Distance` 拼接并在多个结果间使用 `;` 分隔。若备选桩全不可达，输出 `None`。如果数量不够3个，则全部输出。

## 复杂度分析
1. 时间复杂度：O((V + E) log V)。其中 V 为图中点的个数，E 为单向边的数量。主要消耗在 Dijkstra 算法寻找最短路径过程中。最后的筛选及排序仅针对 k 个候选桩，耗时 O(k log k)。总耗时可接受。
2. 空间复杂度：O(V + E)。邻接表需要 O(V + E) 的空间存储网络拓扑结构，距离数组和优先队列占据 O(V) 的空间。

``` cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

struct Edge {
    int to;
    int weight;
};

struct Node {
    int id;
    long long dist;
    bool operator>(const Node& other) const {
        if (dist != other.dist) {
            return dist > other.dist; // 距离小的优先
        }
        return id > other.id; // NodeID小的优先
    }
};

int main() {
    // 优化输入输出流
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    int user_node;
    cin >> user_node;

    int k;
    cin >> k;
    vector<int> candidates(k);
    for (int i = 0; i < k; ++i) {
        cin >> candidates[i];
    }

    // 动态确定最大节点ID，防止越界
    int max_node = n;
    vector<vector<Edge>> graph(max_node + 1);
    
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        if (u > max_node) {
            max_node = u;
            graph.resize(max_node + 1);
        }
        if (v > max_node) {
            max_node = v;
            graph.resize(max_node + 1);
        }
        graph[u].push_back({v, w});
    }

    // Dijkstra 算法求解最短路径
    const long long INF = 1e18;
    vector<long long> dist(max_node + 1, INF);
    priority_queue<Node, vector<Node>, greater<Node>> pq;

    if (user_node <= max_node) {
        dist[user_node] = 0;
        pq.push({user_node, 0});
    }

    while (!pq.empty()) {
        Node current = pq.top();
        pq.pop();

        int u = current.id;
        long long d = current.dist;

        if (d > dist[u]) continue;

        for (const auto& edge : graph[u]) {
            int v = edge.to;
            int w = edge.weight;
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({v, dist[v]});
            }
        }
    }

    // 候选点结果处理
    vector<Node> results;
    for (int c : candidates) {
        if (c <= max_node && dist[c] != INF) {
            // 要求小于3km
            if (dist[c] >= 3000) continue; 
            results.push_back({c, dist[c]});
        }
    }

    sort(results.begin(), results.end(), [](const Node& a, const Node& b) {
        if (a.dist != b.dist) return a.dist < b.dist;
        return a.id < b.id;
    });

    string ans = "";
    int count = 0;
    for (const auto& res : results) {
        if (count >= 3) break;
        if (!ans.empty()) {
            ans += ";";
        }
        ans += to_string(res.id) + "," + to_string(res.dist);
        count++;
    }

    if (count == 0) {
        cout << "None\n";
    } else {
        cout << ans << "\n";
    }

    return 0;
}
```

``` python
import sys
import heapq

def solve():
    # 优化读取大规模输入
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    n = int(input_data[0])
    m = int(input_data[1])
    
    user_node = int(input_data[2])
    k = int(input_data[3])
    
    candidates = []
    idx = 4
    for _ in range(k):
        candidates.append(int(input_data[idx]))
        idx += 1
        
    max_node = n
    edges = []
    for _ in range(m):
        if idx >= len(input_data):
            break
        u = int(input_data[idx])
        v = int(input_data[idx+1])
        w = int(input_data[idx+2])
        edges.append((u, v, w))
        max_node = max(max_node, u, v)
        idx += 3
        
    # 建图
    graph = [[] for _ in range(max_node + 1)]
    for u, v, w in edges:
        graph[u].append((v, w))
        
    INF = float('inf')
    dist = [INF] * (max_node + 1)
    
    # Dijkstra 算法过程
    pq = []
    if user_node <= max_node:
        dist[user_node] = 0
        heapq.heappush(pq, (0, user_node))
        
    while pq:
        d, u = heapq.heappop(pq)
        
        if d > dist[u]:
            continue
            
        for v, w in graph[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                heapq.heappush(pq, (dist[v], v))
                
    # 提取及过滤结果
    results = []
    for c in candidates:
        if c <= max_node and dist[c] != INF:
            if dist[c] >= 3000: continue
            results.append((dist[c], c))
            
    # 按照先距离，后NodeID顺序排序
    results.sort(key=lambda x: (x[0], x[1]))
    
    ans = []
    for d, c in results:
        if len(ans) >= 3:
            break
        ans.append(f"{c},{d}")
        
    if not ans:
        print("None")
    else:
        print(";".join(ans))

if __name__ == '__main__':
    solve()
```

### Go
``` go
package main

import (
	"bufio"
	"container/heap"
	"fmt"
	"os"
	"sort"
	"strconv"
	"strings"
)

type Edge struct {
	to     int
	weight int
}

type Node struct {
	id   int
	dist int64
}

// 定义最小堆支持
type PriorityQueue []*Node

func (pq PriorityQueue) Len() int { return len(pq) }

func (pq PriorityQueue) Less(i, j int) bool {
	if pq[i].dist == pq[j].dist {
		return pq[i].id < pq[j].id
	}
	return pq[i].dist < pq[j].dist
}

func (pq PriorityQueue) Swap(i, j int) { pq[i], pq[j] = pq[j], pq[i] }

func (pq *PriorityQueue) Push(x interface{}) {
	item := x.(*Node)
	*pq = append(*pq, item)
}

func (pq *PriorityQueue) Pop() interface{} {
	old := *pq
	n := len(old)
	item := old[n-1]
	*pq = old[0 : n-1]
	return item
}

func main() {
    // 快速扫描读取数据
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)

	if !scanner.Scan() {
		return
	}
	n, _ := strconv.Atoi(scanner.Text())
	scanner.Scan()
	m, _ := strconv.Atoi(scanner.Text())

	scanner.Scan()
	userNode, _ := strconv.Atoi(scanner.Text())

	scanner.Scan()
	k, _ := strconv.Atoi(scanner.Text())

	candidates := make([]int, k)
	for i := 0; i < k; i++ {
		scanner.Scan()
		candidates[i], _ = strconv.Atoi(scanner.Text())
	}

	maxNode := n
	type InputEdge struct{ u, v, w int }
	edges := make([]InputEdge, m)
	for i := 0; i < m; i++ {
		if !scanner.Scan() {
			break
		}
		u, _ := strconv.Atoi(scanner.Text())
		scanner.Scan()
		v, _ := strconv.Atoi(scanner.Text())
		scanner.Scan()
		w, _ := strconv.Atoi(scanner.Text())
		edges[i] = InputEdge{u, v, w}
		if u > maxNode {
			maxNode = u
		}
		if v > maxNode {
			maxNode = v
		}
	}

	graph := make([][]Edge, maxNode+1)
	for _, e := range edges {
		graph[e.u] = append(graph[e.u], Edge{to: e.v, weight: e.w})
	}

	dist := make([]int64, maxNode+1)
	const INF = int64(1e18)
	for i := range dist {
		dist[i] = INF
	}

	pq := make(PriorityQueue, 0)
	heap.Init(&pq)

	if userNode <= maxNode {
		dist[userNode] = 0
		heap.Push(&pq, &Node{id: userNode, dist: 0})
	}

    // Dijkstra搜索最短路径
	for pq.Len() > 0 {
		current := heap.Pop(&pq).(*Node)
		u := current.id
		d := current.dist

		if d > dist[u] {
			continue
		}

		for _, edge := range graph[u] {
			v := edge.to
			w := int64(edge.weight)
			if dist[u]+w < dist[v] {
				dist[v] = dist[u] + w
				heap.Push(&pq, &Node{id: v, dist: dist[v]})
			}
		}
	}

	var results []Node
	for _, c := range candidates {
		if c <= maxNode && dist[c] != INF {
            if dist[c] >= 3000 {
                continue
            }
			results = append(results, Node{id: c, dist: dist[c]})
		}
	}

	sort.Slice(results, func(i, j int) bool {
		if results[i].dist == results[j].dist {
			return results[i].id < results[j].id
		}
		return results[i].dist < results[j].dist
	})

	var ans []string
	count := 0
	for _, res := range results {
		if count >= 3 {
			break
		}
		ans = append(ans, fmt.Sprintf("%d,%d", res.id, res.dist))
		count++
	}

	if len(ans) == 0 {
		fmt.Println("None")
	} else {
		fmt.Println(strings.Join(ans, ";"))
	}
}
```
