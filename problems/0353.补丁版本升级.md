# DFS
[题目链接](https://kamacoder.com/problempage.php?pid=1433)
## 思路分析

我们大致可以将节点分为两类：

1. 一类是源节点，即前序版本是NA的节点。对于此类节点，我们将其加入到根节点组成的数组中。然后从这些节点开始遍历，记录后续遍历的每个节点的深度。
2. 第二类是中间节点。对于这类节点，我们直接搜索即可，然后就可以记录每个节点的深度。

记录大深度，比对所有节点的深度，将所有与最大深度相等的节点排序后输出。

## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(n)。

### Python
``` python
import sys
from collections import defaultdict, OrderedDict

sys.setrecursionlimit(1000000)

def main():
    data = sys.stdin.read().strip().split()
    if not data:
        return
    
    n = int(data[0])
    idx = 1
    
    roots = []
    paths = defaultdict(list)
    # 使用 OrderedDict 保持插入顺序（按key排序需要手动排序）
    depth = OrderedDict()
    
    for _ in range(n):
        s1 = data[idx]
        s2 = data[idx + 1]
        idx += 2
        
        if s2 == "NA":
            roots.append(s1)
        else:
            paths[s2].append(s1)
    
    max_depth = 1
    
    def dfs(step, node):
        nonlocal max_depth
        depth[node] = step
        max_depth = max(max_depth, step)
        for nxt in sorted(paths.get(node, [])):  # 按字典序遍历
            dfs(step + 1, nxt)
    
    for root in sorted(roots):  # 按字典序遍历根节点
        dfs(1, root)
    
    # 按字典序排序depth
    sorted_depth = OrderedDict(sorted(depth.items()))
    
    result = []
    for node, d in sorted_depth.items():
        if d == max_depth:
            result.append(node)
    
    print(' '.join(result))

if __name__ == "__main__":
    main()
```
### Java
``` java
import java.io.*;
import java.util.*;

public class Main {
    // 使用 TreeMap 自动按key排序
    private static TreeMap<String, Integer> depth = new TreeMap<>();
    private static TreeMap<String, List<String>> paths = new TreeMap<>();
    private static List<String> roots = new ArrayList<>();
    private static int maxDepth = 1;
    
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;
        
        int n = Integer.parseInt(br.readLine().trim());
        
        for (int i = 0; i < n; i++) {
            st = new StringTokenizer(br.readLine());
            String s1 = st.nextToken();
            String s2 = st.nextToken();
            
            if (s2.equals("NA")) {
                roots.add(s1);
            } else {
                paths.computeIfAbsent(s2, k -> new ArrayList<>()).add(s1);
            }
        }
        
        // 对根节点排序
        Collections.sort(roots);
        for (String root : roots) {
            dfs(1, root);
        }
        
        // 对每个节点的子节点排序
        for (Map.Entry<String, List<String>> entry : paths.entrySet()) {
            Collections.sort(entry.getValue());
        }
        
        StringBuilder result = new StringBuilder();
        for (Map.Entry<String, Integer> entry : depth.entrySet()) {
            if (entry.getValue() == maxDepth) {
                result.append(entry.getKey()).append(" ");
            }
        }
        
        if (result.length() > 0) {
            result.setLength(result.length() - 1);
        }
        System.out.println(result);
    }
    
    private static void dfs(int step, String node) {
        depth.put(node, step);
        maxDepth = Math.max(maxDepth, step);
        
        List<String> children = paths.get(node);
        if (children != null) {
            for (String child : children) {
                dfs(step + 1, child);
            }
        }
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
	"sort"
	"strings"
)

var (
	depth   = make(map[string]int)
	paths   = make(map[string][]string)
	roots   []string
	maxDepth = 1
)

func dfs(step int, node string) {
	depth[node] = step
	if step > maxDepth {
		maxDepth = step
	}
	
	// 获取并排序子节点
	children, exists := paths[node]
	if exists {
		// 确保子节点按字典序遍历
		sort.Strings(children)
		for _, child := range children {
			dfs(step+1, child)
		}
	}
}

func main() {
	scanner := bufio.NewScanner(os.Stdin)
	
	// 读取第一行
	scanner.Scan()
	var n int
	fmt.Sscanf(scanner.Text(), "%d", &n)
	
	// 读取剩余行
	for i := 0; i < n; i++ {
		scanner.Scan()
		fields := strings.Fields(scanner.Text())
		s1, s2 := fields[0], fields[1]
		
		if s2 == "NA" {
			roots = append(roots, s1)
		} else {
			paths[s2] = append(paths[s2], s1)
		}
	}
	
	// 对根节点排序
	sort.Strings(roots)
	
	// 对每个节点的子节点排序
	for key := range paths {
		sort.Strings(paths[key])
	}
	
	// 从所有根节点开始DFS
	for _, root := range roots {
		dfs(1, root)
	}
	
	// 收集最深节点并按字典序排序
	var deepNodes []string
	for node, d := range depth {
		if d == maxDepth {
			deepNodes = append(deepNodes, node)
		}
	}
	
	// 按字典序排序结果
	sort.Strings(deepNodes)
	
	// 输出结果
	fmt.Println(strings.Join(deepNodes, " "))
}
```
