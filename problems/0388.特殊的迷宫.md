# Dijkstra / 剪枝优化
[题目链接](https://kamacoder.com/problempage.php?pid=1468)
## 思路分析
这道题的核心难点在于处理“传送移动”。如果在网格中任意两点之间建立传送边，边数会达到 O((NM)^2) = 10^8 级别，必然超时。

但是，只要注意到题目中给定的网格范围 N, M <= 100，我们就能发现一个限制：
在没有任何传送的情况下，仅仅依靠“上下左右”普通移动，从任意一点走到任意一点的最大步数也绝对不会超过 N + M。
也就是说，如果某次传送的代价 a[i][j] ⊕ a[x][y] >= N + M，那么这个传送门是绝对不划算的，因为直接走路过去的花费都不会超过 N + M。

基于这个点，我们做如下剪枝：
1. 记录一个映射表，通过数值能够 O(1) 找到它对应的格子索引。
2. 在普通的 Dijkstra 算法框架中，对于弹出的当前格子 u（上面的数字为 a[u]）：
   - 常规扩展：上下左右走，代价为 1。
   - 传送扩展：枚举合理的传送代价 k，范围从 0 到 N + M。既然我们想花费 k 进行传送，那么目标格子的数字必须是 a[u] ⊕ k。我们直接去映射表中查是否存在这个数字的格子，如果存在，就直接连一条代价为 k 的边。

通过这种剪枝，我们将每个点需要检查的连边数量从 10000 条骤降到至多 200 条，总边数不超过 2 * 10^6，普通的 Dijkstra 即可完成。

## 复杂度分析
1. 时间复杂度：O(NM * log(NM) + NM * (N + M) * log(NM))。网格中有 V = NM 个节点。每个节点扩展 4 个方向和至多 N+M 种合理的传送代价。所以边数 E <= NM * (N + M + 4)。Dijkstra 的时间复杂度为 O(E * log(V))。
2. 空间复杂度：O(NM + max(A))。其中 max(A) <= 200000，我们需要一个长度为 200005 的数组来存储数值到网格节点索引的映射，外加 Dijkstra 自带的距离数组和优先队列。

## 代码实现
### Cpp
``` cpp
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

const long long INF = 1e18;
const int MAX_VAL = 200005;

void solve() {
    int n, m;
    cin >> n >> m;
    int num_cells = n * m;
    vector<int> a(num_cells);
    
    // val_to_grid[x] 表示数字 x 所在的格子编号，-1 表示不存在该数字
    vector<int> val_to_grid(MAX_VAL, -1);
    for (int i = 0; i < num_cells; i++) {
        cin >> a[i];
        val_to_grid[a[i]] = i;
    }
    
    vector<long long> dist(num_cells, INF);
    // 优先队列保存 <距离, 节点编号>
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    
    dist[0] = 0;
    pq.push({0, 0});
    
    int dx[] = {-1, 1, 0, 0};
    int dy[] = {0, 0, -1, 1};
    
    // 终极剪枝：最大的有效传送代价不可能超过曼哈顿极限距离 n + m
    int MAX_COST = n + m; 
    
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        
        if (d > dist[u]) continue;
        
        // 到达右下角终点，直接输出并返回
        if (u == num_cells - 1) {
            cout << d << "\n";
            return;
        }
        
        // 1. 普通移动
        int r = u / m, c = u % m;
        for (int i = 0; i < 4; i++) {
            int nr = r + dx[i], nc = c + dy[i];
            if (nr >= 0 && nr < n && nc >= 0 && nc < m) {
                int v = nr * m + nc;
                if (d + 1 < dist[v]) {
                    dist[v] = d + 1;
                    pq.push({d + 1, v});
                }
            }
        }
        
        // 2. 传送移动（枚举合理的代价 k）
        for (int k = 0; k <= MAX_COST; k++) {
            int target_val = a[u] ^ k; // 反推出想要传送过去的数字
            
            // 如果数字超出了界限，或者网格里根本没这个数字，直接跳过
            if (target_val >= MAX_VAL || val_to_grid[target_val] == -1) continue;
            
            int v = val_to_grid[target_val];
            if (d + k < dist[v]) {
                dist[v] = d + k;
                pq.push({dist[v], v});
            }
        }
    }
}

int main() {
    // 优化 C++ 输入输出流
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int t;
    if (cin >> t) {
        while (t--) solve();
    }
    return 0;
}
```

### Python
``` python
import sys
import heapq

def solve():
    # 一次性读取全部输入以提速
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    t = int(input_data[0])
    idx = 1
    out = []
    MAX_VAL = 200005
    
    for _ in range(t):
        n = int(input_data[idx])
        m = int(input_data[idx+1])
        idx += 2
        
        num_cells = n * m
        a = []
        val_to_grid = [-1] * MAX_VAL
        
        for i in range(num_cells):
            val = int(input_data[idx])
            a.append(val)
            val_to_grid[val] = i
            idx += 1
            
        dist = [float('inf')] * num_cells
        pq = [(0, 0)]
        dist[0] = 0
        
        dx = [-1, 1, 0, 0]
        dy = [0, 0, -1, 1]
        MAX_COST = n + m
        ans = -1
        
        while pq:
            d, u = heapq.heappop(pq)
            if d > dist[u]:
                continue
            
            if u == num_cells - 1:
                ans = d
                break
                
            # 1. 普通移动
            r, c = u // m, u % m
            for i in range(4):
                nr, nc = r + dx[i], c + dy[i]
                if 0 <= nr < n and 0 <= nc < m:
                    v = nr * m + nc
                    if d + 1 < dist[v]:
                        dist[v] = d + 1
                        heapq.heappush(pq, (d + 1, v))
            
            # 2. 传送移动
            for k in range(MAX_COST + 1):
                target_val = a[u] ^ k
                if target_val < MAX_VAL:
                    v = val_to_grid[target_val]
                    if v != -1:
                        if d + k < dist[v]:
                            dist[v] = d + k
                            heapq.heappush(pq, (dist[v], v))
                            
        out.append(str(ans))
        
    print('\n'.join(out))

if __name__ == '__main__':
    solve()
```

``` java
import java.io.InputStream;
import java.io.IOException;
import java.util.PriorityQueue;
import java.util.Arrays;

public class Main {
    static class FastScanner {
        private InputStream stream;
        private byte[] buf = new byte[1024];
        private int head, tail;

        public FastScanner(InputStream stream) { this.stream = stream; }

        int read() {
            if (head >= tail) {
                head = 0;
                try { tail = stream.read(buf, 0, buf.length); } catch (IOException e) { return -1; }
                if (tail <= 0) return -1;
            }
            return buf[head++];
        }

        int nextInt() {
            int c = read();
            while (c <= 32) {
                if (c == -1) return -1;
                c = read();
            }
            int res = 0;
            while (c > 32) {
                res = res * 10 + c - '0';
                c = read();
            }
            return res;
        }
    }

    public static void main(String[] args) {
        FastScanner sc = new FastScanner(System.in);
        int t = sc.nextInt();
        if (t == -1) return;

        int MAX_VAL = 200005;
        StringBuilder sb = new StringBuilder();

        while (t-- > 0) {
            int n = sc.nextInt();
            int m = sc.nextInt();
            int num_cells = n * m;
            
            int[] a = new int[num_cells];
            int[] val_to_grid = new int[MAX_VAL];
            Arrays.fill(val_to_grid, -1);

            for (int i = 0; i < num_cells; i++) {
                a[i] = sc.nextInt();
                val_to_grid[a[i]] = i;
            }

            long[] dist = new long[num_cells];
            Arrays.fill(dist, Long.MAX_VALUE);

            PriorityQueue<Long> pq = new PriorityQueue<>();
            dist[0] = 0;
            // 将 distance 和 u 压缩在一个 long 里，避免包装类的开销
            pq.offer(0L);

            int[] dx = {-1, 1, 0, 0};
            int[] dy = {0, 0, -1, 1};
            int MAX_COST = n + m;
            long ans = -1;

            while (!pq.isEmpty()) {
                long curr = pq.poll();
                long d = curr >>> 32;
                int u = (int) (curr & 0xFFFFFFFFL);

                if (d > dist[u]) continue;

                if (u == num_cells - 1) {
                    ans = d;
                    break;
                }

                // 1. 普通移动
                int r = u / m, c = u % m;
                for (int i = 0; i < 4; i++) {
                    int nr = r + dx[i], nc = c + dy[i];
                    if (nr >= 0 && nr < n && nc >= 0 && nc < m) {
                        int v = nr * m + nc;
                        if (d + 1 < dist[v]) {
                            dist[v] = d + 1;
                            pq.offer(((d + 1) << 32) | v);
                        }
                    }
                }

                // 2. 传送移动
                for (int k = 0; k <= MAX_COST; k++) {
                    int target_val = a[u] ^ k;
                    if (target_val < MAX_VAL && val_to_grid[target_val] != -1) {
                        int v = val_to_grid[target_val];
                        if (d + k < dist[v]) {
                            dist[v] = d + k;
                            pq.offer(((d + k) << 32) | v);
                        }
                    }
                }
            }
            sb.append(ans).append("\n");
        }
        System.out.print(sb.toString());
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"container/heap"
	"fmt"
	"os"
	"strconv"
)

type Item struct {
	d int64
	u int
}

type MinHeap []Item

func (h MinHeap) Len() int           { return len(h) }
func (h MinHeap) Less(i, j int) bool { return h[i].d < h[j].d }
func (h MinHeap) Swap(i, j int)      { h[i], h[j] = h[j], h[i] }
func (h *MinHeap) Push(x interface{}) {
	*h = append(*h, x.(Item))
}
func (h *MinHeap) Pop() interface{} {
	old := *h
	n := len(old)
	x := old[n-1]
	*h = old[0 : n-1]
	return x
}

const INF int64 = 2e18

func solve() {
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Split(bufio.ScanWords)
	out := bufio.NewWriter(os.Stdout)
	defer out.Flush()

	if !scanner.Scan() {
		return
	}
	t, _ := strconv.Atoi(scanner.Text())
	MAX_VAL := 200005

	for iter := 0; iter < t; iter++ {
		scanner.Scan()
		n, _ := strconv.Atoi(scanner.Text())
		scanner.Scan()
		m, _ := strconv.Atoi(scanner.Text())

		num_cells := n * m
		a := make([]int, num_cells)
		val_to_grid := make([]int, MAX_VAL)
		for i := 0; i < MAX_VAL; i++ {
			val_to_grid[i] = -1
		}

		for i := 0; i < num_cells; i++ {
			scanner.Scan()
			a[i], _ = strconv.Atoi(scanner.Text())
			val_to_grid[a[i]] = i
		}

		dist := make([]int64, num_cells)
		for i := 0; i < num_cells; i++ {
			dist[i] = INF
		}

		pq := &MinHeap{}
		heap.Init(pq)

		dist[0] = 0
		heap.Push(pq, Item{d: 0, u: 0})

		dx := []int{-1, 1, 0, 0}
		dy := []int{0, 0, -1, 1}
		MAX_COST := n + m

		ans := int64(-1)

		for pq.Len() > 0 {
			curr := heap.Pop(pq).(Item)
			d := curr.d
			u := curr.u

			if d > dist[u] {
				continue
			}

			if u == num_cells-1 {
				ans = d
				break
			}
            
            // 1. 普通移动
			r, c := u/m, u%m
			for i := 0; i < 4; i++ {
				nr, nc := r+dx[i], c+dy[i]
				if nr >= 0 && nr < n && nc >= 0 && nc < m {
					v := nr*m + nc
					if d+1 < dist[v] {
						dist[v] = d + 1
						heap.Push(pq, Item{d: d + 1, u: v})
					}
				}
			}

            // 2. 传送移动
			for k := 0; k <= MAX_COST; k++ {
				target_val := a[u] ^ k
				if target_val < MAX_VAL {
					v := val_to_grid[target_val]
					if v != -1 {
						if d+int64(k) < dist[v] {
							dist[v] = d + int64(k)
							heap.Push(pq, Item{d: dist[v], u: v})
						}
					}
				}
			}
		}
		fmt.Fprintln(out, ans)
	}
}

func main() {
	solve()
}
```
