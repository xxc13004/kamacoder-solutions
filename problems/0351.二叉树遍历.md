## 思路分析

题目中所给的二叉树是按照完全二叉树的格式来的，所以对于一个完全二叉树，第i层有``` 2 ^ (i - 1) ```个节点。

那么我们提前预处理好每一层需要遍历的节点的数量，然后直接从前往后处理数组即可。

这里需要按照输入格式，将其分解为数组。

## 复杂度分析
1. 时间复杂度：O(n)；
2. 空间复杂度：O(n)。

### Python
``` python
import sys

# 计算每层的节点数：第i层有2^i个节点
CNT = [1 << i for i in range(26)]

# 读取整行输入
s = sys.stdin.readline().strip()
tokens = s.split(',')

idx = 0  # 当前层数（0-based）
cnt = 0  # 当前层已经处理的节点数

output_lines = []
current_line = []

for temp in tokens:
    # 如果当前层已满，进入下一层
    if cnt == CNT[idx]:
        if current_line:  # 如果当前行有内容，输出
            output_lines.append(" ".join(current_line))
        idx += 1
        cnt = 0
        current_line = []
    
    if temp == "#":
        cnt += 1
        continue
    
    current_line.append(temp)
    cnt += 1

# 处理最后一层
if current_line:
    output_lines.append(" ".join(current_line))

sys.stdout.write("\n".join(output_lines))
```

### Java
``` java
import java.util.*;
import java.io.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        PrintWriter pw = new PrintWriter(new BufferedWriter(new OutputStreamWriter(System.out)));
        
        // 计算每层的节点数：第i层有2^i个节点
        int[] CNT = new int[26];
        for (int i = 0; i < 25; i++) {
            CNT[i] = 1 << i;  // 2^i
        }
        
        // 读取整行输入
        String s = br.readLine();
        
        // 使用逗号分割字符串
        String[] tokens = s.split(",");
        
        int idx = 0;  // 当前层数（0-based）
        int cnt = 0;  // 当前层已经处理的节点数
        
        for (String temp : tokens) {
            // 如果当前层已满，进入下一层
            if (cnt == CNT[idx]) {
                idx++;
                cnt = 0;
                pw.println();  // 换行
            }
            
            if (temp.trim().equals("#")) {
                cnt++;
                continue;
            }
            
            pw.print(temp.trim() + " ");
            cnt++;
        }
        
        pw.flush();
    }
}
```

### Go
``` go
package main

import (
	"bytes"
	"io/ioutil"
	"os"
)

func main() {
	// 一次性读取所有输入
	data, _ := ioutil.ReadAll(os.Stdin)
	
	// 计算每层的节点数
	CNT := [26]int{}
	for i := 0; i < 26; i++ {
		CNT[i] = 1 << uint(i)
	}
	
	// 手动按逗号分割字符串
	tokens := splitByComma(data)
	
	idx := 0  // 当前层数
	cnt := 0  // 当前层节点计数
	
	var result bytes.Buffer
	result.Grow(len(data) + 100)  // 预分配空间
	
	firstInLine := true
	
	for i := 0; i < len(tokens); i++ {
		token := tokens[i]
		
		// 如果当前层已满，进入下一层
		if cnt == CNT[idx] {
			idx++
			cnt = 0
			if result.Len() > 0 && result.Bytes()[result.Len()-1] == ' ' {
				// 移除末尾空格并添加换行
				result.Truncate(result.Len() - 1)
			}
			result.WriteByte('\n')
			firstInLine = true
		}
		
		// 检查是否为"#"
		if len(token) == 1 && token[0] == '#' {
			cnt++
			continue
		}
		
		// 去除token两端的空格
		trimmedToken := trimSpaces(token)
		
		// 输出元素
		if !firstInLine {
			result.WriteByte(' ')
		}
		result.Write(trimmedToken)
		firstInLine = false
		cnt++
	}
	
	// 移除末尾多余的空格
	if result.Len() > 0 && result.Bytes()[result.Len()-1] == ' ' {
		result.Truncate(result.Len() - 1)
	}
	
	os.Stdout.Write(result.Bytes())
}

// 按逗号分割字符串
func splitByComma(data []byte) [][]byte {
	var tokens [][]byte
	start := 0
	
	for i := 0; i < len(data); i++ {
		if data[i] == ',' {
			if start < i {
				tokens = append(tokens, data[start:i])
			}
			start = i + 1
		}
	}
	
	// 添加最后一个token
	if start < len(data) {
		tokens = append(tokens, data[start:])
	}
	
	return tokens
}

// 去除字节切片两端的空格
func trimSpaces(b []byte) []byte {
	// 去除开头的空格
	start := 0
	for start < len(b) && (b[start] == ' ' || b[start] == '\t' || b[start] == '\n' || b[start] == '\r') {
		start++
	}
	
	// 去除结尾的空格
	end := len(b)
	for end > start && (b[end-1] == ' ' || b[end-1] == '\t' || b[end-1] == '\n' || b[end-1] == '\r') {
		end--
	}
	
	return b[start:end]
}
```
