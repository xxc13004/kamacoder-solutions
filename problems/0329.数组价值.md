# 贪心
[题目链接](https://kamacoder.com/problempage.php?pid=1409)
## 思路分析
对数组进行排序，要么删除最大的数，要么删除最小的数。结果是``` min(nums[n-1] - nums[1], nums[n-2] - nums[0]) ```。
## 复杂度分析
1. 时间复杂度：O(nlogn)；
2. 空间复杂度：O(1)。

### Python
``` python
import sys

def main() -> None:
    input = sys.stdin.readline
    t = int(input().strip())
    out_lines = []
    
    for _ in range(t):
        n = int(input().strip())
        nums = list(map(int, input().split()))
        nums.sort()
        
        # 计算两种可能的答案
        ans = min(nums[-1] - nums[1], nums[-2] - nums[0])
        out_lines.append(str(ans))
    
    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    main()
```

### Go
``` go
package main

import (
    "bufio"
    "fmt"
    "os"
    "sort"
    "strconv"
    "strings"
)

func main() {
    reader := bufio.NewReader(os.Stdin)
    writer := bufio.NewWriter(os.Stdout)
    defer writer.Flush()
    
    // 读取测试用例数量
    line, _ := reader.ReadString('\n')
    t, _ := strconv.Atoi(strings.TrimSpace(line))
    
    for i := 0; i < t; i++ {
        // 读取数组大小
        line, _ := reader.ReadString('\n')
        n, _ := strconv.Atoi(strings.TrimSpace(line))
        
        // 读取数组元素
        line, _ = reader.ReadString('\n')
        fields := strings.Fields(line)
        
        nums := make([]int, n)
        for j := 0; j < n; j++ {
            nums[j], _ = strconv.Atoi(fields[j])
        }
        
        // 排序
        sort.Ints(nums)
        
        // 计算两种可能的答案
        ans1 := nums[n-1] - nums[1]
        ans2 := nums[n-2] - nums[0]
        if ans1 < ans2 {
            fmt.Fprintln(writer, ans1)
        } else {
            fmt.Fprintln(writer, ans2)
        }
    }
}
```
