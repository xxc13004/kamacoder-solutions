[题目链接](https://kamacoder.com/problempage.php?pid=1455)
## 思路分析

1. 先判断字符串长度是否为3的整数倍，不是直接输出"No"；
2. 将字符串转化为小写；
3. 每隔3位检测一次。看看是否等于"yes"或"sey"，并打好标记，然后输出。

## 复杂度分析
1. 时间复杂度：O(tn)，最大不会超过2e5；
2. 空间复杂度：O(1)。

``` cpp
        // 长度不能被3整除

            cout << "No" << endl;
            continue;
        }
        for (int j = 0; j < n; ++j) {
            if (s[j] >= 'A' && s[j] <= 'Z') s[j] = s[j] + 'a' - 'A';  // 变成小写
        }
        bool flag = true;
        for (int i = 0; i < n; i += 3) {
            string s1 = s.substr(i, 3);
            if (s1 != "yes" && s1 != "sey") {
                cout << "No" << endl;
                flag = false;
                break;
            }
        }
        if (flag) cout << "Yes" << endl;
    }
    return 0;
}
```

### Python
``` python
import sys

def solve() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    t = int(next(it))
    out_lines = []
    for _ in range(t):
        n = int(next(it))
        s = next(it).strip()
        if n % 3 != 0:
            out_lines.append("No")
            continue
        ok = True
        for i in range(0, n, 3):
            # 提取并转换为小写（直接取子串转小写）
            substr = s[i:i+3].lower()
            if substr != "yes" and substr != "sey":
                out_lines.append("No")
                ok = False
                break
        if ok:
            out_lines.append("Yes")
    sys.stdout.write("\n".join(out_lines))

if __name__ == "__main__":
    solve()
```

### Go
``` go
package main

import (
    "bufio"
    "os"
    "strings"
)

func main() {
    scanner := bufio.NewScanner(os.Stdin)
    scanner.Scan()
    t := parseInt(scanner.Text())
    out := make([]string, 0, t)
    for i := 0; i < t; i++ {
        scanner.Scan()
        n := parseInt(scanner.Text())
        scanner.Scan()
        s := scanner.Text()
        if n%3 != 0 {
            out = append(out, "No")
            continue
        }
        ok := true
        for j := 0; j < n; j += 3 {
            sub := strings.ToLower(s[j : j+3])
            if sub != "yes" && sub != "sey" {
                out = append(out, "No")
                ok = false
                break
            }
        }
        if ok {
            out = append(out, "Yes")
        }
    }
    bw := bufio.NewWriter(os.Stdout)
    for _, v := range out {
        bw.WriteString(v)
        bw.WriteByte('\n')
    }
    bw.Flush()
}

func parseInt(s string) int {
    n := 0
    for _, c := range s {
        n = n*10 + int(c-'0')
    }
    return n
}
```
