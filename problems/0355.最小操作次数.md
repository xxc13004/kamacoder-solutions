# 树状数组
[题目链接](https://kamacoder.com/problempage.php?pid=1435)
## 思路分析

题目中的数据是一个矩阵，所以我们可以按照螺旋的方式对这个矩阵进行展开成一维的。然后元素的移动规则就变成了只能往后走（往后面交换），移动的次数是完全等价的。

如此一来，就变成了：

1. 每个元素只能与后一个元素进行交换顺序
2. 要将数组变成递增的

我们思考一下怎么确定要交换的最少次数？其实很简单，那就是求逆序对的数量，因为我们要将小的元素变到前面。他前面所有的比他大的元素都占用了他本来的位置，所以要产生交换。

那么接下来就变成了，求数组中所有的逆序对的数量。可以用树状数组进行求解。

## 复杂度分析
1. 时间复杂度：O(mlogm), m = n * n；
2. 空间复杂度：O(m)。

``` cpp
#include<bits/stdc++.h>
#define endl '\n'
const int N = 100005, MOD = 1000000007;
using namespace std;
using LL = long long;
using PIL = pair<int, LL>;
using PII = pair<int, int>;
using T3I = tuple<int, int, int>;
int t = 0, dir[]{0, 1, 0, -1, 0};
vector<int> spiralOrder(vector<vector<int>>& matrix) {
    int m = matrix.size(), n = matrix[0].size();
    int l = 0, r = n - 1, u = 0, d = m - 1;  // 定义上下左右四条边界
    int dr = 0, x = 0, y = -1;  // 定义方向和当前的坐标
    vector<int> ans(n * m);
    for (int i = 0; i < m * n; ++i) {
        x += dir[dr], y += dir[dr + 1];
        if (y > r) y = r, ++x, ++dr, ++u;
        if (y < l) y = l, --x, ++dr, --d;
        if (x > d) x = d, --y, ++dr, --r;
        if (x < u) x = u, ++y, ++dr, ++l;
        if (dr > 3) dr = 0;
        ans[i] = matrix[x][y];
    }
    return ans;
}
class treeArray {
private:
    vector<int> tree;
    int n;
public:
    treeArray(int n): n(n), tree(n + 1) {}
    void update(int idx, int val = 1) {
        while (idx <= n) {
            tree[idx] += val;
            idx += idx & -idx;
        }
    }
    int query(int idx) {
        int ans = 0;
        while (idx) {
            ans += tree[idx];
            idx -= idx & -idx;
        }
        return ans;
    }
};
int main() {
    cin.tie(nullptr) -> sync_with_stdio(false);
    int n;
    cin >> n;
    vector<vector<int>> matrix(n, vector<int> (n));
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            cin >> matrix[i][j];
    vector<int> nums = spiralOrder(matrix);  // 将这个数组铺平
    // 利用树状数组求解逆序对
    int m = n * n;
    treeArray tree(m);
    LL ans = 0;
    for (int i = 0; i < m; ++i) {
        // 树状数组求的是前面 < x的元素个数，所以用前面的元素总数减掉<的就是大于的，即逆序对的数量
        ans += (i - tree.query(nums[i]));
        if (ans >= MOD) ans -= MOD;
        tree.update(nums[i]);
    }
    cout << ans << endl;
    return 0;
}
```

### Java
``` java
import java.io.*;
import java.util.*;

public class Main {

    static final int MOD = 1000000007;
    static int[] dir = {0, 1, 0, -1, 0};

    // 螺旋展开
    static int[] spiralOrder(int[][] matrix) {
        int m = matrix.length, n = matrix[0].length;
        int l = 0, r = n - 1, u = 0, d = m - 1;
        int dr = 0, x = 0, y = -1;

        int[] ans = new int[m * n];

        for (int i = 0; i < m * n; i++) {
            x += dir[dr];
            y += dir[dr + 1];

            if (y > r) { y = r; x++; dr++; u++; }
            if (y < l) { y = l; x--; dr++; d--; }
            if (x > d) { x = d; y--; dr++; r--; }
            if (x < u) { x = u; y++; dr++; l++; }
            if (dr > 3) dr = 0;

            ans[i] = matrix[x][y];
        }
        return ans;
    }

    // 树状数组
    static class Fenwick {
        int n;
        int[] tree;

        Fenwick(int n) {
            this.n = n;
            tree = new int[n + 1];
        }

        void update(int i) {
            while (i <= n) {
                tree[i]++;
                i += i & -i;
            }
        }

        int query(int i) {
            int s = 0;
            while (i > 0) {
                s += tree[i];
                i -= i & -i;
            }
            return s;
        }
    }

    // 快速输入
    static class FastScanner {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;

        String next() throws IOException {
            while (st == null || !st.hasMoreElements())
                st = new StringTokenizer(br.readLine());
            return st.nextToken();
        }

        int nextInt() throws IOException {
            return Integer.parseInt(next());
        }
    }

    public static void main(String[] args) throws Exception {
        FastScanner fs = new FastScanner();

        int n = fs.nextInt();
        int[][] matrix = new int[n][n];

        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                matrix[i][j] = fs.nextInt();

        int[] nums = spiralOrder(matrix);
        int m = n * n;

        Fenwick bit = new Fenwick(m);
        long ans = 0;

        for (int i = 0; i < m; i++) {
            ans += i - bit.query(nums[i]);
            ans %= MOD;
            bit.update(nums[i]);
        }

        System.out.println(ans);
    }
}
```

### Go
``` go
package main

import (
	"bufio"
	"fmt"
	"os"
)

const MOD = 1000000007

var dir = []int{0, 1, 0, -1, 0}

// 螺旋展开
func spiralOrder(matrix [][]int) []int {
	m, n := len(matrix), len(matrix[0])
	l, r, u, d := 0, n-1, 0, m-1
	dr, x, y := 0, 0, -1

	ans := make([]int, m*n)

	for i := 0; i < m*n; i++ {
		x += dir[dr]
		y += dir[dr+1]

		if y > r {
			y = r
			x++
			dr++
			u++
		}
		if y < l {
			y = l
			x--
			dr++
			d--
		}
		if x > d {
			x = d
			y--
			dr++
			r--
		}
		if x < u {
			x = u
			y++
			dr++
			l++
		}
		if dr > 3 {
			dr = 0
		}
		ans[i] = matrix[x][y]
	}
	return ans
}

// 树状数组
type Fenwick struct {
	n    int
	tree []int
}

func NewFenwick(n int) *Fenwick {
	return &Fenwick{n, make([]int, n+1)}
}

func (f *Fenwick) update(i int) {
	for i <= f.n {
		f.tree[i]++
		i += i & -i
	}
}

func (f *Fenwick) query(i int) int {
	sum := 0
	for i > 0 {
		sum += f.tree[i]
		i -= i & -i
	}
	return sum
}

func main() {
	in := bufio.NewReader(os.Stdin)
	out := bufio.NewWriter(os.Stdout)
	defer out.Flush()

	var n int
	fmt.Fscan(in, &n)

	matrix := make([][]int, n)
	for i := 0; i < n; i++ {
		matrix[i] = make([]int, n)
		for j := 0; j < n; j++ {
			fmt.Fscan(in, &matrix[i][j])
		}
	}

	nums := spiralOrder(matrix)
	m := n * n

	bit := NewFenwick(m)
	ans := 0

	for i := 0; i < m; i++ {
		ans = (ans + i - bit.query(nums[i])) % MOD
		bit.update(nums[i])
	}

	fmt.Fprintln(out, ans)
}
```
